

CREATE   PROCEDURE [dbo].[cw_GetWksSales] @BuySellNo char(12)

AS

BEGIN

--10/16/2015 MCM: Created

--01/09/2016 MCM: Fixed wks no bug



	SET NOCOUNT ON;

	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;



    SELECT SAdjustment,SalesTaxAmt, FXSalesTaxAmt from Buysell where Buysellno = @BuySellNo

END
