CREATE TABLE #BookingContainerList

(

ContainerNo VARCHAR(30)

)

--INVENTORY SALE (CONTAINERS) 

	----------------------------------------------------------------------------------------------------------------

	INSERT #BookingContainerList

	SELECT 	

	Container.ContainerNo

	FROM WksInventory 		

		INNER JOIN Inventory ON WksInventory.LotNo = Inventory.LotNo

		LEFT JOIN WksDetail ON WksInventory.DetailID = WksDetail.DetailID

		LEFT JOIN ContainerContents ON Inventory.LotNo = ContainerContents.LotNo

		LEFT JOIN Container ON Container.ContainerID =  ContainerContents.ContainerID 

		LEFT JOIN Grade ON Inventory.GradeID = Grade.GradeID

		LEFT JOIN BuySell ON WksInventory.BuySellNo = BuySell.BuySellNo

	WHERE 

		Container.BookingNo = @BookingNo 

	AND BuySell.VendorID = '(INV)' AND BuySell.CustID <> '(INV)' 





		--BROKERAGE & INVENTORY SALE (NON CONTAINER)

	----------------------------------------------------------------------------------------------------------------

	INSERT #BookingContainerList

	SELECT 

	BuySell.RefEquipNo as ContainerNo

	FROM BuySell WITH(NOLOCK)

	INNER JOIN WksDetail ON WksDetail.BuysellNo = BuySell.BuySellNo

	LEFT JOIN Grade ON Grade.GradeID = WksDetail.GradeID

	WHERE 

		BuySell.BookingNo = @BookingNo		

		AND CustID <> '(INV)' 

		AND RefEquipNo <> ''

		AND ISNULL(IsContainer,0) = 0 

		GROUP BY RefEquipNo, RefSealNo, BookingNo, BuySell.BuySellNo, Grade.Description, 

		Grade.CommodityDesc,InvoiceDesc,BuySell.ShippingDt,SPrice, SPriceUOM, WksDetail.SCurrencyCd, Specifications, ShowCommDescOnDocs

	HAVING Sum(dbo.CONVERT_TO_UOM(WksDetail.SWeight,WksDetail.SWeightUOM,'M'))>0

	ORDER BY BuySell.ShippingDt, RefEquipNo



	DECLARE @CntrShipped int

	set @CntrShipped =(select count(*) from #BookingContainerList)

-----------------------------------------------------------------------------------------------------------------------------------------------------

DECLARE @LCNo varchar(25)

DECLARE @LCID int

DECLARE @PONo char(15)

DECLARE @ConsigneeNm varchar(100)

DECLARE @ConsigneeAddr varchar(255)

--DECLARE @BeneficiaryNm varchar(100)

--DECLARE @BeneficiaryAddr varchar(255)

--SET @BeneficiaryNm = ''

DECLARE @BeneficiaryCOID char(2)

SET @ConsigneeAddr = ''

SET @LCNo = ''

SET @BeneficiaryCOID = ''



	SELECT @PONo = PONumber	FROM BookSheet WITH(NOLOCK)	WHERE BookSheet.BookingNo = @BookingNo

	IF ISNULL(@PONo,'')<>'' BEGIN

		SELECT @LCID = IssuingLC_ID, @ConsigneeAddr = PO.BLConsignTo	FROM PO WITH(NOLOCK) WHERE PONumber = @PONo

		

		SELECT @LCNo = LCNo, @ConsigneeAddr = ConsigneeAddr,@ConsigneeNm = Counterparty.CompanyNm,

		@BeneficiaryCOID = LC.COID

		--@BeneficiaryAddr = Beneficiary, @BeneficiaryNm = CompanyInfo.CompanyNm

		FROM LC WITH(NOLOCK)

 		LEFT JOIN Counterparty WITH(NOLOCK) ON Counterparty.CpID = LC.ConsigneeCPID

		WHERE LC.ID = @LCID

	END

	



	SELECT BookDate, BookSheet.LoadPort, BookSheet.PortOfDischg, SSLine.CompanyNm, Vessel, VoyageNo, CutOffDate, 

		BookSheet.PONumber,Booksheet.PlaceOfDeliv,

		BookSheet.BookingNo, PO.COID, PO.POReference,PO.Terms,

		dbo.xDoc_GetAddress(PO.CPID,PO.AddrType,0,0) as CustAddr,

		dbo.xDoc_GetAddress(PO.CPID,'',0,0) as CustAddrPrimary,

		dbo.xDoc_GetAddress(BuySell.CustID,BuySell.ShipToID,0,0) as ShipWksCustAddrLocation,

		dbo.xDoc_GetAddress(BuySell.CustID,'',0,0) as ShipWksCustAddr,

		BuySell.Terms as  ShipWksTerms,

		Cust.CompanyNm as CustNm, 

		ETS, ContainerRef as BuysellNo,

		PO.PriceBasis As ShipTerms,

		BookSheet.GradeDesc,

		BookSheet.ContainerQty,BookSheet.ContainerSize,

		@CntrShipped as CntrShipped,

	    BookSheet.ContainerRef as InvoiceNo,

		BookSheet.ShipperRef

		INTO #REPORT

 		FROM BookSheet WITH(NOLOCK) 

 		LEFT JOIN Counterparty SSLine WITH(NOLOCK) ON BookSheet.Carrier = SSLine.CpID

		LEFT JOIN PO WITH(NOLOCK) on Booksheet.PONumber = PO.PONumber And PO.[Source] = 'SO'

 		LEFT JOIN Counterparty Cust WITH(NOLOCK) ON PO.CpID = Cust.CpID

		left join BuySell on BuySell.BuySellNo =  BookSheet.ContainerRef 

 		WHERE BookSheet.BookingNo = @BookingNo



	IF IsNull(@ConsigneeAddr,'')<>''

		UPDATE #REPORT SET CustAddr = @ConsigneeAddr

		

	IF IsNull(@ConsigneeNm,'')<>''

		UPDATE #REPORT SET CustNm = @ConsigneeNm

		

 	IF IsNull(@BeneficiaryCOID,'')<>''

 		UPDATE #REPORT SET COID = @BeneficiaryCOID



IF DB_NAME() = 'cdb_PremierTrading' BEGIN

	DECLARE @GradeNm As VARCHAR(65)

	SELECT @GradeNm = Grade.Description

	FROM BookSheet WITH(NOLOCK) 

		LEFT JOIN PODetail WITH(NOLOCK) on Booksheet.PONumber = PODetail.PONumber 

		LEFT JOIN Grade WITH(NOLOCK) ON Grade.GradeID = PODetail.GradeID

	WHERE BookSheet.BookingNo = @BookingNo



	IF @GradeNm IS NOT NULL

		UPDATE #REPORT SET GradeDesc = REPLACE(REPLACE(REPLACE(GradeDesc, RTRIM(@GradeNm), ''), CHAR(13), ''), CHAR(10), '')

 END	

 	

 	SELECT * FROM #REPORT
