CREATE   PROCEDURE [dbo].[cw_GetGrossProfitBySupplier_test]

		BuySellNo CHAR(12),

    BuySellNo char(12),

    LedgerID INT,

    VendorID char(12) , 

    InvQty decimal(25,2),

    TotalQty decimal(25,2),

    Allocation decimal(25,18),

    Source char(1),

    AdjType char(1),

    RefID char(10)

    CREATE TABLE #REPORT (

    Supplier varchar(50), 

    BuySellNo varchar(15),

	cpid char(12),

    Customer varchar(50),

    TradeDesc varchar(25),

    Invoiced char(1),

    QuantityUOM CHAR(3),

    SupplierCost money,

    Expenses money,

    Cost money,

    GrossProfit money,     

    _Source char(1),

    _AdjType char(1),

    LedgerID INT, 

    OrderNo char(15),

	Dept CHAR(2),

    select 

		gpledger.BUYSELLNO, 

		gpLedger.LedgerID, 

		GPLedger.Source, 

		GPLedger.AdjType,

		(CASE WHEN BuySell.VendorID = '(INV)' THEN 1 ELSE 0 END) IsInventory,

		GPLedger.GrossProfit CkGP		

    FROM GPLedger WITH(NOLOCK) 

		INNER JOIN BUYSELL WITH(NOLOCK)  ON gpledger.buysellno = BUYSELL.BUYSELLNO

    WHERE

		GPLedger.Source='B' AND Buysell.[Status] = 'I' AND VendorID <> '(INV)'  

		AND (GPLedger.TradeDt >= @FromDt And GPLedger.TradeDt <= @ToDt)

		AND (@ForCustCpID = '' OR (@ForCustCpID <> '' AND @ForCustCpID = BuySell.VendorID))

		AND (@TradeType = '' OR @TradeType = Buysell.TradeType)

		gpledger.BUYSELLNO,

		GPLedger.AdjType,

		GPLedger.GrossProfit CkGP		

    FROM GPLedger WITH(NOLOCK) 

		LEFT JOIN BUYSELL WITH(NOLOCK)  ON gpledger.buysellno = BUYSELL.BUYSELLNO

    WHERE

		GPLedger.[Source] <> 'B' AND GPLedger.[Source] <> 'C' AND GPLedger.[Source] <> 'J'	

		AND GPLedger.AdjPosted = 'Y'

		AND (GPLedger.AdjType = 'P' OR GPLedger.AdjType = 'S' OR GPLedger.AdjType = 'B') 

		AND (GPLedger.TradeDt >= @FromDt And GPLedger.TradeDt <= @ToDt)

		AND (@ForCustCpID = '' OR (@ForCustCpID <> '' AND @ForCustCpID = BuySell.VendorID))

		AND (@TradeType = '' OR @TradeType = Buysell.TradeType)







    -- POPULATE INVENTORY ALLOCATION TABLE.

    INSERT #SupplierAlloc(BuySellNo,LedgerID,VendorID,InvQty,Allocation,Source,AdjType,RefID)

    SELECT 

		#BUYSELLLIST.BuySellNo, 

		#BUYSELLLIST.LedgerID, 

		Inventory.VendorID, 

		sum(dbo.calclbs(WksInventory.Weight,WksInventory.UOM)),

		1,

		#BuySellList.Source,

		#BuySellList.AdjType,

		RefID

    FROM #BUYSELLLIST 	

    LEFT JOIN WksInventory WITH(NOLOCK) ON #BUYSELLLIST.BUYSELLNO = WksInventory.BUYSELLNO

    LEFT JOIN Inventory WITH(NOLOCK) ON WksInventory.Lotno = Inventory.Lotno     

	WHERE IsInventory = 1

    Group By #BUYSELLLIST.BuySellNo, #BUYSELLLIST.LedgerID, Inventory.VendorID, #BuySellList.Source, #BuySellList.AdjType,RefID





    -- D.) GET TOTAL QUANTITIES FOR EACH WORKSHEET.

    SELECT BuySellno, LedgerID, Sum(InvQty) AS Total_InvQty

    INTO #TotalInvQty

    FROM #SupplierAlloc

    Group By BuySellNo, LedgerID

    Create index #PK_TotalInvQty ON #TotalInvQty(BuySellNo)

    





    -- E.) CALCULATE ALLOCATION FACTORS. 

    UPDATE #SupplierAlloc 

    SET 

    TotalQty = Total_InvQty,

    Allocation = InvQty/Total_InvQty

    FROM #SupplierAlloc INNER JOIN #TotalInvQty ON #SupplierAlloc.BuySellNo = #TotalInvQty.BuysellNo AND #SupplierAlloc.LedgerID = #TotalInvQty.LedgerID

   WHERE Total_InvQty <> 0















    -- INVENTORY SALES.

    -------------------

 INSERT #REPORT(Supplier,BuySellNo,CpID,Customer,TradeDt,TradeDesc,Invoiced,Quantity,QuantityUOM,Sales,SupplierCost,Expenses,Cost,GrossProfit,_Source,_AdjType,Dept,_CkWtUom, WksType)

    SELECT 

    Supplier.CompanyNm Supplier, 

    #SupplierAlloc.BuySellNo,

	Supplier.CpID,

    Customer.CompanyNm Customer,

    GPLEDGER.TradeDt,

    dbo.SETTRADETYPE(#SupplierAlloc.Source,#SupplierAlloc.AdjType,BuySell.VendorID) AS TradeDesc, 

    CASE Buysell.Status WHEN 'I' THEN 1 ELSE 0 END AS Invoiced,

    (Buysell.TotalWeightLbs*Allocation),	

    'L', 

    round(Sale*Allocation,2) AS Sales,

    round(Cost*Allocation,2) AS SupplierCost,

    round(BuySell.pFreightChg*Allocation,2) AS Expenses,

    round(Cost*Allocation,2) AS Cost,

    round(GrossProfit*Allocation,2) AS GrossProfit,

    'V',    

    #SupplierAlloc.AdjType,

    BuySell.CoID,

    1,

	CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END

    FROM #SupplierAlloc 

    INNER JOIN GPLEDGER WITH(NOLOCK)  ON #SupplierAlloc.LedgerID = GPLedger.LedgerID

    left JOIN BUYSELL WITH(NOLOCK)  ON #SupplierAlloc.BuysellNo = BUYSELL.BuysellNo

    left join Counterparty Customer WITH(NOLOCK) ON Customer.CPID = BuySell.CustID

    left join Counterparty Supplier WITH(NOLOCK) ON Supplier.CPID = #SupplierAlloc.VendorID

    LEFT JOIN Address WITH(NOLOCK) ON BuySell.CustID = Address.CpID AND BuySell.ShipToID = Address.[Type]

    WHERE 

		#SupplierAlloc.Source = 'B'

		AND (BuySell.CoID = @CoID or @CoID = '')		

		AND (@ForCustCpID = '' OR @ForCustCpID = BuySell.VendorID)

		AND (@RefID = '' OR @RefID = BuySell.ReferrerNo1 OR @RefID = BuySell.ReferrerNo2 OR #SupplierAlloc.RefID = @RefID)

		AND (@TradeType = '' OR @TradeType = BuySell.TradeType)

		



    -- INVENTORY SALES ADJUSTMENTS.

    -------------------------------

    INSERT #REPORT(Supplier,BuySellNo,CpID,Customer,TradeDt,TradeDesc,Invoiced,Quantity,QuantityUOM,Sales,SupplierCost,Expenses,Cost,GrossProfit,_Source,_AdjType,Dept, WksType, OrderNo)

    SELECT 

    Supplier.CompanyNm Supplier, 

    #SupplierAlloc.BuySellNo,

	Supplier.CpID,

    Customer.CompanyNm Customer,

    GPLEDGER.TradeDt,

    dbo.SETTRADETYPE(#SupplierAlloc.Source,#SupplierAlloc.AdjType,BuySell.VendorID) AS TradeDesc, --    'Brokerage',   'Inv-Sale',

    CASE Buysell.Status WHEN 'I' THEN 1 ELSE 0 END AS Invoiced,

    AdjWeight*Allocation,	

    AdjWeightUOM, 

    round(Sale*Allocation,2) AS Sales,

    round(Cost*Allocation,2) AS SupplierCost,

    round(BuySell.pFreightChg*Allocation,2) AS Expenses,

    round(Cost*Allocation,2) AS Cost,

    round(GrossProfit*Allocation,2) AS GrossProfit,

    'V',

	CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END,

	RefSONo

    FROM #SupplierAlloc     

    INNER JOIN GPLEDGER WITH(NOLOCK)  ON #SupplierAlloc.LedgerID = GPLedger.LedgerID

    LEFT JOIN BUYSELL WITH(NOLOCK)  ON #SupplierAlloc.BuysellNo = BUYSELL.BuysellNo

    left join Counterparty Customer WITH(NOLOCK) ON Customer.CPID = BuySell.CustID

    left join Counterparty Supplier WITH(NOLOCK) ON Supplier.CPID = #SupplierAlloc.VendorID

    LEFT JOIN Address WITH(NOLOCK) ON BuySell.CustID = Address.CpID AND BuySell.ShipToID = Address.[Type]

    WHERE 

    	#SupplierAlloc.Source <> 'B' 

		AND (BuySell.CoID = @CoID or @CoID = '')		

		AND (@ForCustCpID = '' OR @ForCustCpID = BuySell.VendorID)

		AND (@RefID = '' OR @RefID = BuySell.ReferrerNo1 OR @RefID = BuySell.ReferrerNo2 OR #SupplierAlloc.RefID = @RefID)

		AND (@TradeType = '' OR @TradeType = BuySell.TradeType)

		

    UPDATE #REPORT SET SupplierCost=SupplierCost-Expenses

    Select

    GpLedger.BuySellNo, 

	VENDOR.CpID,

	CUST.CompanyNm, 

    TradeDt, 

    dbo.SETTRADETYPE(GPLedger.Source,GPLedger.AdjType,BuySell.VendorID) AS TradeDesc, --    'Brokerage', 

    CASE Buysell.Status WHEN 'I' THEN 1 ELSE 0 END AS Invoiced,

    BuySell.TotalWeightLbs,

    'L',

    Sale,

	Cost,

    0, 

    Cost, 

    GrossProfit,

    GPLedger.[Source],     

    GPLedger.AdjType,    

    GPLedger.LedgerID,

	BuySell.PfreightChg,

	BuySell.CoID,

	1,

	CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END,

	RefSONo

	FROM #BUYSELLLIST INNER JOIN GPLedger WITH(NOLOCK)  ON #BUYSELLLIST.LedgerID = GPLedger.LedgerID

    LEFT JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno

    LEFT join Counterparty CUST WITH(NOLOCK) ON buysell.custid = CUST.cpid

    LEFT join Counterparty VENDOR WITH(NOLOCK) ON buysell.vendorid = VENDOR.cpid

    LEFT JOIN Address WITH(NOLOCK) ON BuySell.CustID = Address.CpID AND BuySell.ShipToID = Address.[Type]

    WHERE 		

		GPLedger.[Source] = 'B' AND BuySell.[Status] = 'I'

		AND BuySell.VendorID <> '(INV)' AND buysell.VendorID <> '(OTHER)'

		AND (@CoID = '' OR BuySell.CoID = @CoID)

    Select

    CASE Buysell.Status WHEN 'I' THEN 1 ELSE 0 END AS Invoiced,

    gpledger.AdjWeight,

	gpledger.AdjWeightUOM,    

    Sale,

	Cost,

    GPLedger.[Source],

	BuySell.PfreightChg,

	BuySell.CoID,

	CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END,

	FROM #BUYSELLLIST

    left JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno

    WHERE

	-- ASSIGN CORRECT WEIGHT UOM ON REPORT RECORDS.

	INSERT #SetWeightUom

	SELECT 

		Buysellno, 

		(SELECT top 1 PUOM FROm WKsdetail where WKsdetail.BuySellno = #REPORT.BuySellNo ) WTUOM

	from  #REPORT WHERE _CkWtUom = 1

	

	UPDATE #REPORT SET QuantityUOM = #SetWeightUom.QtyUOM 

		FROM #REPORT INNER JOIN #SetWeightUom ON #REPORT.BuySellNo = #SetWeightUom.BuYSellNo

		WHERE (QtyUOM = 'X' OR QtyUOM = 'E' OR QtyUOM = 'B') AND _CkWtUom = 1  



	-- RECALC QTY ON FBM, M3, SQF

	INSERT #SetWeight

	SELECT WksDetail.BuySellNo, SUM(SWEIGHT) 

		FROM WksDetail INNER JOIN #REPORT ON WksDetail.BuySellNo = #REPORT.BuySellNo

		WHERE _source = 'B' 

		Group By WksDetail.BuySellNo



	UPDATE #REPORT SET Quantity = Qty

		FROM #REPORT INNER JOIN #SetWeight ON #REPORT.BuySellNo = #SetWeight.BuySellNo

		WHERE (QuantityUom = 'X' OR QuantityUom = 'E' OR QuantityUom = 'B') AND _CkWtUom = 1  



	---- CONVERT TO REPORTING UOM

	UPDATE #REPORT SET Quantity = dbo.Convert_to_uom(Quantity, QuantityUOM, @ReportUOM)

	IF 	@ReportUOM = 'E'  				

		UPDATE #REPORT SET Quantity = 0

	IF 	@ReportUOM = 'X' OR @ReportUOM = 'B' OR @ReportUOM = 'Q'

					cast ([dbo].[FormatQty](SUM(Quantity),QuantityUOM) as float) As [Weight],

					SUM(Sales) as Sales,

					SUM(SupplierCost) as  Purchase,

					SUM(Expenses) as Expenses ,

					SUM(Cost) as Cost,

					SUM(GrossProfit) as GrossProfit, 

					case when SUM(Sales)=0 then 0 else Sum(GrossProfit) / Sum(Sales) * 100 end as Margin, 

					case when SUM(Quantity)=0 then 0 else Sum(GrossProfit) / Sum(Quantity) end as MarginPerUOM, 

					QuantityUOM,

					'' as '-Hidden-WksType'

			FROM #REPORT 

			GROUP BY Customer,QuantityUOM

			ORDER BY Customer--, TradeDt, BuysellNo
