CREATE   PROCEDURE [dbo].[cw_GetGrossProfitBySuplierTrend_TEST] 

	@LastMonth datetime, 

	@COID char(2), 

	@TradeType varchar(25)

AS

SET NOCOUNT ON

SET ANSI_WARNINGS OFF

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

--3/3/2013 DSH Created by request M. Zachocki AFS.

/*Period=Last Period in 12 month trend*/

--03/15/2013 RS: Period fixed, titles removed, totals added

--05/16/2014 DSH Corrected column lengths for CompanyNm

--09/30/2020 DSH Include "OTHER"



CREATE TABLE #REPORTCOLUMNS(

	ROWNUM INT IDENTITY(1,1),

	LABELNM CHAR(12),

	ReportDt DATETIME

)



	DECLARE @ToDate DATETIME

	DECLARE @FromDate DATETIME

	DECLARE @NextDate DATETIME





	--GET REPORT DATE RANGE.

	DECLARE @StartDt datetime, @EndDt datetime

	SET @EndDt = DATEADD(m,1,@LastMonth) -1

	SET @StartDt = DATEADD(m,-11,@LastMonth) 

	SEt @NextDate = @StartDt





	DECLARE @CrntMonth smallint

	SET @CrntMonth = 1



	DECLARE @Period1 DATE,

			@Period2 DATE,

			@Period3 DATE,

			@Period4 DATE,

			@Period5 DATE,

			@Period6 DATE,

			@Period7 DATE,

			@Period8 DATE,

			@Period9 DATE,

			@Period10 DATE,

			@Period11 DATE,

			@Period12 DATE



	CREATE TABLE #GPDETAILS(

		Vendor CHAR(65),

		WksNo CHAR(12),

		Customer VARCHAR(65),

		TradeDt DATETIME,

		PONumber CHAR(15),

		TradeType CHAR(25),

		Weight DECIMAL(18,5),

		Source CHAR(1),

		Sales MONEY,

		Expenses MONEY,

		Purchases MONEY,

		TotalCost MONEY,

		GrossProfit MONEY,

		Margin DECIMAL(22,5),

		MarginperUOM DECIMAL(22,5),

		Hidden VARCHAR(10)

		)



	CREATE INDEX #IE_DETAILS ON #GPDETAILS(TradeDt)





	CREATE TABLE #DETAILS(

		SupplierNm CHAR(60),

		GP MONEY,

		Qty DECIMAL(18,5),

		TradeDt DATETIME

		)





	--GET GROSS PROFIT BY SUPPLIER DATA.

	INSERT #GPDETAILS

	exec cw_GetGrossProfitBySupplierGrade 'SUPPLIER','0',@StartDt, @EndDt, @CoID, @TradeType,@SupplierID=N'',@ReferrerNo=N'',@ReportUOM=N'L',@CustomerID=N'',@WksNo=NULL,@CountryCode=N'',@ProductCategory=N'',@ProductGrade=N''







	--DELETE FROM #GPDETAILS WHERE Vendor = '(OTHER)'

	DELETE FROM #GPDETAILS WHERE Vendor = '(Gross-Profit Adj)'



	-- CREATE REPORT DETAIL DATA.

	INSERT #DETAILS

	SELECT 

		rtrim(Vendor), 

		GrossProfit,

		[Weight], 

		CONVERT(varchar(2),Month(TradeDt)) + '/1/' + CONVERT(varchar(4),Year(TradeDt))

	FROM #GPDETAILS









	--REPORT TABLE

	CREATE TABLE #REPORT(

		SupplierNm varchar(65),

		Month1 Decimal(18,2) NOT NULL DEFAULT 0,

		Month2 Decimal(18,2) NOT NULL DEFAULT 0,

		Month3 Decimal(18,2) NOT NULL DEFAULT 0,

		Month4 Decimal(18,2) NOT NULL DEFAULT 0,

		Month5 Decimal(18,2) NOT NULL DEFAULT 0,

		Month6 Decimal(18,2) NOT NULL DEFAULT 0,

		Month7 Decimal(18,2) NOT NULL DEFAULT 0,

		Month8 Decimal(18,2) NOT NULL DEFAULT 0,

		Month9 Decimal(18,2) NOT NULL DEFAULT 0,

		Month10 Decimal(18,2) NOT NULL DEFAULT 0,

		Month11 Decimal(18,2) NOT NULL DEFAULT 0,

		Month12 Decimal(18,2) NOT NULL DEFAULT 0,

		Total Decimal(18,2) NOT NULL DEFAULT 0,

		MonthAvg Decimal(18,2) NOT NULL DEFAULT 0

		)



	

	SET @Period1 = @StartDt

	SEt @Period2 = DATEADD(Month,1,@StartDt)

	SEt @Period3 = DATEADD(Month,2,@StartDt)

	SEt @Period4 = DATEADD(Month,3,@StartDt)

	SEt @Period5 = DATEADD(Month,4,@StartDt)

	SEt @Period6 = DATEADD(Month,5,@StartDt)

	SEt @Period7 = DATEADD(Month,6,@StartDt)

	SEt @Period8 = DATEADD(Month,7,@StartDt)

	SEt @Period9 = DATEADD(Month,8,@StartDt)

	SEt @Period10 = DATEADD(Month,9,@StartDt)

	SEt @Period11 = DATEADD(Month,10,@StartDt)

	SEt @Period12 = DATEADD(Month,11,@StartDt)



	----FILL TABLE

	INSERT #REPORT(SupplierNm)

		SELECT DISTINCT SupplierNm FROM #DETAILS ORDER BY SupplierNm



	IF NOT EXISTS(SELECT * FROM #REPORT WHERE SupplierNm IS NULL) 

		INSERT #REPORT(SupplierNm) VALUES(NULL)



	

	UPDATE #REPORT SET Month1 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period1),0)

	UPDATE #REPORT SET Month2 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period2),0)

	UPDATE #REPORT SET Month3 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period3),0)

	UPDATE #REPORT SET Month4 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period4),0)

	UPDATE #REPORT SET Month5 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period5),0)

	UPDATE #REPORT SET Month6 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period6),0)

	UPDATE #REPORT SET Month7 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period7),0)

	UPDATE #REPORT SET Month8 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period8),0)

	UPDATE #REPORT SET Month9 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period9),0)

	UPDATE #REPORT SET Month10 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period10),0)

	UPDATE #REPORT SET Month11 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = @Period11),0)

	UPDATE #REPORT SET Month12 = ISNULL((SELECT SUM(GP) FROM #DETAILS WHERE #DETAILS.SupplierNm = #REPORT.SupplierNm AND #DETAILS.TradeDt = DATEADD(Month,11,@StartDt)),0)

	

	UPDATE #REPORT SET Total = ISNULL(Month1 + Month2 + Month3 + Month4 + Month5 + Month6 + Month7 + Month8 + Month9 + Month10 + Month11 + Month12,0)

	UPDATE #REPORT SET MonthAvg = Total / 12

	





	----DISPLAY

	DECLARE @CurMnth SMALLINT

	SET @CurMnth = 1

	SEt @NextDate = @StartDt

	WHILE @CurMnth <= 12 BEGIN

		INSERT	#REPORTCOLUMNS(LABELNM,ReportDt) 

			SELECT UPPER(SUBSTRING(DATENAME(month,@NextDate),1,3))+' '+CONVERT(CHAR(4),DATEPART(yy,@NextDate)), DATEADD(mm, DATEDIFF(mm, 0, @NextDate), 0) 



		SET @NextDate = DATEADD(m,1,@NextDate)	

		SET @CurMnth = @CurMnth + 1

	END



	-- SET REPORT LABELS AND COLUMNS

		

	SELECT * FROM #REPORTCOLUMNS



	--SELECT @Period1, @Period2, @Period3, @Period4, @Period5, @Period6, @Period7, @Period8, @Period9, @Period10, @Period11, @Period12



	SELECT * FROM #REPORT ORDER BY SupplierNm
