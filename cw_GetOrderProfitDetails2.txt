	@PeriodType char(1), 

	@FromDt datetime, @ToDt datetime, 

	@COID char(2), 

	@TradeType char(25), 

	@CustID char(12), 

	@OrderNo char(15), 

	@ReferrerNo char(10), 

	@UOM char(1) = 'L', 

	@Status varchar(10),

	@IncludeUnshipped tinyint = 0,  

	@ReportByWorksheet tinyint = 0, 

	@UDF1ExpAcct VARCHAR(12) = '',

	@UDF2ExpAcct VARCHAR(12) = ''

SET ANSI_WARNINGS OFF

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

IF @FromDt = '' 

   SET @FromDt = '07/01/2023'



IF @ToDt = '' 

   SET @ToDt = GetDATE()





-- CREATE REPORT TABLE.

CREATE TABLE #REPORT (

	OrderNo char(15),

	OrderDate DateTime,

	OrderStatus Char(10),

	Customer varchar(60),

	Dept varchar(50),

	Tonnage Decimal(18,4) DEFAULT 0,

	UnitType char(1),

	Sales Decimal(18,4) DEFAULT 0,

	SalesAdj Decimal(18,4) DEFAULT 0,

	Purchases Decimal(18,4) DEFAULT 0,

	PurchaseAdj Decimal(18,4) DEFAULT 0,

	Expenses  Decimal(18,4) DEFAULT 0,

	FreightExp Decimal(18,4) DEFAULT 0,

	CommExp Decimal(18,4) DEFAULT 0,

	OtherExp Decimal(18,4) DEFAULT 0,

	GrossProfit Decimal(18,4) DEFAULT 0,

	IntInterest Decimal(18,4) DEFAULT 0,

	NetProfit Decimal(18,4) DEFAULT 0, 

	NetProfitST  Decimal(18,4) DEFAULT 0,

	BudgetProfitST   Decimal(18,4) DEFAULT 0,

	Commission Decimal(18,4) DEFAULT 0,

	SalesRep varchar(30),

	UnitTypePpt char(10),

	Discount Decimal(18,5) DEFAULT 0,

	SupplierNm VARCHAR(130),

	POTerms VARCHAR(50),

	BudgetAmt decimal(18,5),

	GlobalRegion char(30),

	SOTerms varchar(50),

	SOAmount Decimal(18,4) DEFAULT 0,

	SOCurr char(3), 

	SOFxRate Decimal(22,12) DEFAULT 1, 

	SchShipDate datetime,

	TradeType char(25),

	SOQty Decimal(18,4) DEFAULT 0,

	UDFExp1 Decimal(18,4) DEFAULT 0,

	UDFExp2 Decimal(18,4) DEFAULT 0

)

CREATE INDEX #X_REPORT ON #REPORT(OrderNo)



CREATE TABLE #INVOICELIST (

	OrderNo char(15), 

	BuySellNo char(12), 

	PODate datetime, 

	CompanyNm varchar(65), 

	[Status] varchar(10), 

	CompanyDesc varchar(50), 

	TotalWeightLbs decimal(18,4), 

	PFreightChg money, 

	SalesRep varchar(30), 

	SupplierCPID CHAR(12), 

	IntInterest MONEY 

)



    -- WORKSHEET (TRADE) REPORT

	IF @PeriodType = 'T'

		INSERT INTO #INVOICELIST

		SELECT Distinct SPO OrderNo, BuySell.BuysellNo, PODate, Counterparty.CompanyNm, PO.Status, 

		CompanyInfo.CompanyDesc, TotalWeightLbs, PFreightChg, --UnitType ,

		ISNULL((SELECT TOP 1 Referrer.[Name] FROM Referrer INNER JOIN POSalesReps ON POSalesReps.ReferrerNo = Referrer.ReferrerNo WHERE POSalesReps.POID = PO.POID),'') as SalesRep,

		BuySell.VendorID,



		--IntInterest,

		--11/20/23 I.R.

		ISNULL(buysell.IntInterest, 0) 	IntInterest	





		FROM WKSDETAIL 

		WITH(NOLOCK)  INNER JOIN BuySell WITH(NOLOCK)  ON Buysell.Buysellno = WksDetail.BuySellNo

		LEFT JOIN PO WITH(NOLOCK) ON WKSDETAIL.SPO = PO.PONUMBER

		LEFT JOIN CompanyInfo ON BuySell.COID = CompanyInfo.COID

		LEFT JOIN Counterparty ON PO.CPID = Counterparty.CpID

		WHERE (@FromDt = '' OR (Buysell.InvoiceDt >=  @FromDt AND Buysell.InvoiceDt <= @ToDt AND BuySell.Status = 'I'))

		AND CUSTID <> '(INV)' AND CUSTID <> '(PLANT)' AND CUSTID <> '{C}' 

		--AND SPO <> ''

		AND (@CustID = '' OR @CustID = BuySell.CustID)

		AND (@COID = '' OR @COID = Buysell.COID)

		AND (@TradeType = '' OR @TradeType = PO.TradeType)

		AND (@OrderNo = '' OR @OrderNo = SPO)

		AND (@Status = '' OR @Status = PO.Status)

		AND (

				@ReferrerNo = '' OR 

				(

					EXISTS(SELECT POID FROM POSalesReps WHERE POSalesReps.POID = PO.POID AND POSalesReps.ReferrerNo = @ReferrerNo ) OR

					(BuySell.ReferrerNo1 = @ReferrerNo OR BuySell.ReferrerNo2 = @ReferrerNo)

				)

			)

		ORDER BY SPO, BUYSELL.BUYSELLNO

    

	-- ORDER REPORT 

	ELSE

		INSERT INTO #INVOICELIST

		SELECT DISTINCT PO.PONumber, BuySell.BuysellNo, PODate, Counterparty.CompanyNm, PO.Status,

		CompanyInfo.CompanyDesc,TotalWeightLbs, PFreightChg, --UnitType ,

		ISNULL((SELECT TOP 1 Referrer.[Name] FROM Referrer INNER JOIN POSalesReps ON POSalesReps.ReferrerNo = Referrer.ReferrerNo WHERE POSalesReps.POID = PO.POID),'') as SalesRep,

		BuySell.VendorID,

		IntInterest		

		FROM PO WITH(NOLOCK) INNER JOIN wksDetail WITH(NOLOCK) ON PO.PONumber = wksDetail.SPo

		INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = wksDetail.BuySellNo

		INNER JOIN CompanyInfo WITH(NOLOCK) ON PO.COID = CompanyInfo.COID

		INNER JOIN Counterparty WITH(NOLOCK) ON PO.CPID = Counterparty.CpID

		WHERE (@FromDt = ''  OR (@FromDt <= PO.PODate AND PO.PODate <=@ToDt))

		AND CUSTID <> '(INV)' AND CUSTID <> '(PLANT)' AND CUSTID <> '{C}'

		AND (@CustID = '' OR @CustID = BuySell.CustID)

		AND (@COID = '' OR @COID = Buysell.COID)

		AND (@TradeType = '' OR @TradeType = PO.TradeType)

		AND (@OrderNo = '' OR @OrderNo = SPO)

		AND (@Status = '' OR @Status = PO.Status)

		AND (

				@ReferrerNo = '' OR 

				(

					EXISTS(SELECT POID FROM POSalesReps WHERE POSalesReps.POID = PO.POID AND POSalesReps.ReferrerNo = @ReferrerNo) OR

					(BuySell.ReferrerNo1 = @ReferrerNo OR BuySell.ReferrerNo2 = @ReferrerNo)

				)

			)

		ORDER BY PO.PONumber, BUYSELL.BUYSELLNO





	-- FIX "ORDER" DATA WHEN REPORTING WORKSHEETS WITH NO ORDERS.

	IF @PeriodType = 'T' AND @ReportByWorksheet = 0 BEGIN

		UPDATE #INVOICELIST SET 

			OrderNo		= '*'+#INVOICELIST.BuySellNo,

			PODate		= InvoiceDt,

			STATUS		= 'CLOSED',

			CompanyNm	= Counterparty.CompanyNm

		FROM #INVOICELIST 

			INNER JOIN BUYSELL ON Buysell.Buysellno = #INVOICELIST.BuySellNo

			INNER JOIN Counterparty ON BuySell.CustID = Counterparty.CpID

		WHERE OrderNo = ''

	END





	IF @PeriodType = 'T' AND @ReportByWorksheet = 1 BEGIN

		UPDATE #INVOICELIST SET 

			OrderNo		= '*'+#INVOICELIST.BuySellNo,

			PODate		= InvoiceDt,

			STATUS		= 'CLOSED',

			CompanyNm	= Counterparty.CompanyNm

		FROM #INVOICELIST 

			INNER JOIN BUYSELL ON Buysell.Buysellno = #INVOICELIST.BuySellNo

			INNER JOIN Counterparty ON BuySell.CustID = Counterparty.CpID

	END





	

	CREATE INDEX #X_INVLIST ON #INVOICELIST(BuySellNo)

		

	-- GET ORDER SHIPPED SALES WEIGHT.

	SELECT wksDetail.BuySellNo, sum(dbo.CnvToBaseQtyUOM(SWeight, SWeightUOM)) AS TotSalesQty 

	INTO #wksTotSalesQty

	FROM wksDetail WITH(NOLOCK) INNER JOIN #INVOICELIST ON wksDetail.BuySellNo = #INVOICELIST.BuySellNo

	GROUP BY wksDetail.BuySellNo

	

	UPDATE #INVOICELIST SET TotalWeightLbs = TotSalesQty FROM #wksTotSalesQty WHERE #wksTotSalesQty.BuySellNo = #INVOICELIST.BuySellNo



    --B. WRITE ORDER LIST TO REPORT.

    INSERT #REPORT (OrderNo,OrderDate,OrderStatus,Customer,Dept,Tonnage,IntInterest)

    select OrderNo, PODate, Status, CompanyNm, CompanyDesc,SUM(TotalWeightLbs),SUM(IntInterest)

    FROM #INVOICELIST

    GROUP BY OrderNo, PODate, Status, CompanyNm, CompanyDesc

    Order by OrderNo

    

    IF @IncludeUnshipped  = 1 BEGIN

		INSERT #REPORT (OrderNo,OrderDate,OrderStatus,Customer,Dept,Tonnage)

		select PO.PONumber, PODate, Status, Counterparty.CompanyNm, CompanyDesc,

		SUM(dbo.CONVERT_TO_UOM(PODetail.Weight,PODetail.WeightUOM,'L'))

		FROM PO 

		INNER JOIN PODetail ON PO.PONumber = PODetail.PONumber

		INNER JOIN Counterparty ON PO.CPID = Counterparty.CpID

		INNER JOIN CompanyInfo WITH(NOLOCK) ON PO.COID = CompanyInfo.COID

		WHERE  (@FromDt = '' OR (PO.PODate >=  @FromDt AND PO.PODate <= @ToDt))

		AND (@CustID = '' OR @CustID = PO.CPID)

		AND (@COID = '' OR @COID = PO.COID)

		AND (@TradeType = '' OR @TradeType = PO.TradeType)

		AND (@OrderNo = '' OR @OrderNo = PO.PONumber)

		AND (@Status = '' OR @Status = PO.Status)

		AND (

				@ReferrerNo = '' OR 

				(

					EXISTS(SELECT POID FROM POSalesReps WHERE POSalesReps.POID = PO.POID AND POSalesReps.ReferrerNo = @ReferrerNo)

				)

			)

		AND NOT EXISTS(SELECT OrderNo FROM #INVOICELIST WHERE #INVOICELIST.OrderNo = PO.PONumber)

		GROUP BY PO.PONumber, PODate, Status, Counterparty.CompanyNm, CompanyDesc

		Order by PO.PONumber

    

    END







    --C. UPDATE SALES, PURCHASE, GROSS PROFIT AND NET PROFIT COLUMNS

    SELECT 

		OrderNo, 

		SUM(SALE) AS XSALE, 

		SUM(COST) AS XCOST, 

		SUM(GROSSPROFIT) AS XGP, 		

		SUM(Commission) as Comm

    INTO #Temp1

    FROM GPLedger WITH(NOLOCK)  inner join #INVOICELIST ON gpledger.buysellno = #INVOICELIST.BuySellno

	WHERE Source <> 'A'

	AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (TradeDt >= @FromDt AND TradeDt <= @ToDt))) OR @PeriodType = 'O')

    group By OrderNo





    UPDATE #REPORT SET 

		Sales = xSale, 

		Purchases = xCOst, 

		GrossProfit = XGP, 

		NetProfit = XGP - #REPORT.IntInterest - Comm,  

		Commission=Comm

		--IntInterest = #TEMP1.IntInterest, 

		--NetProfit = XGP - #TEMP1.IntInterest - Comm,Commission=Comm		

    FROM #TEMP1 INNER JOIN #REPORT ON #Temp1.OrderNo = #REPORT.OrderNo

    

    UPDATE #REPORT SET Tonnage= dbo.CnvFromBaseQtyUOM(Tonnage,@UOM)



	--SELECT * FROM #REPORT

	--RETURN

	

	--D. BREAKOUT FREIGHT & COMMISSIONS EXPENSES

	SELECT OrderNo, CAT, SUM(WksDelivery.Cost) AS XEXPENSE

	INTO #Temp3

	FROM WksDelivery WITH(NOLOCK)  inner join #INVOICELIST ON WKSDelivery.buysellno = #INVOICELIST.BuySellno

		LEFT JOIN glaccount GL ON GL.GLAcct = WksDelivery.GLAcct

	group By OrderNo, Cat

		

	UPDATE #REPORT SET FreightExp = XEXPENSE

		FROM #TEMP3 INNER JOIN #REPORT ON #Temp3.OrderNo = #REPORT.OrderNo AND #Temp3.Cat = 'FREIGHT'

	UPDATE #REPORT SET CommExp = XEXPENSE

		FROM #TEMP3 INNER JOIN #REPORT ON #Temp3.OrderNo = #REPORT.OrderNo AND #Temp3.Cat = 'COMM EXPENSE'





	--E. GET TOTAL EXPENSES

	SELECT OrderNo, SUM(WksDelivery.Cost) AS XEXPENSE

    INTO #Temp2

    FROM WksDelivery WITH(NOLOCK)  inner join #INVOICELIST ON WKSDelivery.buysellno = #INVOICELIST.BuySellno

    group By OrderNo



    UPDATE #REPORT SET 

		Expenses  = XEXPENSE,

		Purchases = Purchases-XEXPENSE,

		OtherExp  = XEXPENSE - (FreightExp + CommExp)		

    FROM #TEMP2 INNER JOIN #REPORT ON #Temp2.OrderNo = #REPORT.OrderNo





	--F. GET UDF Expenses (BRUNIUS)

	IF @UDF1ExpAcct <> '' OR @UDF2ExpAcct <> '' BEGIN



		SELECT OrderNo, WksDelivery.GLAcct, SUM(WksDelivery.Cost) AS XEXPENSE INTO #Temp4

		FROM WksDelivery WITH(NOLOCK)  inner join #INVOICELIST ON WKSDelivery.buysellno = #INVOICELIST.BuySellno

			LEFT JOIN glaccount GL ON GL.GLAcct = WksDelivery.GLAcct

		WHERE WksDelivery.GLAcct=@UDF1ExpAcct OR WksDelivery.GLAcct=@UDF2ExpAcct 

		GROUP BY OrderNo, WksDelivery.GLAcct



		IF @UDF1ExpAcct <> '' 

			UPDATE #REPORT 

			SET UDFExp1 = XEXPENSE

			FROM #TEMP4 INNER JOIN #REPORT ON #Temp4.OrderNo = #REPORT.OrderNo

			WHERE GLAcct = @UDF1ExpAcct



		IF @UDF2ExpAcct <> '' 				

			UPDATE #REPORT 

			SET UDFExp2 = XEXPENSE

			FROM #TEMP4 INNER JOIN #REPORT ON #Temp4.OrderNo = #REPORT.OrderNo

			WHERE GLAcct = @UDF2ExpAcct



	END













-- GET (LOCAL) PURCHASE ADJUSTMENTS



	--1. GET LIST OF SALES WORKSHEET ADJUSTMENTS IN PERIOD OR ORDER WORKSHEET

	Select #INVOICELIST.OrderNo, SUM(GPLedger.Cost) as adj

    INTO #PURCHADJLIST

    FROM #INVOICELIST

	INNER JOIN GPLedger WITH(NOLOCK) ON GPLedger.BuySellNo = #INVOICELIST.BuysellNo

	INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = #INVOICELIST.BuySellNo

    WHERE Source = 'A' AND ADJType = 'P' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt) OR @PeriodType = 'O')

    AND VendorID <> '(INV)' 

	GROUP BY  #INVOICELIST.OrderNo





    --2. UPDATE REPORT TABLE WITH PURCHASE ADJUSTMENTS.    

    UPDATE #report set 

		#report.PurchaseAdj = #report.PurchaseAdj + adj, 

		#report.GrossProfit = #report.GrossProfit - ADJ, 

		#report.NetProfit = #report.NetProfit - ADJ

    FROM #REPORT INNER JOIN #PURCHADJLIST ON #REPORT.OrderNo = #PURCHADJLIST.OrderNo













    

-- GET INVENTORY PURCHASE ADJUSTMENTS --



    --1. GET LIST OF INVENTORY PURCHASE WORKSHEETS WITH ADJUSTMENTS IN PERIOD.

    Select DISTINCT Inventory.BuySellNo, BuySell.TotalWeightLbs TotalWeight

    INTO #PURCHASELIST

    FROM WksInventory WITH(NOLOCK)  

    INNER JOIN #INVOICELIST ON WksInventory.BuySellNo = #INVOICELIST.BuysellNo

    INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno

    INNER JOIN GPLEDGER WITH(NOLOCK)  ON Inventory.BuySellNo = GPLEdger.BuySellNo

    INNER JOIN BuySell WITH (NOLOCK) ON Inventory.BuySellno = BuySell.BuySellno 

    WHERE Source = 'A' AND ADJType = 'P' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt))) OR @PeriodType = 'O')



		

	UNION

	SELECT DISTINCT GPLedger.BuySellNo, BuySell.TotalWeightLbs TotalWeight

	FROM GPLedger WITH(NOLOCK)

	INNER JOIN #INVOICELIST WITH(NOLOCK) ON GPLedger.BuySellNo = #INVOICELIST.BuySellNo

	INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = #INVOICELIST.BuySellNo

	WHERE VendorID = '(INV)' 

	AND Source = 'A' AND ADJType = 'P' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt))) OR @PeriodType = 'O')



	



    --2. CREATE PURCHASE ADJUSTMENT ALLOCATION TABLE.

    Select #INVOICELIST.OrderNo,     		

    		SUM(GPLEDGER.COST * (dbo.CnvToBaseQtyUOM(WksInventory.Weight, WksInventory.UOM)/  #PURCHASELIST.TotalWeight )) as adj

    INTO #PADJ

    FROM WksInventory WITH(NOLOCK)  

    INNER JOIN #INVOICELIST ON WksInventory.BuySellNo = #INVOICELIST.BuysellNo

    INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno

    INNER JOIN GPLEDGER WITH(NOLOCK)  ON Inventory.BuySellNo = GPLEdger.BuySellNo

    INNER JOIN #PURCHASELIST ON GPLEDGER.BUYSELLNO = #PURCHASELIST.BuySellNo

    WHERE Source = 'A' AND ADJType = 'P' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt))) OR @PeriodType = 'O')

    AND #PURCHASELIST.TotalWeight <> 0 

	GROUP BY #INVOICELIST.OrderNo

	UNION

	SELECT #INVOICELIST.OrderNo, SUM(GPLedger.Cost) as adj

	FROM #INVOICELIST	

	INNER JOIN GPLedger WITH(NOLOCK) ON GPLedger.BuySellNo = #INVOICELIST.BuySellNo

	INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = #INVOICELIST.BuySellNo

	WHERE VendorID = '(INV)' 

	AND Source = 'A' AND ADJType = 'P' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt))) OR @PeriodType = 'O')

	GROUP BY #INVOICELIST.OrderNo



    --3. UPDATE REPORT TABLE WITH PURCHASE ADJUSTMENTS.    

    UPDATE #report set #report.PurchaseAdj = #report.PurchaseAdj + adj, #report.GrossProfit = #report.GrossProfit - ADJ, #report.NetProfit = #report.NetProfit - ADJ

    FROM #REPORT INNER JOIN #PADJ ON #REPORT.OrderNo = #PADJ.OrderNo













-- GET SALES ADJUSTMENT

	

	

	--1. GET LIST OF SALES WORKSHEET ADJUSTMENTS IN PERIOD OR ORDER WORKSHEET

	Select #INVOICELIST.OrderNo, SUM(GPLedger.Sale) as adj

    INTO #SALESADJLIST

    FROM #INVOICELIST

	INNER JOIN GPLedger WITH(NOLOCK) ON GPLedger.BuySellNo = #INVOICELIST.BuysellNo

    WHERE Source = 'A' AND ADJType = 'S' AND AdjPosted = 'Y'

    AND ((@PeriodType = 'T' AND (@FromDt = ''  OR (AdjPostDt >= @FromDt AND AdjPostDt <= @ToDt))) OR @PeriodType = 'O')

	GROUP BY  #INVOICELIST.OrderNo



	--2. UPDATE REPORT TABLE WITH PURCHASE ADJUSTMENTS.    

    UPDATE #report set 

		#report.SalesAdj = #report.SalesAdj + #SALESADJLIST.Adj, 

		#report.GrossProfit = #report.GrossProfit + #SALESADJLIST.Adj, 

		#report.NetProfit = #report.NetProfit + #SALESADJLIST.Adj

    FROM #REPORT INNER JOIN #SALESADJLIST ON #REPORT.OrderNo = #SALESADJLIST.OrderNo







	-- GET SALES DISCOUNT

	Select #INVOICELIST.OrderNo, SUM(DiscountAmt*-1) SalesDisc  

		INTO #SALESDISC

		FROM #INVOICELIST

		INNER JOIN Receipt WITH(NOLOCK) ON Receipt.BuySellNo = #INVOICELIST.BuysellNo

		GROUP BY #INVOICELIST.OrderNo



    UPDATE #report SET #REPORT.Discount = SalesDisc

	FROM #REPORT INNER JOIN #SALESDISC ON #REPORT.OrderNo = #SALESDISC.OrderNo





	--(Old code)

	--UPDATE #REPORT SET #REPORT.Discount = ISNULL((SELECT SUM(DiscountAmt) FROM Receipt INNER JOIN WksDetail WITH(NOLOCK) ON WksDetail.BuySellNo = Receipt.BuySellNo

	--	WHERE WksDetail.SPO = #REPORT.OrderNo),0)









	--E. ASSIGN GROSS NET /ST

    UPDATE #REPORT SET NetProfitST = NetProfit/Tonnage WHERE Tonnage <> 0





	--F. GET BUDGETED GP

    SELECT Budget.PoNumber, isnull(sum(Amount),0) AS BudgetAmount

    INTO #BGP

    FROM Budget INNER JOIN #REPORT ON Budget.PONumber = #REPORT.OrderNo

    AND (ItemType = '[CONTRIBUTION]')

    AND EXISTS(SELECT PONumber FROM Budget A WHERE Amount > 0 AND ItemType = '[PURCHASE]' AND A.PONumber = Budget.PONumber)

    GROUP BY Budget.PONUMBER



    --G. GET BUDGETED WEIGHT    

    SELECT PODetail.PONumber, sum(dbo.CnvToBaseQtyUOM(Weight, WeightUOM)) AS BudgetWtLbs, SUM(ISNULL(Amount,0)) as Amount

    INTO #BWT

    FROM PODETAIL WITH(NOLOCK) INNER JOIN #REPORT ON PODetail.PONumber = #REPORT.OrderNo 

    Group By PODetail.PONumber
