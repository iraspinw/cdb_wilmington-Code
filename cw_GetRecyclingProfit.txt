	IF @View = '(All)' OR @View = 'Plant' BEGIN

		INSERT #wrkSalesGradeDetail(SaleType,GradeID,GradeNm,UnitType,ShipBaseWeight,Sales,SalesExpenses) 	

			SELECT 

				'PASSTHRU',			

				WksDetail.GradeID,

				Grade.[Description],

				COALESCE(WksDetail.UnitType,''),

				dbo.convert_to_uom(SWeight,SWeightUOM,@ScaleWeight), 

				WksDetail.SAmount, 

				CASE WHEN totalweightlbs = 0 THEN 0 ELSE ((sfreightchg-pfreightchg)  /  dbo.convert_to_uom(totalweightlbs,'L',@ScaleWeight)) *  dbo.convert_to_uom(SWeight,SWeightUOM,@ScaleWeight) END  --*(a)			

			FROM WksDetail WITH (NOLOCK) 

				INNER JOIN BuySell WITH (NOLOCK) on WksDetail.BuySellNo = BuySell.BuySellNo

				INNER JOIN GRADE WITH (NOLOCK) ON WksDetail.GradeID = Grade.GradeID								

			WHERE 

			BuySell.VendorID = '(INV)'			 

			AND (BuySell.CustID <> '(INV)' AND CustID <> '(OTHER)')

			AND (@IsAllDates=1 OR (@IsAllDates=0 AND ShippingDt >= @StartDt And ShippingDt <= @EndDt) )	

			AND (BuySell.[Status] = 'C' OR BuySell.[Status] = 'I') 

			AND SAmount <> 0 

			AND (@WhseNm = '' OR (@WhseNm <> '' AND Buysell.ShipFromID = @WhseNm)) 	

			AND (@TradeType = '' OR (@TradeType <> '' AND Buysell.TradeType = @TradeType))

			AND (Grade.InventoryType = 'BULK_WT' OR Grade.InventoryType = 'BULK_WTVAL')

	END







	-- BUILD LIST OF *PASSTHROUGH* GRADES SOLD IN PERIOD. (Bought & Sold as Same Grade)

	INSERT #wrkSalesGradeSummary(SaleType,GradeID,GradeNm,UnitType,ShipBaseWeight,Sales,SalesExpenses,ShipmentCnt,AccountID,Location) 

		SELECT 

			'PASSTHRU',			

			GradeID,

			GradeNm,

			'',

			Sum(ShipBaseWeight), 

			Sum(Sales ), 

			Sum(SalesExpenses),  

			count(GradeID),

			'',''

		FROM #wrkSalesGradeDetail

		Group BY GradeID, GradeNm

			FROM WksDetail 

				INNER JOIN BuySell ON WksDetail.BuySellNo = BuySell.BuySellNo			

				INNER JOIN GRADE ON WksDetail.GradeID = Grade.GradeID

				INNER JOIN COUNTERPARTY ON BuySell.VendorID = Counterparty.CPID

				AND (@IsAllDates=1 OR (ShippingDt >= @StartDt And ShippingDt <= @EndDt) )

				AND (@SalesRep = '' OR (@SalesRep <> '' AND (BuySell.Referrerno1 = @SalesRep OR BuySell.ReferrerNo2 = @SalesRep)))

	-- UPDATE SALES.

		UPDATE #PurchaseSummary SET 		

			Sales = round((BaseWeight-SoldAsBaseWeight) * (SalesPerUOM/@ScaleWeighFactor) +  SoldAsSales ,2),

			SalesPerUOM = (((BaseWeight-SoldAsBaseWeight) * (SalesPerUOM/@ScaleWeighFactor) +  SoldAsSales) / BaseWeight) * @ScaleWeighFactor

			WHERE BaseWeight <> 0 AND IsSoldAsGrade=0 	

		UPDATE #PurchaseSummary SET 		

			Profit = Round(Sales - Cost,2),

			ProfitPerUom = Round(  Round(Sales - Cost,2) / (BaseWeight/@ScaleWeighFactor)  ,2)

			WHERE BaseWeight <> 0 AND IsSoldAsGrade=0 

	END







	-- ASSIGN MARGIN	

	UPDATE #PurchaseSummary SET 				

		Margin = ((BaseWeight * (SalesPerUOM/@ScaleWeighFactor)) - Cost)  / Cost

			WHERE ABS(Cost) > .01 





	-- SPECIAL CUSTOM REPORTING OUTPUT (1)

	IF @SpecialReportOption = 'MARGIN' BEGIN

		SELECT	@StartDt,

				GradeNm, GradeID, 

				round(SUM(BaseWeight/@ScaleWeighFactor),3) WeightST,

				SUM(Sales) Sales, 

				SUM(Purchase) Costs,

				SUM(Sales-Purchase) GrossMargin,				

				SUM(Sales-Purchase)/round(SUM(BaseWeight/@ScaleWeighFactor),3) GrossMarginUOM

			FROM #PurchaseSummary

			GROUP BY GradeNm, GradeID

			ORDER BY GradeNm

		RETURN

	END



	-- SPECIAL CUSTOM REPORTING OUTPUT (2)

	IF @SpecialReportOption = 'MARGIN_ACC' BEGIN

		SELECT	@StartDt,		

				GradeNm, GradeID, 

				AccountNm,

				Location,				

				round(SUM(BaseWeight/@ScaleWeighFactor),3) WeightST,

				SUM(Sales) Sales, 

				SUM(Purchase) Costs,

				SUM(Sales-Purchase) GrossMargin,				

				SUM(Sales-Purchase)/round(SUM(BaseWeight/@ScaleWeighFactor),3) GrossMarginUOM

			FROM #PurchaseSummary

			GROUP BY GradeNm, GradeID, AccountNm, Location

			ORDER BY GradeNm, GradeID, AccountNm, Location

		RETURN

	END









	

	































































	--===========================================================================================================================================================

	--NEW BROKERAGE DATA SECTION

	IF @View = '(All)' OR @View = 'Broker' BEGIN





		UPDATE #PurchaseSummary SET IsPlantReceiving = 1



		IF @SalesRep IS NULL SET @SalesRep = '' 

		IF @BySupID IS NULL SET @BySupID = ''

		IF @BrokerDept IS NULL SET @BrokerDept = ''





		

		-- RESULTS BY TICKET/ACCT

		If @GroupBy <> 'ACCT+GRADE' BEGIN





			INSERT #PurchaseSummary(WarehouseNm,AccountID,AccountNm,Location,GradeID,GradeNm,UnitType,

					BaseWeight,Sales,SalesPerUOM,Purchase,PurchasePerUOM,Freight,FreightPerUom,Adj,AdjPerUOM,  

					IntFreight,Processing,Cost,ShipmentCnt,BuySellNo,ReceivingDt,IsSoldAsGrade,SalesRepNm,ProfitItemID,

					PriceModel,Profit,ProfitPerUom) 

			EXEC cw_GetGrossProfitBySupplierGrade 

				@ReportType=N'SUPPLIER+GRADE+WKS',@GroupBy=0,@FromDate=@StartDt,@ToDate=@EndDt,@CoID=@BrokerDept,@TradeType=@BrokerTradeType,@SupplierID=@BySupID,

				@ReferrerNo=@SalesRep,@ReportUOM=N'S',@CustomerID=N'',@WksNo=NULL,@CountryCode=N'',@ProductCategory=N'',@ProductGrade=N''



		END				



		-- RESULTS BY ACCT GROUP



		If @GroupBy = 'ACCT+GRADE' BEGIN

			INSERT #BrokerageDetails(WarehouseNm,AccountID,AccountNm,Location,GradeID,GradeNm,UnitType,

					BaseWeight,Sales,SalesPerUOM,Purchase,PurchasePerUOM,Freight,FreightPerUom,Adj,AdjPerUOM,  

					IntFreight,Processing,Cost,ShipmentCnt,BuySellNo,ReceivingDt,IsSoldAsGrade,SalesRepNm,ProfitItemID,

					PriceModel,Profit,ProfitPerUom) 

			EXEC cw_GetGrossProfitBySupplierGrade 

				@ReportType=N'SUPPLIER+GRADE+WKS',@GroupBy=0,@FromDate=@StartDt,@ToDate=@EndDt,@CoID=@BrokerDept,@TradeType=@BrokerTradeType,@SupplierID=@BySupID,

				@ReferrerNo=@SalesRep,@ReportUOM=N'S',@CustomerID=N'',@WksNo=NULL,@CountryCode=N'',@ProductCategory=N'',@ProductGrade=N''



			INSERT #PurchaseSummary(WarehouseNm,AccountID,AccountNm,Location,GradeID,GradeNm,UnitType,UnitTypeNm,

					BaseWeight,Sales,SalesPerUOM,Purchase,PurchasePerUOM,Freight,FreightPerUom,Adj,AdjPerUOM,  

					IntFreight,Processing,Cost,ShipmentCnt,BuySellNo,ReceivingDt,IsSoldAsGrade,SalesRepNm,ProfitItemID,

					PriceModel,Profit,ProfitPerUom)		

				SELECT WarehouseNm,AccountID,AccountNm,Location,GradeID,UPPER(GradeNm),UnitType,UnitTypeNm,

					SUM(BaseWeight),SUM(Sales),0,SUM(Purchase),0,SUM(Freight),0,SUM(Adj),0,SUM(IntFreight),SUM(Processing),SUM(Cost),SUM(ShipmentCnt),

					NULL,NULL,1,'',0,'Actual',SUM(Profit),0

					FROM #BrokerageDetails		

					GROUP BY WarehouseNm,AccountID,AccountNm,Location,GradeID,GradeNm,UnitType,UnitTypeNm



			

			UPDATE #PurchaseSummary

			SET PurchasePerUOM = (Purchase/BaseWeight) * @ScaleWeighFactor,					

				FreightPerUom = (Freight/BaseWeight) * @ScaleWeighFactor,

				IntFreightPerUom = (IntFreight/BaseWeight) * @ScaleWeighFactor,	

				ProcessingPerUOM = (Processing/BaseWeight) * @ScaleWeighFactor,								

				AdjPerUom = (Adj/BaseWeight)*@ScaleWeighFactor,

				SalesPerUOM = (Sales/BaseWeight)*@ScaleWeighFactor,

				CostPerUom = (Cost/BaseWeight)*@ScaleWeighFactor,				

				ProfitPerUOM = (Profit/BaseWeight) * @ScaleWeighFactor				

			WHERE BaseWeight <> 0 AND PriceModel = 'Actual' 	

		

		END



		--*1

		UPDATE #PurchaseSummary 

			SET Margin = Profit / Sales WHERE Cost <> 0 AND Sales <> 0 AND PriceModel = 'Actual' 	









		-- REMOVE ADJUSTMENTS FROM ANY PLANT WORKSHEETS

		-------------------------------------------------------------------------------------------------------------------			

		DELETE #PurchaseSummary FROM #PurchaseSummary WHERE #PurchaseSummary.AccountID = '(INV)' 

		--DELETE #PurchaseSummary FROM #PurchaseSummary INNER JOIN BuySell ON #PurchaseSummary.BuySellNo = BuySell.BuySellNo 

		--WHERE #PurchaseSummary.AccountID = '(INV)' AND IsPlantReceiving = 0		

		-------------------------------------------------------------------------------------------------------------------



	END



	--===========================================================================================================================================================





















		



	-- FINAL REPORT BY ACCOUNT.		

	---------------------------

	If @GroupBy = 'ACCT' BEGIN  --Acct = 1 





		CREATE TABLE #LoadCountTable(

			AccountID Char(12),		

			Location Varchar(45),

			BuySellNo CHAR(12) 

			)

		CREATE TABLE #LoadCounts(

			AccountID Char(12),		

			Location Varchar(45),

			ShipmentCnt INT

			)

		INSERT #LoadCountTable SELECT DISTINCT AccountID, Location, BuySellNo FROM #PurchaseSummary  WHERE BuySellno IS NOT NULL

		INSERT #LoadCountTable SELECT DISTINCT AccountID, Location, BuySellNo FROM #purchasedetail  WHERE BuySellno IS NOT NULL

		INSERT #LoadCounts SELECT AccountID, Location, COUNT(*) FROM #LoadCountTable GROUP BY AccountID, Location





		INSERT #PurchaseSummary2 (AccountID,AccountNm,Location,ShipmentCnt,BaseWeight,Purchase,Freight,IntFreight,Processing,Adj,Cost,Sales,Profit,Margin)

			SELECT 

				AccountID,

				AccountNm,

				Location,

				SUM(ShipmentCnt),

				Sum(BaseWeight),

				sum(Purchase),

				SUM(Freight),

				SUM(IntFreight),

				SUM(Processing),				

				SUM(Adj),				

				SUM(Cost),				

				SUM(Sales),

				SUM(Profit),0

				FROM #PurchaseSummary

				GROUP BY AccountID,	AccountNm,Location

				Order By AccountNm, Location



		UPDATE #PurchaseSummary2

			SET PurchasePerUOM = (Purchase/BaseWeight) * @ScaleWeighFactor,					

				FreightPerUom = (Freight/BaseWeight) * @ScaleWeighFactor,

				IntFreightPerUom = (IntFreight/BaseWeight) * @ScaleWeighFactor,	

				ProcessingPerUOM = (Processing/BaseWeight) * @ScaleWeighFactor,								

				AdjPerUom = (Adj/BaseWeight)*@ScaleWeighFactor,

				SalesPerUOM = (Sales/BaseWeight)*@ScaleWeighFactor,

				CostPerUom = (Cost/BaseWeight)*@ScaleWeighFactor,				

				ProfitPerUOM = (Profit/BaseWeight) * @ScaleWeighFactor				

			WHERE BaseWeight <> 0  				

	

		UPDATE #PurchaseSummary2 

			SET Margin = Profit / Sales

			WHERE Cost <> 0 AND Sales <> 0 	

			

		-- update load counts

		UPDATE #PurchaseSummary2 

			SET ShipmentCnt = #LoadCounts.ShipmentCnt

			FROM #PurchaseSummary2 INNER JOIN #LoadCounts ON #PurchaseSummary2.AccountID = #LoadCounts.AccountID AND  #PurchaseSummary2.Location = #LoadCounts.Location



					

				

		SELECT 

			#PurchaseSummary2.AccountID,

			AccountNm As [Account],		

			#PurchaseSummary2.Location,

			GradeID,

			--GradeNm As [Grade],

			ShipmentCnt As [Shipment Count], 

			UnitType,

			UnitTypeNm As [Unit Type],				

			BaseWeight [Weight],

			Purchase,

			PurchasePerUOM As [Purchase Per UOM],

			Freight,

			FreightPerUOM As [Freight Per UOM],	

			IntFreight AS [Inter Freight],

			IntFreightPerUOM,					

			Processing,	

			ProcessingPerUOM As [Processing Per UOM],				

			Adj,

			AdjPerUOM As [Adj Per UOM],			

			Cost,

			CostPerUom As [Cost Per UOM],						

			Sales,

			SalesPerUOM As [Sales Per UOM],		

			Profit,

			ProfitPerUOM As [Profit Per UOM],	

			Margin,

			PriceModel As [Price Model],	

			ProfitItemID,

			'' as [Scale Ticket No.],

			'' as [Receiving Date],

			'' [Ticket Type],

			''  as [Sales Rep]

		 FROM #PurchaseSummary2 			 

			Order By #PurchaseSummary2.AccountNm, #PurchaseSummary2.Location

	

		RETURN

	END		





	-- FINAL REPORT BY ACCOUNT GRADE.	

	---------------------------------		

	If @GroupBy = 'ACCT+GRADE' BEGIN

		SELECT 		

			AccountID,

			AccountNm As [Account],

			Location,

			WarehouseNm,

			GradeID,

			GradeNm As [Grade],

			ShipmentCnt As [Shipment Count],

			UnitType,

			UnitTypeNm As [Unit Type],

			BaseWeight [Weight],

			Purchase,

			PurchasePerUOM As [Purchase Per UOM],				

			Freight,

			FreightPerUom As [Freight Per UOM],

			IntFreight AS [Inter Freight],

			IntFreightPerUOM,		

			Processing,

			ProcessingPerUOM As [Processing Per UOM],

			Adj,

			AdjPerUom As [Adj Per UOM],

			Cost,

			CostPerUOM As [Cost Per UOM],

			Sales,

			SalesPerUOM As [Sales Per UOM],

			Profit,

			ProfitPerUOM As [Profit Per UOM],

			Margin,				

			PriceModel As [Price Model],

			ProfitItemID,

			'' as [Scale Ticket No.],

			'' as [Receiving Date],

			'' [Ticket Type],

			'' as [Sales Rep]

			FROM #PurchaseSummary

			Order By AccountNm, Location



	END





	-- FINAL REPORT BY TICKET.	

	-------------------------		

	If @GroupBy = 'TICKET' BEGIN

	

		

		SELECT 		

			BuysellNo  WksNo,

			AccountID,

			AccountNm As [Account],

			Location,			

			WarehouseNm,

			GradeID,

			GradeNm As [Grade],

			ShipmentCnt As [Shipment Count],

			UnitType,

			UnitTypeNm As [Unit Type],

			BaseWeight [Weight],

			Purchase,

			PurchasePerUOM As [Purchase Per UOM],

			FreightPerUom As [Freight Per UOM],

			IntFreight AS [Inter Freight],

			IntFreightPerUOM,		

			Processing,

			ProcessingPerUOM As [Processing Per UOM],

			Adj,

			AdjPerUom As [Adj Per UOM],

			Cost,

			CostPerUOM As [Cost Per UOM],

			Sales,

			SalesPerUOM As [Sales Per UOM],

			Profit,

			ProfitPerUOM As [Profit Per UOM],

			Margin,				

			PriceModel As [Price Model],

			ProfitItemID,									

			ReceivingDt as [Receiving Date],	

			SalesRepNm	as [Sales Rep]	

			FROM #PurchaseSummary 

			Order By AccountNm, Location, ReceivingDt	

			

					



	END
