		@AltScaleWt CHAR(1),

    -- GET GROSS PROFIT TRANSACTIONS 

    --==============================

	

	--FILL WORKSHEETS.	

	INSERT #BuySelllist

	SELECT	gpLedger.LedgerID,gpledger.BUYSELLNO,GrossProfit,[Source],TradeDT,CustID,VendorID, 

			TotalWeightLbs,Sale,Cost,AdjType,AdjReason,GPLedger.GLAcct,RefPONo,0,

			(CASE WHEN BuySell.VendorID <> '(INV)'  THEN 1 ELSE 0 END)  --DoAllocation

			,0,NULL,

			BuySell.PfreightChg,

			BuySell.SfreightChg,

			CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END,

			CASE WHEN Buysell.VendorID = '(INV)' THEN 'LOT' ELSE '' END

		FROM GPLedger WITH(NOLOCK) 

		INNER JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno	

		WHERE

		GPLedger.[Source] = 'B' AND BUYSELL.[Status] = 'I'

		AND (@CoID = '' OR (BuySell.CoID = @CoID)) 

		AND (TradeDt >=@FromDate OR @FromDate = '')

		AND (TradeDt <=@ToDate OR @ToDate = '')

		AND	(@TradeType = '' OR (BUYSELL.TradeType = @TradeType)) 		

		AND (buysell.custid <> '(INV)' AND buysell.custid <> '(PLANT)' AND buysell.custid  <> '{C}' )

		AND (@ReferrerNo = '' OR @ReferrerNo= BuySell.ReferrerNo1 OR @ReferrerNo= BuySell.ReferrerNo2) 

		AND (@WksNo IS NULL OR buysell.buysellno = @WksNo)



		

	--FILL ADJUSTMENTS.	

	INSERT #BuySelllist

	SELECT	gpLedger.LedgerID,gpledger.BUYSELLNO,GrossProfit,[Source],TradeDT,

			CASE WHEN AdjType='P' THEN GPLedger.AdjCPID ELSE BuySell.CustID END AS CustID,		

			BuySell.VendorID,   

			TotalWeightLbs,Sale,Cost,AdjType,AdjReason,GPLedger.GLAcct,RefPONo, 

			CASE WHEN AdjType='P' THEN dbo.CnvToBaseQtyUOM(AdjWeight,AdjWeightUOM) ELSE 0 END AS AdjustQty, 

			1 --DoAllocation

			,0,NULL,0,0,

			CASE WHEN Buysell.custid = '(INV)' THEN 'R' WHEN buysell.vendorid='(INV)' THEN 'S' ELSE 'W' END,

			CASE WHEN Buysell.VendorID = '(INV)' THEN 'LOT' ELSE '' END

		FROM GPLedger WITH(NOLOCK) 

		LEFT JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno	

		WHERE

		GPLedger.[Source] <> 'B' AND GPLedger.[Source] <> 'C' AND GPLedger.[Source] <> 'J'  --(C,J for internal interest)

		AND (GPLedger.TradeDt >=@FromDate AND GPLedger.TradeDt <=@ToDate) 

		AND (GPLEdger.AdjPosted = 'Y' OR GPLedger.[Source] = 'I')

		AND (@CoID = '' OR (BuySell.CoID = @CoID OR GPLedger.MiscCOID = @COID))

		AND (@TradeType = '' OR (GPLEDGER.MiscTradeType = @TradeType OR BUYSELL.TradeType = @TradeType)) 

		AND (@ReferrerNo = '' OR @ReferrerNo= BuySell.ReferrerNo1 OR @ReferrerNo= BuySell.ReferrerNo2) 	

		AND (@WksNo IS NULL OR buysell.buysellno = @WksNo)







	--SET INVENTORY MODEL

	UPDATE #BuySelllist SET InvModel = 'AVG' , VendorID = '(OTHER)', DoGradeAlloc=1 

		WHERE InvModel = 'LOT' AND NOT EXISTS (SELECT * FROM WksInventory WHERE WksInventory.Buysellno = #BuySelllist.BuysellNo)









		

		IF @ReportType = 'SUPPLIER+GRADE' OR @ReportType = 'GRADE' OR @ReportType =  'SUPPLIER+GRADE+WKS' BEGIN





			--ALLOCATE PROFIT BY GRADE FOR NON INVENTORY TRANACTIONS (WHEN REPORTING BY GRADE}.

			-------------------------------------------------------------------------------------------------- 

			CREATE TABLE #BuysellGradeAllocList(	

				TranSource CHAR(1),		

				LedgerID INT,

				BuySellNo CHAR(12),			

				SeqNo INT,

				GradeID CHAR(15),

				ItemSales MONEY,

				ItemCost MONEY,

				ItemExpense MONEY,

				ItemChargeBack MONEY,

				--ItemGrossProfit MONEY,

				TotalGrossProfit MONEY,

				Sale MONEY,

				ChargeBack MONEY,

				Cost MONEY,

				Expense MONEY,				

				GrossProfit MONEY,

				WeightLbs DECIMAL(22,5),			

				Alloc DECIMAL(22,12),

				TotalWTLBS decimal(22,5)

			)







			INSERT #BuysellGradeAllocList

			SELECT 

				#buyselllist.Source,

				#buyselllist.LedgerID, 

				#buyselllist.buysellno,			

				row_number() OVER (Partition by #buyselllist.LedgerID order by #buyselllist.LedgerID), 

				Wksdetail.GradeID, 

				SAmount,				

				PAmount, 

				0,

				0,				

				0,  			

				#buyselllist.Sale,

				#buyselllist.ChargeBack,

				#buyselllist.Cost,

				#buyselllist.Expense,				

				#buyselllist.GrossProfit,

				dbo.convert_to_UOM(SWeight,SweightUOM,'L') WeightLbs, 

				0 as Alloc,

				0 as TotalWTLBS 			

				from wksdetail 

				inner join #buyselllist on wksdetail.buysellno = #buyselllist.buysellno

				WHERE DoGradeAlloc = 1 











			CREATE TABLE #BuysellGradeAllocList_Totals(

				LedgerID INT,

				TotalWTLBS decimal(22,5)

			)



			INSERT #BuysellGradeAllocList_Totals

				SELECT LedgerID, SUM(WeightLbs)

				FROM #BuysellGradeAllocList

				GROUP BY LedgerID





			UPDATE #BuysellGradeAllocList 

				SET TotalWTLBS = B.TotalWTLBS

				FROM #BuysellGradeAllocList A 

				INNER JOIN #BuysellGradeAllocList_Totals B ON A.LedgerID = B.LedgerID



			

			-- 06/08/2015 RS: Changed

			-- 06/08/2015 DSH: NOTE Only used for allocating expenses by weight.

			UPDATE #BuysellGradeAllocList 

				SET Alloc = WeightLbs / TotalWTLBS WHERE TotalWTLBS <> 0 





				

			--DO EXPENSE ALLOCATION BY WEIGHT.  06/08/2015-DSH

			UPDATE #BuysellGradeAllocList SET 

				ItemExpense = Round(Expense * Alloc,2), 

				ItemChargeback = Round(ChargeBack * Alloc,2)				

				WHERE TotalWTLBS <> 0 



			UPDATE #BuysellGradeAllocList SET 

				Sale = ItemSales+ItemChargeBack,

				Cost = ItemCost+ItemExpense,

				Expense = ItemExpense,				

				GrossProfit = (ItemSales+ItemChargeBack)-ItemCost-ItemExpense

				WHERE TranSource = 'B'



			UPDATE #BuysellGradeAllocList SET 

				Sale = Round(Sale * Alloc,2),

				Cost = Round(Cost * Alloc,2),

				Expense = NULL,				

				GrossProfit = Round(GrossProfit * Alloc,2)

				WHERE TranSource = 'A'







			

			--Adjust for rounding errors.

			-----------------------------------------------------------

			CREATE TABLE #GPLedgerRoundingFix(

				LedgerID INT,

				CalcSale MONEY,

				CalcCost MONEY,

				CalcGP MONEY,

				Sale MONEY,

				Cost MONEY,

				GP MONEY,

				AdjSale MONEY,

				AdjCost MONEY,

				AdjGP MONEY

			)

			

			INSERT #GPLedgerRoundingFix(LedgerID,CalcSale,CalcCost,CalcGP)

				SELECT LedgerID, Sum(Sale),Sum(Cost),Sum(GrossProfit) 

				FROM #BuysellGradeAllocList

				GROUP BY LedgerID



			UPDATE #GPLedgerRoundingFix SET

				#GPLedgerRoundingFix.Sale = #BuySelllist.Sale,

				#GPLedgerRoundingFix.Cost = #BuySelllist.Cost,

				#GPLedgerRoundingFix.GP = #BuySelllist.Grossprofit

			FROM #GPLedgerRoundingFix 

			INNER JOIN #BuySelllist ON #GPLedgerRoundingFix.LedgerID = #BuySelllist.LedgerID



			UPDATE #GPLedgerRoundingFix Set

				AdjSale = Sale - CalcSale, 

				AdjCost = Cost - CalcCost,

				AdjGP = GP - CalcGP

			

			--Apply rounding fixes.	

			UPDATE #BuysellGradeAllocList SET

				Sale = A.Sale + AdjSale,

				Cost = A.COst + AdjCost,

				GrossProfit = A.GrossProfit + AdjGP

			FROM #BuysellGradeAllocList A 

			INNER JOIN #GPLedgerRoundingFix B ON A.LedgerID = B.LedgerID AND A.SeqNo = 1			 

				

			-----------------------------------------------------------



			--Update BuySellList

			UPDATE #BuySelllist SET 

				Sale = B.Sale,

				Cost = B.Cost,

				GrossProfit = B.GrossProfit,

				GradeID = B.GradeID,

				TotalWeightLbs = B.WeightLbs,

				SeqNo = 1		

			FROM #BuySelllist A INNER JOIN #BuysellGradeAllocList B ON A.LedgerID =  B.LedgerID AND B.SeqNo = 1



			INSERT #BuySelllist(LedgerID,BuySellNo,GrossProfit,Source,TradeDt,CustID,VendorID,GradeID,TotalWeightLbs,Sale,Cost,AdjType,AdjReason,GLAcct,PONumber,AdjustQty,SeqNo,DoGradeAlloc) 

			SELECT 

				B.LedgerID,B.BuySellNo,B.GrossProfit,Source,TradeDt,CustID,VendorID,B.GradeID,B.WeightLbs,B.Sale,B.Cost,AdjType,AdjReason,GLAcct,PONumber,AdjustQty,B.SeqNo,1

				FROM #BuySelllist A 

				INNER JOIN #BuysellGradeAllocList B ON A.LedgerID = B.LedgerID AND B.SeqNo > 1



		END









		







    -- CREATE SUPPLIER ALLOCATION RECORDS FOR EVERY TRANSACTION. 

    --==========================================================



    -- BROKERAGE 

	INSERT #SupplierAlloc(AllocType,VendorID,GradeID,BuySellNo,LedgerID,Allocation,InvVal,[Source],Qty,Sales,Cost,GrossProfit,TotalProfit,AdjQty,BrkExp)

        SELECT 'BRK',

				BUYSELL.VendorID, 

				#BUYSELLLIST.GradeID,

				BUYSELL.BuySellNo, 

				LedgerID, 

				1.00, NULL, 

				#BUYSELLLIST.[Source], 				

				#BUYSELLLIST.TotalWeightLbs,

				#BUYSELLLIST.Sale,   

				#BUYSELLLIST.Cost,

				#BUYSELLLIST.GrossProfit,

				#BUYSELLLIST.GrossProfit,

				#BUYSELLLIST.AdjustQty,

				BuySell.PFreightChg	

			FROM BUYSELL INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO			

			WHERE ([Source] = 'B')

				AND

				(

					BuySell.VENDORID <> '(INV)' 

					OR 

					(BuySell.VENDORID = '(INV)' AND NOT EXISTS(SELECT WksInvID FROM WKSInventory WHERE WKSInventory.BuySellno = BuySell.BuySellNo))

				  )

    INSERT #SupplierAlloc(AllocType,VendorID,GradeID,BuySellNo,LedgerID,Allocation,InvVal,[Source],Qty,Sales,Cost,GrossProfit,TotalProfit,AdjQty)

        SELECT 'ADJ',

   -- SALES ADJUSTMENTS.    

    INSERT #SupplierAlloc(AllocType,VendorID, GradeID,BuySellNo,LedgerID,Allocation,InvVal,[Source],Qty,Sales,Cost,GrossProfit,TotalProfit,AdjQty)

    --INVENTORY PURCHASES **NON-CONVERTING **  

    --==============================================   

	INSERT #SupplierAlloc(AllocType,VendorID,GradeID,BuySellNo,LedgerID,WksDetailID,[Source],Qty,InvVal,TotalProfit,Sales,Cost,AdjQty)

        SELECT 

			'INV',

			Inventory.VendorID, 

			Inventory.GradeID,

			WksInventory.BuySellNo, 

			#BUYSELLLIST.LedgerID,

			WksInventory.DetailID,			

			#BUYSELLLIST.[Source],							

			sum( dbo.CnvToBaseQtyUOM(WksInventory.[Weight],WksInventory.UOM) ),			

			sum(WksInventory.wksInvValue),

			#BUYSELLLIST.GrossProfit,

			#BUYSELLLIST.Sale,

			#BUYSELLLIST.Cost,

			COALESCE(#BUYSELLLIST.AdjustQty,0)		

        FROM WksInventory 

			INNER JOIN #BUYSELLLIST ON WksInventory.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

			INNER JOIN Inventory ON WksInventory.Lotno = Inventory.Lotno 

			INNER JOIN BUYSELL InvoiceBuysell WITH(NOLOCK)  ON WksInventory.BUYSELLNO = InvoiceBuysell.BUYSELLNO   

			WHERE WksInventory.[Weight] <> 0 			

			AND Coalesce(Inventory.ConvertingJobNo,0) = 0					--Exclude Converting Lots.

			AND Coalesce(AdjType,'') <> 'P'									--Exclude Purchase Adjustments

			AND Coalesce(AdjReason,'') <> 'TOLERANCE ADJUSTMENT'

        GROUP BY WksInventory.BuySellNo, LedgerID, DetailID,[Source],Inventory.VendorID,Inventory.GradeID,#BUYSELLLIST.GrossProfit,#BUYSELLLIST.Sale,#BUYSELLLIST.Cost,#BUYSELLLIST.AdjustQty

    INSERT #_SubInventory	

        SELECT 

			WksInventory.BuySellNo,

			#BUYSELLLIST.LedgerID,

			WksInventory.DetailID,	--Sales Worksheet Detail ID		

			WksInventory.WksInvID,  

			#BUYSELLLIST.[Source],							

			InventorySubLots.Lotno,

			row_number() OVER (Partition by WksInventory.WksInvID, #BUYSELLLIST.LedgerID order by WksInventory.WksInvID, #BUYSELLLIST.LedgerID), 	

			--row_number() OVER (Partition by #BUYSELLLIST.LedgerID, InventorySubLots.Lotno order by #BUYSELLLIST.LedgerID,InventorySubLots.Lotno), 						

			InventorySubLots.VendorID, 

			Inventory.GradeID, 

			InventorySubLots.InvValuePrct,

			InventorySubLots.WeightPrct,

			WksInventory.WksInvValue,

			dbo.CnvToBaseQtyUOM(WksInventory.[Weight],WksInventory.UOM),

			NULL,NULL,#BUYSELLLIST.AdjustQty			

		FROM #BUYSELLLIST 

		INNER JOIN WksInventory ON  #BUYSELLLIST.BUYSELLNO = WksInventory.BUYSELLNO 

        INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno 

        INNER JOIN InventorySubLots ON InventorySubLots.Lotno = Inventory.LotNo         

        WHERE WksInventory.[Weight] <> 0 		

			AND Coalesce(ConvertingJobNo,0) <> 0				--Exclude Converting Lots.

			AND Coalesce(#BUYSELLLIST.AdjType,'') <> 'P'						--Exclude Purchase Adjustments		

			AND Coalesce(AdjReason,'') <> 'TOLERANCE ADJUSTMENT'





	UPDATE #_SubInventory SET 

		InvValue = Round(TotalInvValue * InvValuePrct,2),

		InvWeight = Round(TotalWeightLbs * WeightPrct,0)





			

	-- TRUE-UP COST ALLOCATION VARIANCES.

	CREATE TABLE #_TrueUp(WksInvID INT, LedgerID INT, Lotno CHAR(10), TotalCost Money, TotalWeight Decimal(18,5),

		WHERE #_SubInventory.LotSeq = 1 AND Variance <> 0

		FROM #_SubInventory INNER JOIN #_TrueUp ON #_SubInventory.LedgerID = #_TrueUp.LedgerID AND #_SubInventory.WksInvID = #_TrueUp.WksInvID

	/* Special Case: Get value of any product sales not tied to inventory. Treat these as sales chargebacks. */

	INSERT #ChargeBackItems

			FROM WksDetail INNER JOIN #BUYSELLLIST ON WksDetail.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

			WHERE #BUYSELLLIST.VendorID = '(INV)' AND Source = 'B'

			GROUP BY WksDetail.BuysellNo

    UPDATE #SupplierAlloc 		

		SET

			InvSale = ( (#WKSDETAIL.SAmount / #WKSDETAIL.TotalQty) * Qty), 

						

			--Worksheet Charegbacks per base UOM. 

			InvSaleChgBk = (((BuySell.SFreightChg + COALESCE(#ChargeBackItems.ChargeBackAmt,0)) / #SupplierAlloc.TotalQty) * Qty),			

			

			--Worksheet Expenses per base UOM. 

			ExpVal =((( BuySell.PFreightChg) / #SupplierAlloc.TotalQty) * Qty)		 	  

			

    FROM #WKSDETAIL WITH(NOLOCK) INNER JOIN #SupplierAlloc ON #WKSDETAIL.WksDetailID = #SupplierAlloc.WksDetailID

		INNER JOIN BUYSELL WITH(NOLOCK) ON BUYSELL.BUYSELLNO = #SupplierAlloc.BuySellNo		

		LEFT JOIN #ChargeBackItems ON BUYSELL.BuySellNo = #ChargeBackItems.BuySellNo

				(@SupplierID = '' OR @SupplierID = #SupplierAlloc.VendorID	)	

				AND (@CustomerID = '' OR @CustomerID = BuySell.CustID)    				

			GROUP BY #SupplierAlloc.VendorNm, BuySell.ShipFromID, Grade.Description

			ORDER BY VendorNm, ShipFrom, [Description]

		END

	END





















	-- PRODUCE FINAL REPORT: GROSS PROFIT BY PRODUCT.

	--===============================================

	

	

	IF @ReportType = 'GRADE' BEGIN



		UPDATE #SupplierAlloc SET

			SalesPerUOM = #SupplierAlloc.Sales / ISNULL(dbo.Convert_to_UOM(#SupplierAlloc.Qty,'L',@ReportUOM),0),

			CostsPerUom  = #SupplierAlloc.Cost / ISNULL(dbo.Convert_to_UOM(#SupplierAlloc.Qty,'L',@ReportUOM),0),

			MarginPerUOM = #SupplierAlloc.GrossProfit / ISNULL(dbo.Convert_to_UOM(#SupplierAlloc.Qty,'L',@ReportUOM),0)

			WHERE ISNULL(#SupplierAlloc.Qty,0) <> 0













		--Detail Report

		IF @GroupBy = 0 





			SELECT 				

				CASE WHEN Grade.[Description] IS NULL THEN 'UNASSIGNED' ELSE Grade.[Description] END AS GradeNm, 

				InvoiceDt PostingDt,

				BUYSELL.ShippingDt,

				CASE WHEN Source='B' THEN 'Invoice' ELSE 'Adjustment' END TranType,

				#SupplierAlloc.BuySellNo TranNo, 

				ISNULL(dbo.Convert_to_UOM(#SupplierAlloc.Qty,'L',@ReportUOM),0) AS Quantity,

				SalesPerUOM,

				#SupplierAlloc.Sales,

				CostsPerUOM,

				#SupplierAlloc.Cost,

				MarginPerUOM,

				#SupplierAlloc.GrossProfit Margin ,

				#SupplierAlloc.BuySellNo as [Wks #]

				FROM #SupplierAlloc 

				LEFT JOIN BUYSELL WITH(NOLOCK) ON #SupplierAlloc.BUYSELLNO = BUYSELL.BUYSELLNO	        

				LEFT JOIN Counterparty Supplier ON Supplier.CPID = #SupplierAlloc.VendorID				

				LEFT JOIN Counterparty Customer ON Customer.CPID = BuySell.CustID

				LEFT JOIN GRADE ON #SupplierAlloc.GradeID = Grade.GradeID				

			WHERE 

				(@SupplierID = '' OR @SupplierID = #SupplierAlloc.VendorID	)	

				AND (@CustomerID = '' OR @CustomerID = BuySell.CustID)   

				AND (@ProductCategory = '' OR Grade.ProductCategoryID = @ProductCategory)

				AND (@ProductGrade = '' OR Grade.GradeID = @ProductGrade) 

				ORDER BY Grade.Description

		

		--Summary Report	

		IF @GroupBy = 1 BEGIN

			

			CREATE TABLE #Summary(

				GradeNm CHAR(100),

				GradeID CHAR(15),

				Weight DECIMAL(22,5),

				Sales MONEY,

				SalesPerUOM MONEY,

				Costs MONEY,

				CostsPerUOM MONEY,

				Margin MONEY,

				MarginPerUOM MONEY,

				ProductCategory char(30)

			)



			INSERT #Summary

			SELECT 

				CASE WHEN Grade.[Description] IS NULL THEN 'UNASSIGNED' ELSE UPPER(Grade.[Description]) END AS GradeNm, 				

				Grade.GradeID,

				SUM(ISNULL(dbo.Convert_to_UOM(#SupplierAlloc.Qty,'L',@ReportUOM),0)) AS Quantity,

				SUM(#SupplierAlloc.Sales),0,

				SUM(#SupplierAlloc.Cost),0,

				SUM(#SupplierAlloc.GrossProfit),0,

				Grade.ProductCategoryID AS [Product Category]

				FROM #SupplierAlloc 

				LEFT JOIN BUYSELL WITH(NOLOCK) ON #SupplierAlloc.BUYSELLNO = BUYSELL.BUYSELLNO	        

				LEFT JOIN Counterparty Supplier ON Supplier.CPID = #SupplierAlloc.VendorID				

				LEFT JOIN Counterparty Customer ON Customer.CPID = BuySell.CustID

				LEFT JOIN GRADE ON #SupplierAlloc.GradeID = Grade.GradeID

			WHERE 

				(@SupplierID = '' OR @SupplierID = #SupplierAlloc.VendorID	)	

				AND (@CustomerID = '' OR @CustomerID = BuySell.CustID)   

				AND (@ProductGrade = '' OR Grade.GradeID = @ProductGrade) 

				AND (@ProductCategory = '' OR Grade.ProductCategoryID = @ProductCategory) 				

				GROUP BY Grade.[Description], Grade.GradeID, Grade.ProductCategoryID

				ORDER BY Grade.[Description], Grade.GradeID  --IS NULL THEN 'UNASSIGNED' ELSE Grade.[Description] END 



			UPDATE #Summary SET

				SalesPerUOM = Sales/Weight,

				CostsPerUOM = Costs/Weight,

				MarginPerUOM = Margin/Weight

			WHERE Weight <> 0 



			SELECT * FROM #Summary







		END

	END
