CREATE PROC [dbo].[cw_Closing_Alert_AdjPostPriorToInvoice] @DoColumnDefOnly BIT, @FromDt datetime,@ToDT datetime,@COID char(2)

AS

SET NOCOUNT ON

--09/19/2012 DSH Created for Closing Adjustment Ledger.

--03/09/2020 Alex Added  CustID,VendorID



	/* DEFINE REPORT COLUMNS SECTION. */

	IF @DoColumnDefOnly = 1 BEGIN

	

		CREATE TABLE #ReportDef(

			ColName char(25),

			ColAlign char(1),

			ColWidth int,

			ColDataType char(10),

			ColNumFormat char(15)

		)

		

		INSERT #ReportDef SELECT 'Invoice No.','L',1250,'TEXT',''		

		INSERT #ReportDef SELECT 'Worksheet No.','L',1250,'TEXT',''		

		INSERT #ReportDef SELECT 'Received Date','L',1250,'DATE',''		

		INSERT #ReportDef SELECT 'Invoice Date','L',1250,'DATE',''		

		INSERT #ReportDef SELECT 'Vendor','L',2100,'TEXT',''

		INSERT #ReportDef SELECT 'Reason','L',1500,'TEXT',''

		INSERT #ReportDef SELECT 'Amount','R',1300,'NUM','#,###,##0.00'		

		INSERT #ReportDef SELECT 'Currency','L',700,'TEXT',''		

		INSERT #ReportDef SELECT 'User','L',1000,'TEXT',''

		INSERT #ReportDef SELECT 'HC Amount','R',1300,'NUM','#,###,##0.00'		

		

		SELECT ColName, ColAlign, ColWidth, ColDataType, ColNumFormat FROM #ReportDef

		RETURN

	END	

	

	

	CREATE TABLE #Report( 

		InvoiceNo CHAR(100),			

		BuySellNo CHAR(12),

		RcvDate DATETIME, 	

		InvDate DATETIME, 	

		CompanyNm VARCHAR(65),

		AdjReason VARCHAR(100),

		FxAmount MONEY,

		FxCurrency CHAR(3),		

		UserID CHAR(10),  

		Amount MONEY,

		[-Hidden-CustID] char(12),

		[-Hidden-VendorID] char(12)

	)

	

	

	

	INSERT #Report

    SELECT 

		Payables.InvoiceNo,

		GPLedger.BuySellNo, 

		payables.RcvDt,

		payables.InvoiceDt,

		CompanyNm,

		AdjReason,			

		AdjFxAmount, AdjCurrencyCd,

	    GPLedger.UserID,		

		GPLedger.Cost	,

	    CustID,

	    VendorID			

    FROM gpledger WITH(NOLOCK) inner join buysell WITH(NOLOCK) ON gpledger.BuySellNo = BuySell.BuySellNo

    inner join payables WITH(NOLOCK) on (gpledger.ledgerid = payables.itemid and gpledger.buysellno = payables.buysellno)     

    inner join counterparty ON gpledger.AdjCPID = counterparty.CPID

    WHERE Source = 'A' AND AdjType = 'P' AND GPLedger.AdjCompleted='Y' AND AdjPosted='Y'

    AND (@CoID = '' OR @CoID = BuySell.CoID) 

    AND payables.RcvDt < payables.InvoiceDt

    AND BuySell.InvoiceDt >= @FromDt and BuySell.InvoiceDt <= @ToDt

    Order By AdjReqDate



	----

	INSERT #REPORT(InvoiceNo,Amount)			

		SELECT 

			'TOTAL', 

			SUM(Amount)

		FROM #Report 

	

	--SELECT * FROM #Report

	





	select 

		InvoiceNo [Invoice No],			

		BuySellNo [Worksheet No],

		RcvDate , 	

		InvDate [Invoice Date], 	

		CompanyNm [Vendor],

		AdjReason ,

		FxAmount  ,

		FxCurrency ,		

		--UserID [User],  

		UserDef.UserNm [User],  

		Amount ,

		[-Hidden-CustID],

		[-Hidden-VendorID] 

		FROM #Report

		left join UserDef on UserDef.UserID=#Report.USERID

--go
