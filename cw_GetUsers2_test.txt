CREATE PROC [dbo].[cw_GetUsers2_test] @UserName varchar(100)='', @DBID int, @Status varchar = '' AS

--08/22/2014 MEC: Created

--07/24/2015 MCM: Modified to work with master user tables

--08/14/2019 Alex: Returned 

--10-17-2022 Dakota: made changes, added active users query when status = '2'

 

	SET NOCOUNT ON;

	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;



	--SELECT UserID As [USER ID], UserNm As [USER NAME], Case When Active = 1 Then 'Active' Else 'Inactive' End AS [STATUS]	

	--FROM cieTrade_Sys.dbo.Users WITH(NOLOCK)

	--WHERE (@UserName = '' OR UserNm Like @UserName + '%')

	--AND (@Status = '' OR @Status = ISNULL(Active,1))

	--ORDER BY UserNm, UserID



	-- u: User

	-- ua: UserApp

	-- ud: UserDef

	-- db: DBUser

	IF(@Status = '')

	BEGIN

		SELECT

			cm.CustomerID As [-Hidden-OwnerUserID],

			u.LoginID As [Login ID],

			u.UserName As [User Name],

			u.UserName As [User Details],

			--u.IsLocked,

			CASE WHEN ISNULL(u.IsLocked,0) = 1 THEN 'Inactive' ELSE 'Active' END AS [Status],	--10/14/2022 Dakota: changed to IsLocked

			ur.RoleNm As [Role],

			u.UserEmail AS [Email],

			u.UserID As [-Hidden-USER ID],

			isnull(ud.EmailSignature,'') as EmailSignature,   --08/14/2019 Alex:

			CASE WHEN ISNULL(AllowLogin,0) = 1 THEN 'Active' ELSE 'Inactive' END AS [Session],	--10/14/2022 Dakota: new is active field

			isnull(ud.lastLogin,'--') as [Last Login],	--10/14/2022 Dakota: last login for user

			isnull(ud.TimeZoneNm,0) as TimeZone

		FROM

			[CIE-SQLPROD01].[cieXChange].[dbo].DBUser db 

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].[User] u ON db.UserID = u.UserID

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].UserApp ua ON db.UserID = ua.UserID

			INNER JOIN UserDef ud ON ud.UserID = u.cieTradeUserID

			LEFT JOIN [CIE-SQLPROD01].[cieXChange].[dbo].Company cm ON db.UserID = cm.OwnerUserID

			LEFT JOIN UserRole ur ON ur.RoleID = ud.RoleID

		WHERE

			(@UserName = '' OR u.UserName LIKE @UserName + '%')

			AND LoginID <> 'xa' and u.username <> 'TestUser'

			AND db.[DBID]=@DBID 

			AND [AppName]='cietrade.net'

			--and (loginID <> 'admin' OR DBName = 'cieTradeNET')			-- Added this line to include admin account only for cieTradeNET Database

		ORDER BY u.UserName

	END

	ELSE IF(@Status = '0' or @Status = '1')

	BEGIN

			SELECT

			cm.CustomerID As [-Hidden-OwnerUserID],

			u.LoginID As [Login ID],

			u.UserName As [User Name],

			u.UserName As [User Details],

			CASE WHEN ISNULL(u.IsLocked,0) = 1 THEN 'Inactive' ELSE 'Active' END AS [Status],  --10/14/2022 Dakota: changed to IsLocked in cieXchange

			ur.RoleNm As [Role],

			u.UserEmail AS [Email],

			u.UserID As [-Hidden-USER ID],

			isnull(ud.EmailSignature,'') as EmailSignature,  --08/14/2019 Alex:

			CASE WHEN ISNULL(AllowLogin,0) = 1 THEN 'Active' ELSE 'Inactive' END AS [Session],	--10/14/2022 Dakota: new is active field

			isnull(ud.lastLogin,'--') as [Last Login],	--10/14/2022 Dakota: last login for user

			ud.TimeZoneNm as TimeZone

	

		FROM

			[CIE-SQLPROD01].[cieXChange].[dbo].DBUser db 

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].[User] u ON db.UserID = u.UserID

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].UserApp ua ON db.UserID = ua.UserID

			INNER JOIN UserDef ud ON ud.UserID = u.cieTradeUserID

			LEFT JOIN [CIE-SQLPROD01].[cieXChange].[dbo].Company cm ON db.UserID = cm.OwnerUserID

			LEFT JOIN UserRole ur ON ur.RoleID = ud.RoleID

			

		WHERE

			(@UserName = '' OR u.UserName LIKE @UserName + '%')

			AND LoginID <> 'xa' and u.username <> 'TestUser'

			AND db.[DBID]=@DBID 

			AND [AppName]='cietrade.net'

			and loginID <> 'admin'

			And isnull(u.IsLocked,0) = @Status

		ORDER BY u.UserName

	END

	ELSE BEGIN

		SELECT

			cm.CustomerID As [-Hidden-OwnerUserID],

			u.LoginID As [Login ID],

			u.UserName As [User Name],

			u.UserName As [User Details],

			CASE WHEN ISNULL(u.IsLocked,0) = 1 THEN 'Inactive' ELSE 'Active' END AS [Status],  --10/14/2022 Dakota: changed to IsLocked in cieXchange

			ur.RoleNm As [Role],

			u.UserEmail AS [Email],

			u.UserID As [-Hidden-USER ID],

			isnull(ud.EmailSignature,'') as EmailSignature,  --08/14/2019 Alex:

			CASE WHEN ISNULL(AllowLogin,0) = 1 THEN 'Active' ELSE 'Inactive' END AS [Session],	--10/14/2022 Dakota: new is active field

			isnull(ud.lastLogin,'--') as [Last Login],	--10/14/2022 Dakota: last login for user

			isnull(ud.TimeZoneNm,0) as TimeZone

		FROM

			[CIE-SQLPROD01].[cieXChange].[dbo].DBUser db 

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].[User] u ON db.UserID = u.UserID

			INNER JOIN [CIE-SQLPROD01].[cieXChange].[dbo].UserApp ua ON db.UserID = ua.UserID

			INNER JOIN UserDef ud ON ud.UserID = u.cieTradeUserID

			LEFT JOIN [CIE-SQLPROD01].[cieXChange].[dbo].Company cm ON db.UserID = cm.OwnerUserID

			LEFT JOIN UserRole ur ON ur.RoleID = ud.RoleID

			

		WHERE

			(@UserName = '' OR u.UserName LIKE @UserName + '%')

			AND LoginID <> 'xa' and u.username <> 'TestUser'

			AND db.[DBID]=@DBID  

			AND [AppName]='cietrade.net'

			and loginID <> 'admin'

			And AllowLogin = '1'

		ORDER BY u.UserName

	END





--go
