SELECT a1.buysellno, a1.[Status], a1.Logistics, convert(date, getdate())

from tblTrackStatusLogistics a1

where not exists (

SELECT b1.buysellno,

ISNULL(dbo.[CnvWksStatus_Name](b1.[Status],''),'') 

,isnull(upper([dbo].[CnvShipStatus_Name](b1.ShippingOK)),'') 

from buysell b1

where year(b1.shippingdt) >= 2025 and b1.buysellno = a1.BuySellNo

and ISNULL(dbo.[CnvWksStatus_Name](b1.[Status],''),'') = a1.[status]

and isnull(upper([dbo].[CnvShipStatus_Name](b1.ShippingOK)),'') = a1.[Logistics])
