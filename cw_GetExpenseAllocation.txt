CREATE   PROCEDURE [dbo].[cw_GetExpenseAllocation] @TransNo int AS

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED



-- GET DATA FIELD FOR EXPENSE ALLOCATION FORM.

--'EXPALLOC-CHG: DSH 09/19/2006 (allow partial inv on expense allocations)

-- 02/26/2007 JW: add fxrate

--1/3/2008 BF: added TradeType (new column)

--1/31/2008 DSH: Added In-Date (new column)

--3/31/2008 JW: Added ExpenseAlloc_Detail.FxContractNo

--05/05/2011 RS: ExpenseAlloc_Source.Weight added

--01/10/2014 DSH Added invvalue support in ExpenseAlloc_Source table.

--08/09/2016 MEC: Copied for cieTrade.net

--05/19/2022 ZT: Added LastEdit

--11/28/2022 Nick: Added Bulk Source, IsBulk

	SELECT 

		TransNo, COID, [Description], TransDate, UserID, 

		WarehouseNm, ContainerNo, IsPosted, 

        BuySellNo, Manifest, TradeType, PostingDt, WhseSelected,

        (Select Top 1 Posted From Payables WITH(NOLOCK) Inner Join ExpenseAlloc_Detail WITH(NOLOCK) On ItemID = ID Where IsNull(Posted,0) = 1 And ItemIDSource = 'E' And TransNo = @TransNo) As IsPaid,

		isnull(IsBulk, 0) as IsBulk

    FROM ExpenseAllocation WITH(NOLOCK) 

    WHERE TransNo = @TransNo



	SELECT 

		ExpenseAlloc_Source.LotNo, Grade.[Description], QtyOnHand, 

		ExpenseAlloc_Source.WeightUOM, Units, NoSheets, Inventory.Price, 

        Inventory.PriceUOM, RcvDt, ExpenseAlloc_Source.warehousenm, 

        ExpenseAlloc_Source.HCInvValue, ExpenseAlloc_Source.[Weight],

        Inventory.WarehouseNm, ExpenseAlloc_Source.UnitPriceChange,

        ExpenseAlloc_Source.UnitAmountChange, Inventory.InDate,

		convert(int,Inventory.lastedit) As [LastEdit]

	FROM (ExpenseAlloc_Source WITH(NOLOCK) 

		INNER JOIN ExpenseAllocation WITH(NOLOCK) ON ExpenseAllocation.TransNo = ExpenseAlloc_Source.TransNo)

		INNER JOIN Inventory WITH(NOLOCK) ON ExpenseAlloc_Source.LotNo = Inventory.LotNo

		INNER JOIN GRADE WITH(NOLOCK) ON Inventory.GradeID = Grade.GradeID

	WHERE ExpenseAlloc_Source.TransNo = @TransNo

		And (IsNull(IsPosted,0) = 1 OR QtyOnHand > 0)



	SELECT 

		CompanyNm, GLAcct, ExpenseAlloc_Detail.CpID, Amount, 

		FxAmount, PriceUOM, Charge, CurrencyCd, FxRate, FxContractNo, IsFreight

	FROM ExpenseAlloc_Detail WITH(NOLOCK) 

		INNER JOIN Counterparty WITH(NOLOCK) ON ExpenseAlloc_Detail.CpID = Counterparty.CpID

	WHERE TransNo = @TransNo



	--Bulk Source

	SELECT

		ExpenseAlloc_SourceBulk.WarehouseNm,

		ExpenseAlloc_SourceBulk.GradeID,

		ExpenseAlloc_SourceBulk.Cost,

		Grade.[Description]

	FROM ExpenseAlloc_SourceBulk

	INNER JOIN Grade ON Grade.GradeID = ExpenseAlloc_SourceBulk.GradeID

	WHERE ExpenseAlloc_SourceBulk.TransNo = @TransNo
