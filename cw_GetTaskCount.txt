CREATE   PROCEDURE [dbo].[cw_GetTaskCount] @UserID char(10) = '', @ReturnCount varchar(15) OUTPUT

--05/25/17 MN: Created

AS

set nocount on

set transaction isolation level read uncommitted



	SELECT @ReturnCount = COUNT(TaskID) FROM CRMTask

	WHERE (TaskComplete <> 1 OR TaskComplete IS NULL)

	AND (@UserID = '' OR UserID = @UserID)



RETURN @ReturnCount
