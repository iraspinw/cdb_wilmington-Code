Set NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

	CREATE TABLE #CommercialInvoiceList(

		SealNo VARCHAR(30),

		BookingNo CHAR(25),

		BuySellNo CHAR(12),

		ContainerNo VARCHAR(30),

		WT DECIMAL(22,10),

		SWeightUOM CHAR(3),

		Units INT,

		UnitType VARCHAR(30),

		[Description] VARCHAR(100),

		InvoiceDesc  VARCHAR(100),

		ShippingDt DATE,

		SPrice DECIMAL(22,5),

		SPriceUOM CHAR(10),

		SFxAmount MONEY,

		SCurrencyCd CHAR(3),

		GrossWT DECIMAL(22,5),

		TareWT DECIMAL(22,5),

		ShippedCount INT,

		Specifications VARCHAR(500),

		CalcWeight DECIMAL(22,8),

		RowId INT IDENTITY(1,1)	,

		WTKg DECIMAL(22,10),

	    GrossWTKg DECIMAL(22,5),

		TareWTKg DECIMAL(22,5)

	

	)



	





	-- DISPLAY IN QTY UOM

	DECLARE @SWTUOM char(1)	

	SET @SWTUOM = ISNULL((SELECT TOP 1 TotalWeightUOM FROM Booksheet WITH(NOLOCK) WHERE BookingNo = @BookingNo),'L')



	DECLARE @ShippedCount int, @ContainerSalesAmt MONEY



	SELECT @ContainerSalesAmt = SUM(SFXAmount) FROM WksDetail INNER JOIN BuySell ON WksDetail.BuYSellNo = BuYSell.BuYSellNo 

		WHERE BuySell.BookingNo = @BookingNo AND BuySell.CustID <> '(INV)'





	Set @ShippedCount = (SELECT Count(Distinct BuySell.BuySellNo) ShippedCount

		FROM BuySell WITH(NOLOCK)

		INNER JOIN WksDetail ON WksDetail.BuysellNo = BuySell.BuySellNo

		LEFT JOIN Grade ON Grade.GradeID = WksDetail.GradeID

		WHERE BuySell.BookingNo = @BookingNo And IsNull(RefEquipNo,'') <> '')







	--INVENTORY SALE (CONTAINERS) 

	----------------------------------------------------------------------------------------------------------------



	INSERT #CommercialInvoiceList

	SELECT distinct 	

		Container.SealNo, 

		Container.BookingNo, 

		BuySell.BuySellNo,

		Container.ContainerNo,		

		sum(dbo.CONVERT_TO_UOM_XP(WksInventory.[Weight],WksInventory.UOM,@SWTUOM)),   ----dbo.CONVERT_TO_UOM(NetWeight,Container.UOM,@SWTUOM) as WT,		

		dbo.CnvUOM_Names(@SWTUOM),

		isnull(Container.units,0), --isnull(WksDetail.Units,0), --isnull(Inventory.Units,0), 

		dbo.CnvUnitName(ISNULL(container.UnitType,'')),

		Grade.[Description] As [Description],

		WksDetail.InvoiceDesc,

		BuySell.ShippingDt,

		WksDetail.SPrice, 

		dbo.CnvUOM_Names(WksDetail.SPriceUOM) as SPriceUOM,

		0 SFxAmount,  --WksDetail.SFxAmount as 

		WksDetail.SCurrencyCd,

		isnull(Container.GrossWeight,sum(dbo.CONVERT_TO_UOM(Inventory.GrossWt,WksInventory.UOM,@SWTUOM))),

		Container.Tare,

		@ShippedCount, 

		CASE WHEN IsNull(ShowCommDescOnDocs,0) = 1 Then IsNull(WKSDetail.Specifications,'') + ' ' + CommodityDesc Else IsNull(WKSDetail.Specifications,'') End As Specifications,

		0,

	    sum(dbo.CONVERT_TO_UOM_XP(WksInventory.[Weight],WksInventory.UOM,'KG')),

		sum(dbo.CONVERT_TO_UOM_XP(Inventory.GrossWt,WksInventory.UOM,'KG')),

		sum(dbo.CONVERT_TO_UOM_XP(Inventory.TareWt,WksInventory.UOM,'KG'))

	 --dbo.CONVERT_TO_UOM_XP(Container.GrossWeight,WksInventory.UOM,'KG'),

	--dbo.CONVERT_TO_UOM_XP(Container.Tare,WksInventory.UOM,'KG')

	FROM WksInventory 		

		INNER JOIN Inventory ON WksInventory.LotNo = Inventory.LotNo

		LEFT JOIN WksDetail ON WksInventory.DetailID = WksDetail.DetailID

		LEFT JOIN ContainerContents ON Inventory.LotNo = ContainerContents.LotNo

		LEFT JOIN Container ON Container.ContainerID =  ContainerContents.ContainerID 

		LEFT JOIN Grade ON Inventory.GradeID = Grade.GradeID

		LEFT JOIN BuySell ON WksInventory.BuySellNo = BuySell.BuySellNo

	WHERE Container.BookingNo = @BookingNo 

	AND BuySell.VendorID = '(INV)' AND BuySell.CustID <> '(INV)' 

	group by Container.SealNo, Container.BookingNo, BuySell.BuySellNo,Container.ContainerNo,WksDetail.Units,container.UnitType,	Grade.[Description],

		     WksDetail.InvoiceDesc,BuySell.ShippingDt,WksDetail.SPrice, WksDetail.SPriceUOM ,WksDetail.SCurrencyCd,Container.GrossWeight,Container.Tare,

			 ShowCommDescOnDocs,WKSDetail.Specifications,CommodityDesc,Container.units



	UPDATE #CommercialInvoiceList Set CalcWeight = dbo.convert_to_uom(Wt,@SWTUOM,SPriceUOM)



	UPDATE #CommercialInvoiceList Set SFxAmount = Round(CalcWeight * SPrice,2)



	--TRUE-UP

	DECLARE @CalcTotalSales MONEY

	DECLARE @Adj MONEY

	SELECT @CalcTotalSales  = sum(SFxAmount) from #CommercialInvoiceList

	SET @Adj = @ContainerSalesAmt- @CalcTotalSales

	UPDATE #CommercialInvoiceList SET SFxAmount = SFxAmount + @Adj WHERE RowID = 1

	





	--BROKERAGE & INVENTORY SALE (NON CONTAINER)

	----------------------------------------------------------------------------------------------------------------

	INSERT #CommercialInvoiceList

	SELECT 

		BuySell.RefSealNo As SealNo, 

		BookingNo, 

		BuySell.BuySellNo,

		BuySell.RefEquipNo as ContainerNo,

		dbo.FormatQty(SUM(dbo.CONVERT_TO_UOM(WksDetail.SWeight,SWeightUOM,@SWTUOM)),@SWTUOM) as WT,

		dbo.CnvUOM_Names(@SWTUOM) as SWeightUOM,

		Sum(WksDetail.Units) As Units,

		dbo.CnvUnitName((Select Top 1 Unittype From wksdetail with (nolock) Where WksDetail.BuySellNo = BuySell.BuySellNo)) As UnitType, 

		Grade.[Description] As [Description],

		WksDetail.InvoiceDesc,

		BuySell.ShippingDt,

		WksDetail.SPrice, 

		dbo.CnvUOM_Names(WksDetail.SPriceUOM), 

		SUM(WksDetail.SFxAmount) as SFxAmount,

		WksDetail.SCurrencyCd,

		dbo.FormatQty(SUM(dbo.CONVERT_TO_UOM(WksDetail.GrossWt,SWeightUOM,@SWTUOM)),@SWTUOM) as GrossWT,

		dbo.FormatQty(SUM(dbo.CONVERT_TO_UOM(WksDetail.TareWt,SWeightUOM,@SWTUOM)),@SWTUOM) as TareWT,

		@ShippedCount AS [ShippedCount],

		CASE WHEN IsNull(ShowCommDescOnDocs,0) = 1 Then IsNull(WKSDetail.Specifications,'') + ' ' + CommodityDesc Else IsNull(WKSDetail.Specifications,'') End As Specifications,

		0,

	   sum( dbo.CONVERT_TO_UOM_XP(WksDetail.SWeight,SWeightUOM,'KG')),

	   sum(dbo.CONVERT_TO_UOM_XP(WksDetail.GrossWt,SWeightUOM,'KG')),

	   sum(dbo.CONVERT_TO_UOM_XP(WksDetail.TareWt,SWeightUOM,'KG'))

	FROM BuySell WITH(NOLOCK)

	INNER JOIN WksDetail ON WksDetail.BuysellNo = BuySell.BuySellNo

	LEFT JOIN Grade ON Grade.GradeID = WksDetail.GradeID

	WHERE 

		BuySell.BookingNo = @BookingNo

		AND BuySell.RefEquipNo <> ''

		AND CustID <> '(INV)' 

		AND ISNULL(IsContainer,0) = 0 

		GROUP BY RefEquipNo, RefSealNo, BookingNo, BuySell.BuySellNo, Grade.Description, 

		Grade.CommodityDesc,InvoiceDesc,BuySell.ShippingDt,SPrice, SPriceUOM, WksDetail.SCurrencyCd, Specifications, ShowCommDescOnDocs,WksDetail.SWeight,WksDetail.SWeightUOM,WksDetail.GrossWt,WksDetail.TareWt

	HAVING Sum(dbo.CONVERT_TO_UOM(WksDetail.SWeight,WksDetail.SWeightUOM,'M'))>0

	ORDER BY BuySell.ShippingDt, RefEquipNo



	

	--GET CHARGEBACKS

	INSERT #CommercialInvoiceList(BookingNo,BuySellNo,ContainerNo,Wt,SWeightUOM,[Description],InvoiceDesc, ShippingDt, SPrice, SPriceUOM, SFxAmount, SCurrencyCd, ShippedCount)

	SELECT 

		BuySell.BookingNo,

		BuySell.BuySellNo,

		'',0,NULL,

		InvoiceDesc,

		InvoiceDesc, 

		BuySell.ShippingDt, 

		PriceCust,

		PricePerCust,

		FxAmountCust, 

		CurrencyCdCust,

		@ShippedCount  --, , GLAcct,  

		FROM BuySell WITH(NOLOCK)

		INNER JOIN WksDelivery WITH(NOLOCK) ON WksDelivery.BuysellNo = BuySell.BuySellNo

	WHERE BuySell.BookingNo = @BookingNo 
		AND WksDelivery.ChgToCustAmt <> 0
