CREATE PROC [dbo].[cw_GetSupplierFromOrder] @PONO varchar(15) as

set transaction isolation level read uncommitted

set nocount on



--03/08/2019 MEC: Added Company Name for display on the Container Insert



--cw_GetSupplierFromOrder '100324'



select top 1

	po.cpid, addrtype, CompanyNm,

	CASE WHEN IsNull(IsContract,0) = 1 THEN isnull(PO.FxRate,1) ELSE isnull(CurrencyRate.FxRate,1) END FxRate, 

	CurrCode, ContractNo, IsContract, COID, TradeType

from po 

inner join CounterParty on po.CpID = CounterParty.CpID 

left join CurrencyRate on CurrencyRate.CurCode = counterparty.CurrCode and exchdate <= GETDATE()

where ponumber = @pono and source='PO'

order by ExchDate desc
