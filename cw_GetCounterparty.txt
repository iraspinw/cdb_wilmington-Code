	-- ADDED for Contact Support *BEGIN*

	DECLARE @APContact varchar(50) = ''

	DECLARE @APVoiceNo varchar(50) = ''

	DECLARE @APFaxNo varchar(30) = '' -- field deprecated

	DECLARE @APEmail varchar(300) = ''

	SELECT Top 1 @APContact = ContactNm, @APVoiceNo = PhoneBusiness, @APEmail = Email From Contact WITH(NOLOCK) 

	INNER JOIN ContactRoleAssignment WITH(NOLOCK) ON Contact.CT_ID = ContactRoleAssignment.CT_ID 

	WHERE RoleNm In ('Collections', 'Primary') And CpID = @CpID Order By RoleNm

	

	DECLARE @ContactEmail varchar(150) = ''

	DECLARE @Contact varchar(200) = ''

	DECLARE @Telephone varchar(50) = ''

	DECLARE @Email varchar(300) = '' -- field may be deprecated

	SELECT Top 1 @Contact = ContactNm, @Telephone = PhoneBusiness, @ContactEmail = Email From Contact WITH(NOLOCK) 

	Where CpID = @CpID AND IsNull(IsPrimary,0) = 1

	--INNER JOIN ContactRoleAssignment WITH(NOLOCK) ON Contact.CT_ID = ContactRoleAssignment.CT_ID 

	--WHERE RoleNm In ('Billing', 'Primary') And CpID = @CpID Order By IsNull(IsPrimary,0) Desc, RoleNm

	-- ADDED for Contact Support *END*



    SELECT CpID,CompanyNm,Role,'' as Contact,'' as VoiceNo,'' as FaxNo,'' as EMail,ReferenceID,GLAcctSupplier,GroupNm,

           InvoiceNote,ShippingLineRemarks,Terms,CustSvcRep,UserNotes,CreditLimit,@APContact APContact,@APVoiceNo APVoiceNo,ReqReview,

           AltRemitInfo,GLAcctExpense,CurrCode,ActiveStatus,TermsCode,@APFaxNo APFaxNo,MinWt,MinWtUOM,

           SendRcvWtFax,CommissionRole,ShipAddrForInvoice,

           FreightType,WebLogin,WebUserID,WebPassword,IndustryNm, SalesTypeID, Website, convert( int,CpLastEdit) AS EditStamp, SCAC,CustomsCode,

	       EDIVan, EDIVanID, EDIVanIDQualifier, EDIContact, EDIContactEmail, ActiveCreditProtocolID,@APEmail APEMail, AutoFeedAPVoucher, IsInterCompany,

		   AvgDSODelay, DistributionMode, OurCustNo, UserDefined1, UserDefined2, OnHold, UserDefined3, UserDefined4,InvoiceType, BillingCycle,

		   StatementType, ReportUOM,BillOn, ReverseBilling, WebIsAdmin, RemitID, BatchPay, CPSource, PrintPriceLetter, ForeignLang_CompanyNm, ForeignLang_AddressBlock,

		   EDIOutStartDt, ScaleCheck, LocationEmailContract,CommAgentID,CommGLAcct,CommRate,CommRatePer,CommCurrency, AcctingIDCust, AcctingIDVendor, isFreight, InvoiceFmt,

		   UserDefined1,UserDefined2,UserDefined3,UserDefined4,isnull(RemitID,0) as RemitID,isnull(ReverseBilling,0) as ReverseBilling,isnull(BillingCycle,'') as BillingCycle,isnull(DistributionMode,'') as DistributionMode,

		   ReportGroup,isnull(BatchInvoiceGroup,0) as BatchInvoiceGroup,

		   isnull(isBillingAddressOnly,0) as isBillingAddressOnly,

		   isnull(BANK_Name, '') As BANK_Name,

		   isnull(BANK_PaymentType, '') As BANK_PaymentType,

		   isnull(BANK_AcctNo, '') As BANK_AcctNo,

		   isnull(BANK_RoutingNo, '') As BANK_RoutingNo,

		   isnull(BANK_SwiftNo, '') As BANK_SwiftNo,

		    isnull(BANK_Email, '') As BANK_Email,

		   RevBillingDocFormID,

		   RevBillingDocUOM,

		   isnull(PaymentDays, 0) as PaymentDays,

		   isnull(ReasonCode, '') as ReasonCode,

		   isnull(ReasonDescription, '') as ReasonDescription,

		   Counterparty.CounterpartyLogo

		   



    FROM COUNTERPARTY WITH(NOLOCK) WHERE CPID = @CpID



    SELECT Address.Type,Addr1,Addr2,Addr3,City,Region,PostalCd,Country.Name AS [Country],Instructions,Address.Notes,

           EDI_ShipToCode,AvgBalewtLBS,AllocID, [Address].SalesTaxCodeID,SalesTaxCode.SalesTaxCode, PriceRegionID, EDI_ShipFromCode, TaxID, directions, 

		   CRVCertType, CRVCertNum, InternalFreightCost,@ContactEmail BillingEmail,@Contact Contact,@Telephone Telephone, @Email Email, CountryID,Addr4

	FROM Address WITH(NOLOCK) 

	Left Join SalesTaxCode With(NoLock) On SalesTaxCode.SalesTaxCodeID = [Address].SalesTaxCodeID

	LEFT JOIN Country With(NoLock) On Address.Country = Country.CountryID

	where Address.CpID =  @CpID  AND InvoiceAddr = 'Y' 



    -- SALES REPS

    SELECT referrer.name as Name, Counterparty_Referrer.ReferrerNo, Rate, Charge, IsDefault, ChgPer  

	FROM Counterparty_Referrer WITH(NOLOCK) 

    INNER JOIN referrer WITH(NOLOCK) ON  Counterparty_Referrer.ReferrerNo = referrer.ReferrerNo

    WHERE  CPID = @CpID

	order by isdefault desc

    --SELECT ItemID, Description,a.GRADEID,a.UnitType,a.BW,a.BWUOM,a.COLOR,a.FINISH,a.RW,a.RWUOM,a.Core,a.CoreUOM,a.Caliper,a.SheetSIZE,a.SheetSizeUOM,

    --       a.MWeight,a.Grain,a.JobLot,a.Condition,a.Spec,a.Notes,a.PriceIdea,a.PriceIdeaUOM,a.PriceIdeaCurrency, a.SpeciesID, SKU, AltDesc,

	-- CHECK IS CUSTOMTER

	declare @IsCustomer bit

	SET @IsCustomer = 0

	IF EXISTS(SELECT CpID From Counterparty WHERE CpID = @CpID and [Role] in ('A', 'C', 'P', 'X'))

		SET @IsCustomer = 1



	-- CHECK IS SUPPLIER

	declare @IsSupplier bit

	SET @IsSupplier = 0

	IF EXISTS(SELECT CpID From Counterparty WHERE CpID = @CpID and [Role] in ('S', 'A', 'P', 'V'))

		SET @IsSupplier = 1



	-- CHECK IS FREIGHT/OTHER

	declare @IsFreightOther bit

	SET @IsFreightOther = 0

	IF EXISTS(SELECT CpID From Counterparty WHERE CpID = @CpID and [Role] in ('D', 'A', 'V', 'X'))

		SET @IsFreightOther = 1



-- CHECK IS Intercompany

	declare @IsIntercompany bit

	SELECT @IsIntercompany = IsIntercompany FROM Counterparty WHERE CpID = @CpID



	---- CHECK IS WAREHOUSE 

	--declare @IsWarehouse bit

	SELECT @IsCustomer As IsCustomer, @IsSupplier As IsSupplier, @IsFreightOther As IsExpense, 0 As IsWarehouse, @IsIntercompany As IsIntercompany





	SELECT TOP 1 * FROM Contact WHERE CpID = @CpID AND IsNull(IsPrimary,0) = 1 ORDER BY ContactNm DESC



	

	SELECT 

		Certification.CertificationID, 

		RegistrationNo, 

		Certification.CertificationType

	FROM Counterparty_Certification 

	INNER JOIN Certification WITH(NOLOCK) ON Counterparty_Certification.CertificationTypeID = Certification.CertificationID

	WHERE CpID = @CpID



	SELECT CounterpartyGradeCodes.GradeID, [Description] AS [ProductNm], EDICode, EDIRef FROM CounterpartyGradeCodes 

	INNER JOIN Grade WITH(NOLOCK) ON CounterpartyGradeCodes.GradeID = Grade.GradeID

	WHERE CpID = @CpID



	-- PRODUCTS OF INTEREST

    SELECT ItemID, Description,a.GRADEID,a.UnitType,a.BW,a.BWUOM,a.COLOR,a.FINISH,a.RW,a.RWUOM,a.Core,a.CoreUOM,a.Caliper,a.SheetSIZE,a.SheetSizeUOM,

           a.MWeight,a.Grain,a.JobLot,a.Condition,a.Spec,a.Notes,a.PriceIdea,a.PriceIdeaUOM,a.PriceIdeaCurrency, a.SpeciesID, a.SKU, AltDesc,

           InterestType,StartDt,ExpDt,Qty,QtyUOM, PPrice, SupplierID

    FROM Counterparty_ProductsOfInterest a WITH(NOLOCK) 

    INNER JOIN GRADE WITH(NOLOCK) ON a.GRADEID = GRADE.GRADEID

    WHERE CPID = @CpID



	-- CHECK IS CUSTOMTER

	declare @IsCustomerUse bit

	SET @IsCustomerUse = 0

	IF EXISTS(SELECT Top 1 CustID FROM BuySell WITH(NOLOCK) WHERE BuySell.CustID = @CpID)

		SET @IsCustomerUse = 1

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 CpID FROM PO WITH(NOLOCK) WHERE PO.CpID = @CpID AND Source = 'SO')

			SET @IsCustomerUse = 1

	END

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 Account FROM ServiceRecord WITH(NOLOCK) WHERE Account = @CpID)

			SET @IsCustomerUse = 1

	END

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 CustID FROM BillSheet WITH(NOLOCK) WHERE CustID = @CpID)

			SET @IsCustomerUse = 1

	END

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 AdjCpID FROM GPLedger WITH(NOLOCK) WHERE AdjCpID = @CpID And AdjType = 'S')

			SET @IsCustomerUse = 1

	END

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 AccountID FROM DispatchJob WITH(NOLOCK) WHERE AccountID = @CpID)

			SET @IsCustomerUse = 1

	END

	IF @IsCustomerUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 Customer FROM Equipment_Location WITH(NOLOCK) WHERE Customer = @CpID)

			SET @IsCustomerUse = 1

	END



	-- CHECK IS SUPPLIER

	declare @IsSupplierUse bit

	SET @IsSupplierUse = 0

	IF EXISTS(SELECT Top 1 VendorID FROM BuySell WITH(NOLOCK) WHERE BuySell.VendorID = @CpID)

		SET @IsSupplierUse = 1

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 CpID FROM PO WITH(NOLOCK) WHERE PO.CpID = @CpID AND Source = 'PO')

			SET @IsSupplierUse = 1

	END

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 CpID FROM Budget WITH(NOLOCK) WHERE Budget.CpID = @CpID)

			SET @IsSupplierUse = 1

	END

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 AdjCpID FROM GPLedger WITH(NOLOCK) WHERE AdjCpID = @CpID And AdjType = 'P')

			SET @IsSupplierUse = 1

	END

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 CarrierID FROM DispatchJob WITH(NOLOCK) WHERE CarrierID = @CpID)

			SET @IsSupplierUse = 1

	END

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 FrtCarrierID FROM ServiceRecord WITH(NOLOCK) WHERE FrtCarrierID = @CpID)

			SET @IsSupplierUse = 1

	END

	IF @IsSupplierUse = 0 BEGIN

		IF EXISTS(SELECT Top 1 Receiver FROM ServiceRecord WITH(NOLOCK) WHERE Receiver = @CpID)

			SET @IsSupplierUse = 1

	END



	-- CHECK IS SERVICE

	declare @IsServiceUse bit

	SET @IsServiceUse = 0

	IF EXISTS(SELECT Top 1 VendorID FROM wksDelivery WITH(NOLOCK) WHERE wksDelivery.VendorID = @CpID)

		SET @IsServiceUse = 1



	-- CHECK IS FREIGHT

	declare @IsFreightUse bit

	SET @IsFreightUse = 0

	IF EXISTS(SELECT Top 1 VendorID FROM wksDelivery WITH(NOLOCK) WHERE wksDelivery.VendorID = @CpID AND IsNull(WksDelivery.isFreight,0) = 1)

		SET @IsFreightUse = 1



	SELECT @IsCustomerUse As CustomerInUse, @IsSupplierUse As SupplierInUse, @IsServiceUse As ServiceInUse, @IsFreightUse As FreightInUse
