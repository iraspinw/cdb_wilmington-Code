CREATE   PROCEDURE [dbo].[cw_getbookinglist] 

	@BookingNo varchar(25), 

	@CutoffFromDt datetime, 

	@CutoffToDt datetime,

	@OceanCarrier char(12),

	@CustomerNm	varchar(50),

	@Delivery varchar(25) = '',

	@POL varchar(25),

	@POD varchar(25),

	@DaterangeFlag int = 0,

	@RateType int = 0, -- 1-RATE UNASSIGNED, 2-FREIGHT RATE,  3-OCEAN FREIGHT RATE

	@PlaceOfReceipt varchar(25)

AS

--1/19/2005: dsh Added filter for Documents sent.

--02/07/2005 - removed superfluous left join 'to userdef 

--12/29/2005 DSH: Added time to cutoff date range of 11:59PM

--9/26/2006 JW: Added PlaceOfDeliv

--10/01/2007 MM: Added PlaceOfDeliv to SELECT

--10/08/2008 RS: DocCutOffDate added

--09/17/2009 MCM: Added GradeDesc

--02/26/2010 BF: added/removed fields for udom changes

--04/01/2010 RS: RcvdConfirmDt added

--06/17/2010 DSH Added Column for last invoice no.

--11/22/2010 RS: POReference added per IFP

--04/13/2012 RS: FeederVessel column added

--05/08/2012 MCM: Changed LastInvoiceNo to only return when PONumber has a value

--05/09/2012 RS: ContainerRef added 

--08/27/2012 RS: @SONo filter fixed; (SELECT POReference FROM PO...) changed to (SELECT TOP 1 POReference FROM PO)

--02/14/2013 DSH Added earliest return date. (MEC:  Added default setting to 'Y')

--04/05/2013 RS: @PODCountryID,@POLCountryID added

--08/19/2013 DSH Added Shippers Ref.

--08/26/2013 RS: LastInvShipDt added per IFP

--12/19/2013 RS: @FFID added per 3Stone

--04/09/2014 RS: @Vessel added per KC

--07/11/2014 RS: CutOffDate and DocCutOffDate changed to include the time

--01/26/2015 DSH Updated to remove date convert function calls hardcoded to MDY

--02/03/2015 EMR: Added status

--03/20/2015 RS: Added Voyage

-- Copied for cieTrade.net from q_GetBookingList

--12/03/2015 DSH Revised.

--11/22/2019 Alex Added @RateType.

--12/2/2019 Alex Added DocCutOffDate.

--03/5/2020 Alex: Added GradeDesc

--04/14/2020 Alex: @AES_instance

--05/08/2020 DSH: added Suppliers Column

--05/27/2020 Alex: Added Status

--08/21/2020 DSH Corrected Supplier Column to show suppliers from allocated bookings

--01/30/2023 Nick: Added DateRangeFlag of 4 for BookDate Filtering



SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

SET NOCOUNT ON



	declare @AES_instance  int



	select @AES_instance=count(* )

	from BookSheet 

	inner join BuySell on BuySell.BookingNo = BookSheet.BookingNo

	where BookSheet.BookingNo = @BookingNo and isnull(rtrim(ltrim(BuySell.AesReturnedCode)),'') <>''









	CREATE TABLE #BookingList(

		BookSheetID INT,

		BookingNo CHAR(25),

		CutOffDate DATETIME,

		CutoffTime VARCHAR(25),

		EarliestReturnDt DATETIME,

		OceanCarrier VARCHAR(100),

		LoadPort VARCHAR(25),

		DestPort VARCHAR(25),

		ETA DATETIME,

		ETD DATETIME,

		ContainersBooked INT DEFAULT 0,

		ContainersShipped INT DEFAULT 0,

		ContainersOpen INT DEFAULT 0,

		ContainersReserved INT DEFAULT 0,

		Vessel VARCHAR(30),

		VoyageNo VARCHAR(15),

		OrderNo CHAR(15),

		Customer VARCHAR(100),

		BookingDate DATETIME,

		BookingDeptStatus CHAR(12),

		ContainerRef varchar(100),

		FreightFwdr varchar(65),

		NotifyParty text,

		DocCutOffDate DATETIME,

		GradeDesc varchar(max),

		Notes varchar(4000),

		ShipperRef varchar(100),

		PlaceOfDeliv varchar(25),

		ReceiptPlace varchar(25),

		Suppliers VARCHAR(1200),

		UserDefined1 varchar(50),

		Rate varchar(30),

		FreightRateUOM varchar(15)

	)

	



	INSERT #BookingList

	SELECT  

		BookSheet.BookSheetID, 

		BookSheet.BookingNo, 

		CutOffDate,

		cast(DATEPART(hour, CutOffDate) as varchar) + ':' + RIGHT('0' + cast(DATEPART(minute, CutOffDate) as varchar),2),

		EarliestReturnDt,

		SSLine.CompanyNm as OceanCarrier, 	

		UPPER(LoadPort), 

		UPPER(PortOfDischg), 

		DATEADD(dd, DATEDIFF(dd, 0, ETA), 0)  as ETA, 

		DATEADD(dd, DATEDIFF(dd, 0, ETS),0) as ETD, 

		ContainerQty ContainersBooked, 

		(SELECT COUNT(Distinct BuySellNo) FROM Buysell WHERE Buysell.BookingNo = BookSheet.BookingNo and VendorID <> '(OTHER)' AND (CustID = '(INV)' OR (IsNull(RefEquipNo,'') <> '' AND IsNull(RefSealNo,'') <> ''))) ContainersShipped,

		0 ContainersOpen, 

		ISNULL((SELECT SUM(POBooking.ContainerQty) FROM BookSheet BB WITH(NOLOCK) inner join POBooking WITH(NOLOCK) ON BB.BookSheetID = POBooking.BookSheetID INNER JOIN PO WITH(NOLOCK) on PO.POID = POBooking.POID WHERE Booksheet.BookSheetID = BB.BookSheetID AND


 PO.Source = 'PO'), 0) ContainersReserved,

		Vessel,

		VoyageNo,

		BookSheet.PONumber OrderNo, 

		CustomerCO.CompanyNm Customer,

		BookDate,

		BookingDeptStatus,

		ContainerRef,

		Fwdr.CompanyNm FreightFwdr,

		SalesPO.BLNotifyParty,

		BookSheet.DocCutOffDate,

		BookSheet.GradeDesc,

		BookSheet.UserNotes,

		BookSheet.ShipperRef,

		BookSheet.PlaceOfDeliv,

		BookSheet.ReceiptPlace,

		(SELECT UPPER(STUFF((  		

			SELECT  ','+RTRIM(CompanyNm) FROM ( 		

				SELECT DISTINCT C.CompanyNm FROM Counterparty C INNER JOIN PO ON C.CPID = PO.CPID INNER JOIN POBooking ON POBooking.POID = PO.POID WHERE POBooking.BookSheetID = BookSheet.BookSheetID

			)

		Z FOR XML PATH('')  ),1,1,''))) Suppliers,

		Booksheet.UserDefined1,

		Rate,

		FreightRateUOM

	FROM BookSheet WITH(NOLOCK) 

	LEFT JOIN Counterparty SSLine WITH(NOLOCK) on BookSheet.Carrier = SSLine.CpID

	LEFT JOIN PO as SalesPO WITH(NOLOCK) ON BookSheet.PONumber=SalesPO.PONumber

	LEFT JOIN CounterParty as CustomerCO WITH(NOLOCK) ON SalesPO.CPID=CustomerCO.CPID	

	LEFT JOIN Ports POL WITH(NOLOCK) ON POL.portnm = Booksheet.LoadPort

	LEFT JOIN Ports WITH(NOLOCK) ON Ports.portnm = Booksheet.PortOfDischg

	LEFT JOIN CounterParty Fwdr WITH(NOLOCK) ON Fwdr.CpID = BookSheet.FreightFwdr

	WHERE 

		(@BookingNo = '' or BookSheet.BookingNo like @BookingNo + '%')		

		AND (

		      (@BookingNo <> '' or (CutOffDate = '' OR CutOffDate >= @CutoffFromDt and  (@CutoffToDt = '' OR CutOffDate <= @CutoffToDt + '11:59 PM'))) and @DaterangeFlag=0

			  or

			  (@BookingNo <> '' or (CutOffDate = '' OR ETA >= @CutoffFromDt and  (@CutoffToDt = '' OR ETA <= @CutoffToDt + '11:59 PM'))) and @DaterangeFlag=1

			  or

			  (@BookingNo <> '' or (CutOffDate = '' OR ETS >= @CutoffFromDt and  (@CutoffToDt = '' OR ETS <= @CutoffToDt + '11:59 PM'))) and @DaterangeFlag=2

			   or

			  (@BookingNo <> '' or (CutOffDate = '' OR DocCutOffDate >= @CutoffFromDt and  (@CutoffToDt = '' OR DocCutOffDate <= @CutoffToDt + '11:59 PM'))) and @DaterangeFlag=3

			  or

			  (@BookingNo <> '' or (CutOffDate = '' OR BookDate >= @CutoffFromDt and  (@CutoffToDt = '' OR BookDate <= @CutoffToDt + '11:59 PM'))) and @DaterangeFlag=4

			 )

		AND (@OceanCarrier = '' OR (BookSheet.Carrier = @OceanCarrier))

		AND (@CustomerNm = '' OR (@CustomerNm = CustomerCO.CompanyNm))

		AND (@POL = '' OR (@POL = LoadPort))

		AND (@POD = '' OR (@POD = PortOfDischg))

		AND (@PlaceOfReceipt = '' OR (@PlaceOfReceipt = BookSheet.ReceiptPlace))

		AND (@Delivery = '' OR (@Delivery = BookSheet.PlaceOfDeliv))

		AND (     @RateType = 0 

		      OR 

			      (

			       (@RateType = 1 and isnull(BookSheet.FreightRateID,'')='' and isnull(BookSheet.Rate,'')='') or (@RateType = 2   and BookSheet.FreightRateID=0 and isnull(BookSheet.Rate,'')<>'' ) or (@RateType = 3 and BookSheet.FreightRateID<>0 and isnull(BookShee
t.Rate,'')<>'')

				  )

			)









	

	UPDATE #BookingList SET ContainersOpen = ContainersBooked-ContainersShipped

	UPDATE #BookingList SET ETA = NULL WHERE ETA < '1/1/1980'

	UPDATE #BookingList SET ETD = NULL WHERE ETD < '1/1/1980'





	--get list of ponumbers for booking

	CREATE TABLE #ponumbers(

		BookingNo CHAR(25),

		PONos varchar(500)		

	)



	insert #ponumbers

	select 

	   distinct #BookingList.BookingNo,

		stuff((

			select ',' + RTrim(IsNull(po.PONumber,''))

			from BookSheet bs

			inner join POBooking WITH(NOLOCK) ON bs.BookSheetID = POBooking.BookSheetID

			INNER JOIN PO WITH(NOLOCK) on PO.POID = POBooking.POID

			where bs.BookingNo = #BookingList.BookingNo

			order by bs.BookingNo

			for xml path('')

		),1,1,'') as PONos



	FROM #BookingList WITH(NOLOCK)

	group by #BookingList.BookingNo





	

	-- FINAL LIST

	SELECT 

		BookSheetID [-Hidden-BookSheetID],

		#BookingList.BookingNo As [{C}Booking No],

		CutOffDate As [Cutoff Date],

		CutoffTime AS [Cutoff Time],

		EarliestReturnDt [ERD],

		OceanCarrier As [Ocean Carrier],

		LoadPort As [POL],

		DestPort As [POD],

		ETA,

		ETD,

		NotifyParty [Notify Party],

		ContainersBooked As [Containers],

		ContainersShipped As [Shipped],

		ContainersOpen As [Open],

		ContainersReserved AS [Allocated],

		Vessel As [Vessel Name],

		ContainerRef AS [Invoice No],

		VoyageNo As [Voyage No],

		OrderNo As [Order No],

		Customer As [Customer],

		BookingDate As [Booking Date],

		BookingDeptStatus as [Status],

		FreightFwdr [Forwarder],

		PONos [PO No],

		DocCutOffDate [Document Cut-Off Date],

		GradeDesc as [Description of Goods],

		Notes,

		ShipperRef AS [Shipper Reference],

		PlaceOfDeliv AS [Place of Delivery],

		ReceiptPlace AS [Place of Receipt],

		case when @AES_instance>0 then 'Yes' else 'No' end as AES,

		Suppliers,

		UserDefined1,

		Rate AS [Freight Rate],

		--FreightRateUOM

		case when FreightRateUOM = '(' then '' else FreightRateUOM end as FreightRateUOM

	FROM #BookingList LEFT JOIN #ponumbers ON #BookingList.BookingNo = #ponumbers.BookingNo

	ORDER BY [{C}Booking No]
