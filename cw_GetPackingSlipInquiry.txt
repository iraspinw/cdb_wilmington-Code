CREATE PROC [dbo].[cw_GetPackingSlipInquiry] 

	@BuySellNo char(12), 

	@FromDt datetime, 

	@ToDt Datetime,

	@WhseStatus varchar(300) = '', 

	@WarehouseID CHAR(45) = '',

	@EngOrSp char(5) = 'en-US'

AS



--03/31/2017 MEC: Created

--08/18/2022 Nick: EquipmentNo now accepts up to 30 chars (same as buysell table)

-- cw_GetPackingSlipInquiry '', '', '', ''



/*

Statuses:

	SCHEDULED

	OPEN

	CLOSED

*/





SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;





	CREATE TABLE #PackingSlips(

		PackingSlipNo CHAR(12),

		ShippingDt DATE,

		Customer VARCHAR(100),

		DeliveryTo CHAR(200),

		OrderNo CHAR(15),

		FreightCarrier VARCHAR(100),

		EquipmentNo VARCHAR(30),

		ShipFromWarehouse VARCHAR(100),

		SalesRep VARCHAR(50),

		WarehouseStatus VARCHAR(10)

	)





	INSERT #PackingSlips	

	SELECT		

		B.BuySellNo, 

		ShippingDt,

		UPPER(C.CompanyNm),

		UPPER(A.City)+', '+UPPER(A.Region)+'  ['+UPPER(RTRIM(B.ShipToID))+']',

		B.RefSONo,

		ISNULL( 

			CASE WHEN B.RefFreightCarrier = '(OTHER)' THEN 

			(SELECT TOP 1 InvoiceDesc From WksDelivery Where BuySellNo = B.BuySellNo And VendorID = '(OTHER)') 

			ELSE 

			(SELECT CompanyNm FROM Counterparty WHERE CPID = B.RefFreightCarrier)

			END,

			''),

		B.RefEquipNo As [Equipment Number],

		B.ShipFromID,

		UPPER(R.[Name]),

		B.WarehouseStatus

	

	FROM BuySell B

	INNER JOIN Counterparty C ON B.CustID = C.CpID  

	INNER JOIN Address A ON B.ShipToID = A.Type AND B.CustID = A.CPID

	LEFT JOIN WksPackingSlip ON B.BuySellNo = WksPackingSlip.BuySellNo	

	LEFT JOIN Referrer R ON B.ReferrerNo1 = R.ReferrerNo

	

	

	WHERE 

		( (@WhseStatus = '' AND IsNull(B.WarehouseStatus,'') <> '' ) OR (@WhseStatus <> '' AND @WhseStatus = B.WarehouseStatus) )

		AND ( (ISNULL(@FromDt, '') = '' OR B.ShippingDt >= @FromDt) AND (ISNULL(@ToDt,'') = '' OR B.ShippingDt <= @ToDt) )

		AND (@WarehouseID= '' OR @WarehouseID = B.ShipFromID)

		AND (B.Status <> 'X' AND B.Status <> 'I')

		AND (Upper(VendorID) = '(INV)')

	

--LEFT JOIN Inventory WITH(NOLOCK) ON WksPackingSlip.LotNo = Inventory.LotNo







	IF(@EngOrSp = 'en-US')

		BEGIN

			SELECT DISTINCT

				PackingSlipNo As [Packing-Slip#],

				ShippingDt As [Shipping-Date],

				Customer,

				DeliveryTo As [Delivery-To],

				FreightCarrier As Carrier,

				EquipmentNo As [Equipment No],

				ShipFromWarehouse As [Ship From Whse],

				SalesRep As [Sales Rep],

				WarehouseStatus As [Warehouse-Status],

				OrderNo AS [Order No]

			FROM #PackingSlips Order By ShippingDt

		END

		IF(@EngOrSp = 'es')

		BEGIN

			SELECT DISTINCT

				PackingSlipNo As [Hoja del Embarque],

				ShippingDt As [Fecha del Embarque],

				Customer As Cliente,

				DeliveryTo As [Entrega A],

				FreightCarrier As [Transportista],

				EquipmentNo As [Equipment No],

				ShipFromWarehouse As [Envi¾ Desde],

				SalesRep As [Representante],

				WarehouseStatus As [Estatus de la Bodega],

				OrderNo AS [N·mero de Orden]

			FROM #PackingSlips Order By ShippingDt

		END
