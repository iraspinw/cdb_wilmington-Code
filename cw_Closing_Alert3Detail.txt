CREATE PROC [dbo].[cw_Closing_Alert3Detail] @DoColumnDefOnly BIT, @FromDt datetime,@ToDT datetime,@COID char(2), @UsePostingDt BIT

AS

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

--Expenses Against Unposted Sales Details

--09/24/2012 DSH Created for Closing Adjustments Ledger.

--03/09/2020 Alex Added  CustID,VendorID



    -- GET ACCOUNTING SYSTEM.

    Declare @AccountingSystem varchar(30)

    SELECT TOP 1 @AccountingSystem = AccountingSystem from AccountingCompany WHERE AccountingSystem IS NOT NULL AND AccountingSystem <> '(NONE)'





	/* DEFINE REPORT COLUMNS SECTION. */

	IF @DoColumnDefOnly = 1 BEGIN

	

		CREATE TABLE #ReportDef(

			ColName char(25),

			ColAlign char(1),

			ColWidth int,

			ColDataType char(10),

			ColNumFormat char(15)

		)

		

		INSERT #ReportDef SELECT 'Invoice No','L',1250,'TEXT',''

		INSERT #ReportDef SELECT 'Worksheet No.','L',1250,'TEXT',''

		INSERT #ReportDef SELECT 'Batch No.','L',1250,'TEXT',''

		IF @UsePostingDt = 1

			INSERT #ReportDef SELECT 'Received Date','L',1250,'DATE',''

		ELSE

			INSERT #ReportDef SELECT 'Invoice Date','L',1250,'DATE',''		

		INSERT #ReportDef SELECT 'Vendor','L',3100,'TEXT',''

		INSERT #ReportDef SELECT 'Source','L',800,'TEXT',''

		INSERT #ReportDef SELECT 'User','L',1000,'TEXT',''

		INSERT #ReportDef SELECT 'Amount','R',1300,'NUM','#,###,###0.00'		

		

		SELECT ColName, ColAlign, ColWidth, ColDataType, ColNumFormat FROM #ReportDef

		RETURN

	END	

	

	

	CREATE TABLE #Report( 	

		InvoiceNo CHAR(20), 

		BuySellNo CHAR(12),

		APBatchNo CHAR(15),		

		TransDt DATETIME,  		

		CompanyNm VARCHAR(65),

		ItemIDSource CHAR(1), 

		UserID CHAR(30),  

		Amount MONEY,

		Seq Int,

		[-Hidden-CustID] char(12),

		[-Hidden-VendorID] char(12)

	)

	

	

	

	/* PRODUCE DETAIL REPORT RESULTS BY POSTING DATE. */

	IF @UsePostingDt = 1 BEGIN

		INSERT #Report

		SELECT 

			InvoiceNo, 

			Payables.BuySellNo,

			payables.APBatchNo,		

			Payables.InvoiceDt,  		

			Counterparty.CompanyNm,

			ItemIDSource, 

			UserID, 

			Amount,1 ,

			CustID,

			VendorID

		FROM Payables WITH(NOLOCK) 

		INNER JOIN buysell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo

		LEFT JOIN Counterparty ON Payables.CpID = Counterparty.CpID

		WHERE 

		payables.itemidsource<>'A' AND 

		Payables.RcvDt <= @ToDT 	

		AND Payables.Posted = 1

		AND BuySell.Status <> 'I'

		AND (@COID = '' OR @COID = BuySell.COID) 

		ORDER BY PAYABLES.InvoiceDt

	END	





	/* PRODUCE REPORT DETAIL BY INVOICE DATE. */

	IF @UsePostingDt = 0 BEGIN

		INSERT #Report

		SELECT 

			InvoiceNo, 

			Payables.BuySellNo,

			payables.APBatchNo,		

			Payables.InvoiceDt,  		

			Counterparty.CompanyNm,

			ItemIDSource, 

		UserID,  

			Amount,1 ,

			CustID,

			VendorID

		FROM Payables WITH(NOLOCK) 

		INNER JOIN buysell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo	

		LEFT JOIN Counterparty ON Payables.CpID = Counterparty.CpID

		WHERE 

		payables.itemidsource<>'A' AND 

		Payables.InvoiceDt <= @ToDT 		

		AND Payables.Posted = 1		

		AND BuySell.Status <> 'I'

		AND (@COID = '' OR @COID = BuySell.COID) 

		ORDER BY PAYABLES.InvoiceDt		

	END	





	--

	INSERT #REPORT(InvoiceNo,Amount,Seq)			

		SELECT 

			'TOTAL', 

			SUM(Amount),2

		FROM #Report 

	

	

	--SELECT * FROM #Report Order by Seq, TransDt

		

		SELECT 

		InvoiceNo [Invoice No], 

		BuySellNo [Worksheet No],

		APBatchNo [Batch No],		

		TransDt [Invoice Date],  		

		CompanyNm [Vendor],

		ItemIDSource [Source], 

		--UserID [User], 

		UserDef.UserNm  [User],

		Amount ,

		Seq ,

		[-Hidden-CustID],

		[-Hidden-VendorID] 



		FROM #Report

		left join UserDef on UserDef.UserID=#Report.USERID

		 Order by Seq, TransDt
