



CREATE   PROCEDURE [dbo].[cw_ClosingAdjLedger_Default2_For_Copilot] 

@dt1 as datetime = '01/01/2026', 

@dt2 as datetime = '03/31/2026', 

@CoID as char(2) = '',  

@UsePostingDt BIT = 0, 

@DoColumnDefOnly bit = 0, 

@DetailReportTag VARCHAR(100) = '' 



AS  

SET NOCOUNT ON

SET ANSI_WARNINGS OFF

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED



    CREATE TABLE #PayableTrueUp(

		PayableID int,

		BuySellNo char(12),

		ActualAmt Money,

		CalcAmt Money,

		Variance Money				

	)        

	

	   CREATE TABLE #PayablesToAlloc(

        BuySellNo char(15),

        PayableID int,

        TradeType char(20),

        PayableAmt money              

    )

    CREATE INDEX #IE1_PayablesToAlloc ON #PayablesToAlloc(BuySellNo)



    CREATE TABLE #PayableProducts(

        ItemID int identity(1000,1),

        PayableID int,        

        BuySellNo char(12),        

        Seq INT,

        TradeType char(20),

        GradeID char(15),

        GLAccountNm char(30),

        GLAcct char(12),

        Cost DECIMAL(22,12),

        WksCost Money,

        PayableAmt Money,

        RptCost Money        

    )     

    

        CREATE TABLE #PayableWksCost(

		PayableID int,

		BuySellNo char(12),

		WksCost Money

	)    

	

		CREATE TABLE #ICTrades(

			ICType CHAR(30),

			GLAccountNm CHAR(30),

			GLAcctNo CHAR(12),

			Amount MONEY

		)

	



    -- GET FUNCTIONAL CURRENCY

  Declare @SysCurrency char(3)

    SELECT @SysCurrency = CurrCode from systeminfo





    

    -- GET ACCOUNTING SYSTEM.

    Declare @AccountingSystem varchar(30)

    SELECT TOP 1 @AccountingSystem = AccountingSystem from AccountingCompany WHERE AccountingSystem IS NOT NULL AND AccountingSystem <> '(NONE)'



	/* DEFINE REPORT COLUMNS SECTION. */

	IF @DoColumnDefOnly = 1 BEGIN

		CREATE TABLE #ReportDef(

			ColName char(25),

			ColAlign char(1),

			ColWidth int,

			ColDataType char(10),

			ColNumFormat char(15)

		)

		

		IF @DetailReportTag = '' BEGIN

			INSERT #ReportDef SELECT 'Class','L',1350,'TEXT',''

			INSERT #ReportDef SELECT 'Group','L',1350,'TEXT',''

			INSERT #ReportDef SELECT 'Account Name','L',4100,'TEXT',''

			INSERT #ReportDef SELECT 'GL Acct','L',1100,'TEXT',''

			INSERT #ReportDef SELECT 'Segment','L',1100,'TEXT',''

			INSERT #ReportDef SELECT 'Debit','R',1500,'NUM','#,###,##0.00'

			INSERT #ReportDef SELECT 'Credit','R',1500,'NUM','#,###,##0.00'

			

			SELECT ColName, ColAlign, ColWidth, ColDataType, ColNumFormat FROM #ReportDef

			RETURN

		END		

		

		

		--Default Drill Down Report. 

		IF @DetailReportTag <> '' BEGIN

			INSERT #ReportDef SELECT 'Worksheet No.','L',1500,'TEXT',''

			INSERT #ReportDef SELECT 'Posting Date','L',1500,'TEXT',''

			INSERT #ReportDef SELECT 'Description','L',3200,'TEXT',''

			INSERT #ReportDef SELECT 'Acct No.','L',1100,'TEXT',''									

			INSERT #ReportDef SELECT 'Amount','R',1500,'NUM','#0.00'			

			

			SELECT ColName, ColAlign, ColWidth, ColDataType, ColNumFormat FROM #ReportDef

			RETURN

		END		

	END	

		

	DECLARE @SalesAcct char(12)

	DECLARE @SalesDiscAcct char(12)

	DECLARE @ARAcct char(12)

	DECLARE @ARAccrualAC char(12)

	DECLARE @APAcct char(12)

	DECLARE @PurchAcct char(12)

	DECLARE @Inventory char(12)

	DECLARE @InventoryPurchase char(12)

	DECLARE @BankFeeAc char(12)

	DECLARE @InterestFeeAc char(12)

	DECLARE @UnappledCashAc char(12)

	DECLARE @APAccrualAC char(12)

	DECLARE @CommExpAcct char(12)	

	DECLARE @OffsetAcct  char(12)	

	

	

	SELECT TOP 1 @SalesAcct = GLAcct from glaccount where cat = 'SALES'

	SELECT TOP 1 @SalesDiscAcct = GLAcct from glaccount where cat = 'SALES DISCOUNT'

	SELECT TOP 1 @ARAcct = GLAcct from glaccount where cat = 'A/R'

	SELECT TOP 1 @APAcct = GLAcct from glaccount where cat = 'A/P'

	SELECT TOP 1 @PurchAcct = GLAcct from glaccount where cat = 'PURCHASES'

	SELECT TOP 1 @CommExpAcct = GLAcct from glaccount where cat = 'COMM EXPENSE'

	SELECT TOP 1 @Inventory = GLAcct from glaccount where cat = 'INVENTORY'

	SELECT TOP 1 @InventoryPurchase = GLAcct from glaccount where cat = 'INVENTORY TRADE'

	SELECT TOP 1 @BankFeeAc = GLAcct from glaccount where cat = 'BANK FEES'

	SELECT TOP 1 @InterestFeeAc = GLAcct from glaccount where cat = 'INTEREST CHARGE'

	SELECT TOP 1 @UnappledCashAc= GLAcct from glaccount where cat = 'UNAPPLIED CASH'	

	SELECT TOP 1 @APAccrualAC= GLAcct from glaccount where cat = 'A/P ACCRUAL'	

	SELECT TOP 1 @ARAccrualAC= GLAcct from glaccount where cat = 'A/R ACCRUAL'		

	SELECT TOP 1 @OffsetAcct = GLAcct from glaccount where cat = 'OFFSET'	 

	IF @OffsetAcct = '' SET @OffsetAcct = '*OFFSET'

	



	--CHECK FOR A/R MODULE

	DECLARE @IsARModule BIT

	DECLARE @AROffsetAc CHAR(12)

	DECLARE @AROffsetAcNm CHAR(45)

	SET @IsARModule = 0

	SELECT @IsARModule = ISNULL(EnableCieTradeAR,0) FROM systeminfo WHERE ID = '00'

	IF @IsARModule = 0 BEGIN

		SET @AROffsetAc = @ARAccrualAC

		SET @AROffsetAcNm = 'Accrued Sales'

	END

	ELSE BEGIN

		SET @AROffsetAc = @ARAcct

		SET @AROffsetAcNm = 'Accounts Receivable'

	END





	IF @InventoryPurchase IS NULL SET @InventoryPurchase=@PurchAcct



    CREATE TABLE #LedgerDetail(

		SectionSeq int,

		AcctClass Char(15),

		AcctGrp Char(15),

		AcctGrpOrd int, 								

		TradeType varchar(25),

		AcctNo varchar(20), 

		AcctNm varchar(65),		

		AltAcctNm varchar(65),

		TranNo varchar(20),

		TranDt datetime,		

		DbCr char(2),

		DebitAmt Money,

		CreditAmt Money,

		HCAmount Money,

		Reference varchar(20),

		IsTotalLn BIT,

		ActionTag VARCHAR(130),

		IsOffsetTx SMALLINT DEFAULT 0,

		ExpItemID INT,

		TranFormType CHAR(10),

		WksNo CHAR(12),

		IsIntercompany CHAR(1),

		IntercoAcct CHAR(12),

		[COID] [char](2),

		[Period] char(7)

	)



	--CREATE INDEX #IE_LedgerDetail ON #LedgerDetail(CurCode, AcctGrp, AcctNm, AcctNo, Dept)



    CREATE TABLE #LedgerDrillDown(		

		TransNo varchar(20),

		TranType CHAR(20),

		TransDt datetime,

		AcctNm varchar(45),

		AcctNo varchar(20), 		

		HCAmount Money,

		TranFormType CHAR(10),

		Customer varchar(100),

		Supplier varchar(100),

		[COID] [char](2),

		[Period] char(7)

	)











	/* A/R SECTION [q_Closing_1a] */  

	----------------------------------------------------------------------------------------=    

	

	

	----Accounts Receivable.	

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [period]) 

    SELECT 

		1,'AR','SALES',1,			

		COALESCE(BuySell.TradeType,''),			

		@AROffsetAc,			--@ARAccrualAC, --@ARAcct, 

		@AROffsetAcNm,			--'Accrued Sales', 

		SUM(round(SalesAmt+SFreightChg+COALESCE(SalesTaxAmt,0),2)),'TOTAL',

		buysell.coid,

		FORMAT(BuySell.InvoiceDt, 'MM/yyyy') [period]

	FROM BuySell WITH(NOLOCK) 

    WHERE InvoiceDt >= @dt1 and InvoiceDt <= @dt2 

		AND (CustID <> '(INV)'  AND CustID <> '{C}' AND CustID <> '(PLANT)') 

		AND Status = 'I'  and (IsAgency IS NUlL OR IsAgency <> 1)

		AND (  (@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )

	Group By buysell.coid,FORMAT(InvoiceDt, 'MM/yyyy'), BuySell.TradeType  --, buysell.CustID, buysell.VendorID

	order by buysell.coid,FORMAT(InvoiceDt, 'MM/yyyy'),  BuySell.TradeType	



	

	--Product Sales.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

	SELECT 

		1,'AR','SALES',2,			

		COALESCE(BuySell.TradeType,''),			

		COALESCE(GLAccount.GLAcct,@SalesAcct),   

		COALESCE(GLAccount.Name,'Sales (Default)'), 

		BuySell.BuySellNo,

		BuySell.InvoiceDt,

		Round(SAmount*-1,2),

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier,

		buysell.coid,

		FORMAT(BuySell.InvoiceDt, 'MM/yyyy') [period]

	FROM BuySell WITH(NOLOCK) 

		INNER JOIN wksDetail WITH(NOLOCK) ON BuySell.BuySellNo = wksDetail.BuySellNo		

		INNER JOIN Grade WITH(NOLOCK) ON Grade.GradeID = wksDetail.GradeID 

		LEFT JOIN GLAccount WITH(NOLOCK) ON GLAccount.GLAcct = Grade.GLAcct

		LEFT JOIN Counterparty C ON C.CPID = Buysell.CustID

	WHERE BuySell.InvoiceDt >= @dt1 and BuySell.InvoiceDt <= @dt2 and (BuySell.CustID <> '(INV)' AND BuySell.CustID <> '{C}'  AND CustID <> '(PLANT)')

		And Status = 'I'  and (IsAgency IS NUlL OR IsAgency <> 1)

		AND ((@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )



	

	--Sales Chargebacks.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,AltAcctNm,HCAmount,IsOffsetTx,ExpItemID,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

    SELECT 

		1,'AR','SALES',2,  --3	

		COALESCE(BuySell.TradeType,''),	

		WksDelivery.SalesGLAcct,

		ISNULL('Sales-' + SalesGL.Name,'(SALES-CHGBACK)'),		

		BuySell.BuySellNo,

		BuySell.InvoiceDt,

		SalesGL.Name,		

		round(ChgToCustAmt*-1,2),

		CASE WHEN UPPER(BuySell.VendorID) = '(OTHER)' AND WksDelivery.Invoicedesc = 'OFFSET' THEN -1 ELSE 0 END IsOffsetTx,

		WksDelivery.ItemID,

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier,

		buysell.coid,

		FORMAT(BuySell.InvoiceDt, 'MM/yyyy') [period]

    FROM WksDelivery WITH(NOLOCK) 

		INNER JOIN BuySell WITH(NOLOCK) On WKSDelivery.BuySellNo = BuySell.BuySellNo      

        LEFT JOIN GLAccount SalesGL WITH(NOLOCK) ON WKSDelivery.Salesglacct = SalesGL.glacct

		LEFT JOIN Counterparty C ON C.CPID = Buysell.CustID

    WHERE InvoiceDt >= @dt1 and InvoiceDt <= @dt2 

        AND (CustID <> '(INV)'  AND CustID <> '{C}'  AND CustID <> '(PLANT)')  

        AND Status = 'I'  

		and (IsAgency IS NUlL OR IsAgency <> 1)

        AND ChgToCustAmt <> 0

        AND ((@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')))





	UPDATE #LedgerDetail SET AcctNm = AltAcctNm WHERE AltAcctNm IS NOT NULL 



	------(Sales Taxes)

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

	SELECT 

		1,'AR','SALES',2,	--4	

        COALESCE(BuySell.TradeType,''),	

		SalesTax.SalesTaxAcct,								

		'Sales-Tax '+SalesTax.TaxDescription,   

		BuySell.BuySellNo,

		BuySell.InvoiceDt,		            

        (COALESCE(TaxAmount,0)*-1),

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier,

		buysell.COID,

		FORMAT(BuySell.InvoiceDt, 'MM/yyyy') [period]

    FROM BuySell WITH(NOLOCK)

    LEFT JOIN wksSalesTax WITH(NOLOCK)  ON wksSalesTax.BuySellNo = BuySell.BuySellNo

    LEFT JOIN SalesTax WITH(NOLOCK) ON SalesTax.SalesTaxID = wksSalesTax.SalesTaxID		

	LEFT JOIN Counterparty C ON C.CPID = Buysell.CustID    

    WHERE 

    BuySell.InvoiceDt >= @dt1 and BuySell.InvoiceDt <= @dt2 

    and (BuySell.CustID <> '(INV)' AND BuySell.CustID <> '{C}'  AND CustID <> '(PLANT)')

    AND wksSalesTax.Source = 'B'

    And Status = 'I' 

    and COALESCE(IsAgency,0) <> 1

    AND SalesTaxAmt <> 0 

    AND (  (@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )







	

	-- TAG SALES-CHARGEBACK ITEMS THAT ARE AP OFFSETS

	------------------------------------------------------

	UPDATE #LedgerDetail SET AcctNo = @OffsetAcct, AcctNm = 'Offset', #LedgerDetail.IsOffsetTx=1

	FROM #LedgerDetail INNER JOIN Payables ON #LedgerDetail.ExpItemID = Payables.ItemID 

	WHERE #LedgerDetail.IsOffsetTx = -1 AND ItemIDSource = 'D' AND COALESCE(ChgBackWks,'') <> ''

	------------------------------------------------------





	--/* NET SALES TOTALS */

	------------------------------------------------------------------------------------------=    

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ISTotalLn,ActionTag, COID, [period]) 

    SELECT 1,'AR','SALES-TOTALS',5,'','',' ',SUM((HcAmount*-1)),1,'TOTAL',COID, [period]

	    FROM #LedgerDetail WHERE AcctCLass = 'AR' AND (AcctNo <> @AROffsetAc) AND Coalesce(ISTotalLn,0) <> 1 

		GROUP BY COID, [period]





	--/* SALES CREDIT MEMOS */

	------------------------------------------------------------------------------------------=    

	--Sales Memo.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

    SELECT 

		2,'AR','AR-ADJ-CR',1,		

		BuySell.TradeType,

		GPLedger.GLAcct, 

		dbo.TitleCase(GLAccount.[Name]),		

		GPLedger.BuySellNo,

		GPLedger.TradeDt,

    	GrossProfit*-1,

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier,

		buysell.COID,

		FORMAT(GpLedger.TradeDt, 'MM/yyyy') [period]

	FROM GPLedger WITH(NOLOCK) 

	INNER JOIN Buysell WITH(NOLOCK) ON GpLedger.BuySellno = BuySell.BuySellNo 

	INNER JOIN GLAccount WITH(NOLOCK) ON GpLedger.glacct = GLaccount.GLacct  

	LEFT JOIN Counterparty C ON C.CPID = GPLedger.AdjCPID

	WHERE AdjType = 'S' 

		And GrossProfit < 0 

		and GpLedger.TradeDt >= @dt1 and GpLedger.TradeDt <= @dt2 

		AND (Source <> 'D' AND Source <> 'F' AND Source <> 'N') 

		AND AdjPosted = 'Y' 

	    AND (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00') OR @COID = '')





	---- Tax Adjustment

   INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

   SELECT 

	    2,'AR','AR-ADJ-CR',2,	

	    BuySell.TradeType,	

	    Coalesce(SalesTaxAcct,''),

	    'Sales-Tax '+COALESCE(SalesTax.TaxDescription,''),

		GPLedger.BuySellNo,

		GPLedger.TradeDt,	

		(COALESCE(TaxAmount,0)*-1),                      

	    CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END	,

		C.IsIntercompany,

		C.GLAcctSupplier,

		BuySell.COID,

		FORMAT(GPLedger.TradeDt, 'MM/yyyy') [period]

		FROM GPLedger WITH(NOLOCK) INNER JOIN Buysell WITH(NOLOCK) ON GpLedger.BuySellno = BuySell.BuySellNo 

		LEFT JOIN wksSalesTax WITH(NOLOCK) ON wksSalesTax.LedgerID = GPLedger.LedgerID		

		LEFT JOIN SalesTax WITH(NOLOCK) ON SalesTax.SalesTaxID = wksSalesTax.SalesTaxID		

		LEFT JOIN Counterparty C ON C.CPID = GPLedger.AdjCPID    

	WHERE AdjType = 'S' and TradeDt >= @dt1 and TradeDt <= @dt2 And AdjSalesTaxAmt < 0 

	AND (GPLedger.[Source] <> 'D' AND GPLedger.[Source] <> 'F' AND GPLedger.[Source] <> 'N') 

	AND AdjPosted = 'Y' 

    AND (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00') OR @COID = '')

    AND AdjSalesTaxAmt <> 0 





	----(A/R Offset Transaction)

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [period]) 

	    SELECT 2,'AR',AcctGrp,3,TradeType,@AROffsetAc,@AROffsetAcNm,SUM(HcAmount)*-1,'TOTAL',coid, [period]

	FROM #LedgerDetail WHERE AcctGrp = 'AR-ADJ-CR'

	    GROUP BY coid, [period],AcctGrp, TradeType



	----Total Sales Credits	

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ISTotalLn,ActionTag, COID, [period]) 

        SELECT 2,'AR','AR-ADJ-CR',4,'','','Totals ',SUM(HcAmount),1,'TOTAL', COID, [period]

	 FROM #LedgerDetail WHERE AcctGrp = 'AR-ADJ-CR' AND (AcctNo <> @AROffsetAc)

	 GROUP BY COID, [period]











	--/* SALES DEBIT MEMOS */

	------------------------------------------------------------------------------------------=    

	----Sales Adjustment.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, COID, [period]) 

    SELECT 

	    3,'AR','AR-ADJ-DB',2,	

		BuySell.TradeType,		    

	    GPLedger.GLAcct,

	    dbo.TitleCase(GLAccount.[Name]),	

		GPLedger.BuySellNo,

		GPLedger.TradeDt,	    		

		GrossProfit*-1,

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier, 

		buysell.coid,

		FORMAT(GpLedger.TradeDt, 'MM/yyyy') [period]

		FROM GPLedger WITH(NOLOCK) INNER JOIN Buysell WITH(NOLOCK) ON GpLedger.BuySellno = BuySell.BuySellNo 

		INNER JOIN COUNTERPARTY WITH(NOLOCK) ON Buysell.CustID = Counterparty.CPID

		INNER JOIN CompanyInfo Dept WITH(NOLOCK) ON Buysell.COID = Dept.COID

		INNER JOIN GLAccount WITH(NOLOCK) ON GpLedger.glacct = GLaccount.GLacct 

		LEFT JOIN Counterparty C ON C.CPID = GPLedger.AdjCPID     

		WHERE AdjType = 'S' and TradeDt >= @dt1 and TradeDt <= @dt2 And GrossProfit > 0 

		AND (Source <> 'D' AND Source <> 'F' AND Source <> 'N') 

		AND AdjPosted = 'Y' 

	    AND (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00') OR @COID = '')

	

	---- Tax Adjustment

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType,IsIntercompany,InterCoAcct, coid,[period]) 

    SELECT 

		3,'AR','AR-ADJ-DB',3,	

		BuySell.TradeType,	

	  Coalesce(SalesTaxAcct,''),

	    'Sales-Tax '+COALESCE(SalesTax.TaxDescription,''),	

		GPLedger.BuySellNo,

		GPLedger.TradeDt,	   

		(COALESCE(TaxAmount,0)*-1),

		CASE WHEN BuySell.VendorID = '(INV)' THEN 'WKI' ELSE 'WKS' END,

		C.IsIntercompany,

		C.GLAcctSupplier,

		buysell.COID,

		FORMAT(GpLedger.TradeDt, 'MM/yyyy') [period]

		FROM GPLedger WITH(NOLOCK) INNER JOIN Buysell WITH(NOLOCK) ON GpLedger.BuySellno = BuySell.BuySellNo 

		LEFT JOIN wksSalesTax WITH(NOLOCK)  ON wksSalesTax.LedgerID = GPLedger.LedgerID

		LEFT JOIN SalesTax WITH(NOLOCK) ON SalesTax.SalesTaxID = wksSalesTax.SalesTaxID	

		LEFT JOIN Counterparty C ON C.CPID = GPLedger.AdjCPID     			    

		WHERE AdjType = 'S' and TradeDt >= @dt1 and TradeDt <= @dt2 And AdjSalesTaxAmt > 0 

		AND (GPLedger.[Source] <> 'D' AND GPLedger.[Source] <> 'F' AND GPLedger.[Source] <> 'N') 

		AND AdjPosted = 'Y' 

	    AND (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00') OR @COID = '')

	    AND AdjSalesTaxAmt <> 0 









	----A/R Offset Transaction.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [period]) 

        SELECT 3,'AR',AcctGrp,1,TradeType,@AROffsetAc,@AROffsetAcNm,SUM(HcAmount)*-1,'TOTAL', COID, [period]

	    FROM #LedgerDetail WHERE AcctGrp = 'AR-ADJ-DB'

	    GROUP BY COID, [period], AcctGrp, TradeType



	----Total Sales Debits	

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ISTotalLn,ActionTag, COID, [period]) 

        SELECT 3,'AR','AR-ADJ-DB',4,'','','Totals ',SUM(HcAmount),1,'TOTAL', COID, [period]

	    FROM #LedgerDetail WHERE AcctGrp = 'AR-ADJ-DB' AND (AcctNo <> @AROffsetAc)

		GROUP BY COID, [period]





	--/* NET AR TOTALS */

	------------------------------------------------------------------------------------------=    

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,IsTotalLn,ActionTag, COID, [period]) 

        SELECT 4,'AR','AR-TOTALS',1,'','',' ',SUM((HcAmount*-1)),1,'TOTAL', COID, [period]

	    FROM #LedgerDetail WHERE AcctCLass = 'AR' AND AcctGrp <> 'SALES-TOTALS' AND  (AcctNo <> @AROffsetAc) AND COALESCE(IsTotalLN,0) <> 1

		GROUP BY COID, [period]









	-----------------------------------------------------------------------------------------------------------------   

	-- ASSIGN INTERCOMPANY SALES

	UPDATE #LedgerDetail SET AcctNo = IntercoAcct, AcctNm = 'Intercompany Sales'  WHERE ISNULL(IsIntercompany,0) = 1 AND AcctClass = 'AR'

	-----------------------------------------------------------------------------------------------------------------



	--/* PrePosted Sales Adjustments */

	------------------------------------------------------------------------------------------=    

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount,TranFormType, coid, [period]) 

    SELECT 

		4,'POSTED-SALESADJ','',1,	

		BuySell.TradeType,					

		GlAcct,

		AdjReason,

		GPLedger.BuySellNo,

		FROM payables WITH(NOLOCK) inner join buysell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo

		WHERE RcvDt <= @dt2 and buysell.invoicedt > @dt2 and payables.posted = 1  and payables.itemidsource<>'A'

		AND buysell.Status = 'I' 

		AND (@CoID = '' OR @CoID = BuySell.CoID) 

		group by buysell.COID, FORMAT(RcvDt, 'MM/yyyy')



		UNION

	--Converting Jobs

	SELECT Coalesce(SUM(Amount),0)  Amount, ConvertingJob.COID, FORMAT(ConvertingJob.PostDt, 'MM/yyyy')  [Period]

		FROM payables WITH(NOLOCK) inner join ConvertingJob WITH(NOLOCK) ON Payables.BuySellNo = ConvertingJob.BuySellNo

		where  RcvDt <= @dt2 and ConvertingJob.PostDt > @dt2 and payables.posted = 1  and payables.itemidsource<>'A'

		and ConvertingJob.Status = 'I' 

		AND (@CoID = '' OR @CoID = ConvertingJob.CoID) 

		group by ConvertingJob.COID , FORMAT(ConvertingJob.PostDt, 'MM/yyyy')



		UNION

	--Expense Allocations

	SELECT Coalesce(SUM(Amount),0) Amount,ExpenseAllocation.COID , FORMAT(payables.RcvDt,'MM/yyyy')  [Period]

		FROM payables WITH(NOLOCK) inner join ExpenseAllocation WITH(NOLOCK) ON Payables.BuySellNo = ExpenseAllocation.BuySellNo

		where  payables.RcvDt <= @dt2 and ExpenseAllocation.PostingDt > @dt2 and payables.posted = 1  and payables.itemidsource<>'A'

		and ExpenseAllocation.ISPOsted = 1

		AND (@CoID = '' OR @CoID = ExpenseAllocation.CoID) 

		group by ExpenseAllocation.COID, FORMAT(payables.RcvDt,'MM/yyyy')

		) a1

		GROUP BY COID, [Period]



	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag,COID, [Period]) 

	--SELECT 20,'ALERTS','',2,'','','1.) Expenses Against Future Sales', Round(@AlertTotal,2),'ALERT-EXPONFUTSALE-RCVDT'

	SELECT 20,'ALERTS','',2,'','','1.) Expenses Against Future Sales', Round(#Alert1.Amount,2),'ALERT-EXPONFUTSALE-RCVDT', COID, [Period]

	FROM #Alert1

	if @UsePostingDt = 0 begin

		Coalesce(SUM(payables.Amount),0),

		'ALERT-EXPONUNPOSTSALE-RCVDT'

		,BuySell.COID

		,FORMAT(Payables.InvoiceDt, 'MM/yyyy') [Period]

		from payables WITH(NOLOCK) inner join buysell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo

		where Payables.InvoiceDt <= @dt2 and buysell.Status <> 'I' and payables.posted = 1 and payables.itemidsource<>'A'

		AND (@CoID = '' OR @CoID = BuySell.CoID) 

		group by buysell.coid, FORMAT(Payables.InvoiceDt, 'MM/yyyy')

	end

	else begin

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [Period]) 

	SELECT 

		20,'ALERTS','',3,'','',

		'2.) Expenses Against Unposted Invoices', -- by Rcvd Date',	

		Coalesce(SUM(payables.Amount),0),

		'ALERT-EXPONUNPOSTSALE-RCVDT',

		buysell.COID

		,FORMAT(Payables.RcvDt, 'MM/yyyy') [Period]

		from payables WITH(NOLOCK) inner join buysell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo

		where Payables.RcvDt <= @dt2 and buysell.Status <> 'I' and payables.posted = 1 and payables.itemidsource<>'A'

		AND (@CoID = '' OR @CoID = BuySell.CoID) 

		group by buysell.COID, FORMAT(Payables.RcvDt, 'MM/yyyy')

	end

    --ALERT 3 Unposted Adjustments.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, COID, [Period]) 

    SELECT 

		20,'ALERTS','',4,'','',

		'3.) Unposted Adjustments',	

		SUM(GPLedger.Cost),

		'ALERT-ADJUNPOSTED'

		, buysell.COID

		--,FORMAT(buysell.InvoiceDt, 'MM/yyyy') [period]

		--,FORMAT(gpledger.AdjReqDate, 'MM/yyyy') [period]

		,FORMAT(gpledger.AdjReportDt, 'MM/yyyy') [period]

    from gpledger WITH(NOLOCK) inner join buysell WITH(NOLOCK) ON gpledger.BuySellNo = BuySell.BuySellNo

    WHERE Source = 'A' AND AdjType = 'P' AND GPLedger.AdjCompleted='Y' AND AdjPosted='N'

    AND (@CoID = '' OR @CoID = BuySell.CoID) 

	group by buysell.COID,FORMAT(gpledger.AdjReportDt, 'MM/yyyy')





    --ALERT 4 Adjustment Post Date Prior to Invoice Date.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [period]) 

    SELECT 

		20,'ALERTS','',5,'','',

		' Adjustment Post Date Prior to Invoice Date',	

         sum(gpledger.cost),

        'ALERT-ADJPRIORINVDT'

		,buysell.COID

		,FORMAT(BuySell.InvoiceDt, 'MM/yyyy') [Period]



   from gpledger WITH(NOLOCK) 

    inner join buysell WITH(NOLOCK) ON gpledger.BuySellNo = BuySell.BuySellNo

    inner join payables WITH(NOLOCK) on (gpledger.ledgerid = payables.itemid and gpledger.buysellno = payables.buysellno)

    WHERE Source = 'A' AND AdjType = 'P' AND GPLedger.AdjCompleted='Y' AND AdjPosted='Y'

    AND (@CoID = '' OR @CoID = BuySell.CoID) 

    AND payables.RcvDt < payables.InvoiceDt

    AND BuySell.InvoiceDt >= @dt1 and BuySell.InvoiceDt <= @dt2

	group by buysell.coid,FORMAT(BuySell.InvoiceDt, 'MM/yyyy')



	    --Net-Zero AP FX Invoice Postings.

    SELECT payables.apbatchno, payables.currencycd, sum(amount) as amt, sum(fxamount*-1) as fxamount,buysell.coid 

	,FORMAT(rcvdt, 'MM/yyyy') [Period]

    INTO #fxAdj

    from payables WITH(NOLOCK) inner join buysell ON payables.buysellno = buysell.buysellno

    where rcvdt >= @dt1 and rcvdt <= @dt2 

		AND payables.CurrencyCd <> @SysCurrency 

		AND (@CoID = '' OR @CoID = BuySell.CoID) 

    group by buysell.coid ,FORMAT(rcvdt, 'MM/yyyy'), payables.apbatchno, payables.currencycd 

        having sum(fxamount) = 0

  

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag, coid, [period]) 

    SELECT 

		20,'ALERTS','',6,'','',

		' Net Zero AP FX Postings',	

		sum(amt*-1),

		'ALERT-APNETZERO',

		coid,

		[period]

    FROM #fxadj	

	group by COID,[period]

	---------------------------------



	--1. Assign basic debit and credit values.

	UPDATE #LedgerDetail SET DebitAmt = abs(HCAmount), CreditAmt = 0  WHERE HcAmount > 0 AND AcctClass <> 'ALERTS'

    UPDATE #LedgerDetail SET CreditAmt = abs(HCAmount), DebitAmt = 0  WHERE HcAmount <= 0 AND AcctClass <> 'ALERTS' 



	--2. For Sales: Assign +Debits --> -Credits.

	UPDATE #LedgerDetail SET DebitAmt = 0, CreditAmt = (HCAmount*-1) WHERE AcctGrp = 'SALES' AND COALESCE(ISTotalLn,0) <> 1 AND COALESCE(ActionTag,'') <> 'TOTAL' AND HCAmount > 0 



	--3. For Posted Sales Adjustments: Assign -Credits --> +Debits.

	UPDATE #LedgerDetail SET DebitAmt = (HCAmount), CreditAmt = 0   WHERE AcctClass = 'POSTED-SALESADJ' AND COALESCE(ActionTag,'') <> 'TOTAL' AND HCAmount < 0 

    

	-- 4.  For Expenses and taxes: Assign -credits --> +Debits.  

	UPDATE #LedgerDetail SET DebitAmt = HCAmount, CreditAmt = 0  WHERE ( AcctClass = 'EXPENSES' OR  AcctGrp = 'TAXES-PAID') AND AcctGrpOrd=1 AND COALESCE(ActionTag,'') <> 'TOTAL' and HCAmount < 0 

   

	

	/* COPY DEBIT ON TOTAL LINE */

	UPDATE #LedgerDetail SET CreditAmt = HCAmount, DebitAmt= HCAmount WHERE ISTotalLn = 1 AND HCAmount <> 0
