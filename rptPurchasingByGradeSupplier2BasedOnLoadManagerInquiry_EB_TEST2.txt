	LEFT JOIN [CompanyInfo] W WITH(NOLOCK) ON W.COID = B.COID

	left join po SO on SO.PONumber =  B.RefSONo

	left join Booksheet BS on BS.BookingNo =  B.BookingNo

	left join Grade G on G.GradeID =  B.RefGradeID

	LEFT JOIN [Address] ShipFromAddress WITH(NOLOCK) ON ShipFromAddress.CpID=B.VendorID and ShipFromAddress.Type = B.ShipFromID

	LEFT JOIN [UserDef] UD WITH(NOLOCK) ON UD.UserID = B.UserID

	   , WeightScalesMTD.[Net-wt] Tons

	   , ISNULL(cte1.[Total Sales], 0) [Total Sales]

	   , ISNULL(cte1.TotalProfit, 0) TotalProfit

	   , ISNULL(cte1.FromDate, @FromMnth) FromDate

	   , ISNULL(cte1.ToDate, @ToMnth) ToDate

	   , WeightScalesMTD.ShippingDtPeriod [Shipping Date Period]

	   , ISNULL(cte1.[Sales Amount], 0) [Sales Amount]

	   , ISNULL(cte1.LOADS, 0) LOADS

FROM WeightScalesMTD



LEFT JOIN cte1 on TRIM(WeightScalesMTD.Branch) = TRIM(cte1.Branch) and TRIM(WeightScalesMTD.SupplierID) = TRIM(cte1.SupplierID)

and TRIM(WeightScalesMTD.Custid) = TRIM(cte1.CustID) and TRIM(WeightScalesMTD.GradeGroup) = TRIM(cte1.GradeGroup)  

and TRIM(WeightScalesMTD.[commodity-grade]) = TRIM(cte1.Gradename)  

ORDER BY WeightScalesMTD.Branch, WeightScalesMTD.SupplierID
