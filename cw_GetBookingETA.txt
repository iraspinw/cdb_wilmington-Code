CREATE PROC [dbo].[cw_GetBookingETA] 

	

	@BookingNo char(25)



as 



--06/09/2020 MEC: Added for Mehali to set the ETA on a booking to the shipdate on the worksheet



	SET NOCOUNT ON;

	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;



	SELECT ETA FROM BookSheet WHERE BookingNo = @BookingNo
