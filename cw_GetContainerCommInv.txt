CREATE   PROCEDURE [dbo].[cw_GetContainerCommInv] 



	@BookingNo char(25), 

	@BuySellNo char(12)



AS



--08/14/2018 MEC: Created



-- cw_GetContainerCommInv '12345BOOKING', '501058'



DECLARE @LCNo varchar(25) = ''

DECLARE @LCID int

DECLARE @PONo char(15)

DECLARE @ConsigneeNm varchar(100)

DECLARE @ConsigneeAddr varchar(255)

DECLARE @BeneficiaryCOID char(2)



SET @ConsigneeAddr = ''

SET @LCNo = ''

SET @BeneficiaryCOID = ''



	SELECT @PONo = PONumber	FROM BookSheet WITH(NOLOCK)	WHERE BookSheet.BookingNo = @BookingNo

	IF ISNULL(@PONo,'')<>'' BEGIN

		SELECT @LCID = IssuingLC_ID	FROM PO WITH(NOLOCK) WHERE PONumber = @PONo

		

		SELECT @LCNo = LCNo, @ConsigneeAddr = ConsigneeAddr,@ConsigneeNm = Counterparty.CompanyNm,

		@BeneficiaryCOID = LC.COID

		FROM LC WITH(NOLOCK)

 		LEFT JOIN Counterparty WITH(NOLOCK) ON Counterparty.CpID = LC.ConsigneeCPID

		WHERE LC.ID = @LCID

	END

	

	SELECT BookDate, BookSheet.LoadPort, BookSheet.PortOfDischg, SSLine.CompanyNm, Vessel, VoyageNo, CutOffDate, 

		PO.PONumber,Booksheet.PlaceOfDeliv,

		BookSheet.BookingNo, PO.COID, PO.POReference,PO.Terms,

		dbo.xDoc_GetAddress(BuySell.CustID,BuySell.ShipToID,0,0) as CustAddr,

		Case When datalength(PO.BLConsignTo) > 0 Then PO.BLConsignTo Else dbo.xDoc_GetAddress(PO.CPID,PO.AddrType,0,0) END as ConsignAddr,

		Cust.CompanyNm as CustNm, 

		ETS, ETA, BuysellNo,

		PO.PriceBasis As ShipTerms,

		BookSheet.GradeDesc,

		BLNo, Car.CompanyNm Carrier,

		(SELECT Count(*) From Container Where SaleBuySellNo = @BuySellNo) As Containers,

		BuySell.Notes,

		VatNo,

		PO.POReference RefNo,

		InvoiceDt,

		ShippingDt OnBoardDate,

		@LCNo AS [txtLCNo]

	INTO #REPORT

 	FROM BookSheet WITH(NOLOCK) 

		INNER JOIN BuySell WITH (NOLOCK) ON BookSheet.BookingNo = BuySell.BookingNo 

 		LEFT JOIN Counterparty SSLine WITH(NOLOCK) ON BookSheet.Carrier = SSLine.CpID

		LEFT JOIN PO WITH(NOLOCK) on PO.PONumber = BuySell.RefSONo And PO.[Source] = 'SO'

 		LEFT JOIN Counterparty Cust WITH(NOLOCK) ON PO.CpID = Cust.CpID

		LEFT JOIN CounterParty Car WITH(NOLOCK) ON Car.CpID = Carrier

 	WHERE BookSheet.BookingNo = @BookingNo AND BuySell.BuySellNo = @BuySellNo



	--IF IsNull(@ConsigneeAddr,'')<>''

	--	UPDATE #REPORT SET CustAddr = @ConsigneeAddr

	

	IF IsNull(@ConsigneeNm,'')<>''

		UPDATE #REPORT SET CustNm = @ConsigneeNm

	

 	IF IsNull(@BeneficiaryCOID,'')<>''

 		UPDATE #REPORT SET COID = @BeneficiaryCOID

 	

 	SELECT * FROM #REPORT
