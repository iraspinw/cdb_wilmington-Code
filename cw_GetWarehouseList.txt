	if @IncludeAll = 1 begin

		SELECT 'ALL WAREHOUSES' as CompanyNm, '', 0 

		UNION ALL

		SELECT [Type] CompanyNm, WarehouseID, [Address].InActive

			FROM [Address] WITH (NOLOCK) 

			--INNER JOIN COUNTERPARTY WITH (NOLOCK) ON WarehouseID = Counterparty.CPID  

			WHERE [Address].CpID = '(INV)' AND [Address].[Type] <> '(CONTAINER)'

	

	end

	else begin

		SELECT [Type] CompanyNm, WarehouseID, [Address].InActive

			FROM [Address] WITH (NOLOCK) 

			--INNER JOIN COUNTERPARTY WITH (NOLOCK) ON WarehouseID = Counterparty.CPID  

			WHERE [Address].CpID = '(INV)' AND [Address].[Type] <> '(CONTAINER)' Order By [Type]
