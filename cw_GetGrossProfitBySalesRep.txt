	@ReportType CHAR(12), 

	@FromDate datetime, 

	@ToDate datetime, 

	@CoID char(2), 

    @TradeType varchar(25), 

	@ReferrerNo char(10), 

	@ExclUnassigned as bit, 

	@ExclSalesDisc as bit, 

    @DoSummary bit = 0, 

	@UsePaymentDate bit = 0,	--REPORT BY PAYMENT DATE	

	@ReportUOM CHAR(1), 

	@WksNo CHAR(12) = NULL, 	

	@AccountingAR bit = 0 

	

	



AS

SET NOCOUNT ON

SET ANSI_WARNINGS OFF

	IF @FromDate = '' OR @ToDate = '' BEGIN

		SET @FromDate = '1/1/1980' 

		SET @ToDate = DATEADD(DAY,1,GETDATE())

	END

    

    

    Create Table #zeroalloc(

		BuySellNo char(12),    

		LedgerID int,

		Cnt decimal(25,18)

	)



    --Table to account for Per Each Items added to Inventory Sales.

    Create Table #ChargeBackItems(

		BuySellNo char(12),    		

		ChargeBackAmt decimal(25,8)

    )

	Create Index #IE_ChargeBackItems ON #ChargeBackItems(BuySellNo)    



    

    Create Table #SalesRepAlloc(

		AllocID int IDENTITY (1000, 1) NOT NULL ,

		BuySellNo char(12),

		WksDetailID int,

		LedgerID int,

		[Source] char(1),

		ReferrerNo char(10),

		Role char(1),

		VendorID char(12) , 

		Allocation decimal(25,18) DEFAULT 0,

		TotalVal decimal(25,2) DEFAULT 0,

		InvVal decimal(25,2) DEFAULT 0,

		ExpVal decimal(25,2) DEFAULT 0,

		InvSale decimal(25,2) DEFAULT 0,    

		InvSaleChgBk decimal(25,2) DEFAULT 0, 

		ProfitPerItem decimal(25,8) DEFAULT 0,

		TotalWeightLbs decimal(25,8) DEFAULT 0,		

		TotalProfit decimal(25,2) DEFAULT 0,    

		Wt decimal(25,8) DEFAULT 0,

		WtUOM char(1),

		WeightAlloc decimal(14,3),

		TradeDesc varchar(25),

		Rep_Sales Money DEFAULT 0,

		Rep_Cost Money DEFAULT 0,

		Rep_GrossProfit Money DEFAULT 0,

		AdjQty decimal(25,8),

		TxType CHAR(3),

		TotalEAQty decimal(25,8) DEFAULT 0,

		EaQty  decimal(25,8) DEFAULT 0

 )

   Create Index #IE_SalesRepAlloc ON #SalesRepAlloc(BuySellNo)



	Create Table #BuySellList (

		LedgerID INT,

		BuySellNo CHAR(12),

		GrossProfit Money,

		[Source] CHAR(1),

		TradeDt Datetime,

		--ClosePeriodDt Datetime, --ADDED BY MN

		CustID CHAR(12),

		VendorID CHAR(12),

		TotalWeightLbs Decimal(22,12),

		Sale Money,

		Cost Money,

		AdjType CHAR(1),

		AdjReason VARCHAR(30),

		GLAcct VARCHAR(12),

		AdjustQty decimal(25,8),

		Rep1 Char(10),

		Rep2 Char(10),

		TxType CHAR(3),

		PmtRcvdDate DATE

	)

	Create Index #PK_BuysellList ON #BuySelllist(BuysellNo)	





	



	Create TABLE #_SubInventory(

		BuysellNo CHAR(12),

		LedgerID INT,

		DetailID INT,

		WksInvID int,

		Source CHAR(1),

		LotNo CHAR(15),	

		LotSeq INT DEFAULT 0,	

		VendorID CHAR(15),

		RefID char(10),	

		InvValuePrct decimal(18,5),

		WeightPrct decimal(18,5),

		TotalInvValue Money,

		TotalWeightLbs Decimal(18,5),			

		InvValue Money,

		InvWeight Decimal(18,5),

		AdjQty decimal(25,8)		

	)





	/* DEFINE SPLIT FACTOR */

	-------------------------

	Declare @SplitFactor Money

	SET @SplitFactor = .5		

	IF @ReportType <> 'SPLIT PROFIT'

		SET @SplitFactor = 1	





    /* GET GROSS PROFIT TRANSACTIONS */

	--==============================

	

	--FILL WORKSHEETS.	

	INSERT #BuySelllist

	SELECT	gpLedger.LedgerID,gpledger.BUYSELLNO,GrossProfit,[Source],TradeDT,CustID,VendorID, 

			TotalWeightLbs,Sale,Cost,AdjType,AdjReason,GPLedger.GLAcct,0,

			Coalesce(ReferrerNo1,''),Coalesce(ReferrerNo2,''), CASE WHEN CustID <> '(INV)' AND VendorID <> '(INV)' THEN 'BRK' ELSE 'INV' END,

			NULL   

		FROM GPLedger WITH(NOLOCK) INNER JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno	

		WHERE

		GPLedger.[Source] = 'B' AND BUYSELL.[Status] = 'I'

		AND (@CoID = '' OR (BuySell.CoID = @CoID)) 

		AND	(@TradeType = '' OR (BUYSELL.TradeType = @TradeType)) 

		AND (

				(@UsePaymentDate= 1 AND SPayments <> 0 and SBalancedue <= 0) 

			OR 

				(@UsePaymentDate=0 AND TradeDt >=@FromDate  AND TradeDt <=@ToDate)

			)

		AND (buysell.custid <> '(INV)' AND buysell.custid <> '(PLANT)' AND buysell.custid  <> '{C}' )

		AND (@WksNo IS NULL OR buysell.buysellno = @WksNo)

		

	--FILL ADJUSTMENTS.	

	INSERT #BuySelllist

	SELECT	gpLedger.LedgerID,

			gpledger.BUYSELLNO,

			GrossProfit,

			[Source],

			TradeDT,

			--GPLedger.ClosePeriodDt,

			CASE WHEN AdjType = 'B' THEN GPLedger.AdjCPID ELSE CustID END,

			VendorID, 

			TotalWeightLbs,

			Sale,Cost,

			AdjType,

			AdjReason,

			GPLedger.GLAcct,

			dbo.CnvToBaseQtyUOM(AdjWeight,AdjWeightUOM),

			Coalesce(ReferrerNo1,''),

			Coalesce(ReferrerNo2,''), 

			CASE WHEN CustID <> '(INV)' AND VendorID <> '(INV)' THEN 'BRK' ELSE 'INV' END,

			NULL        

		FROM GPLedger WITH(NOLOCK) LEFT JOIN BuySell WITH(NOLOCK)  ON gpledger.buysellno = buysell.buysellno	

		WHERE

		GPLedger.[Source] <> 'B' AND GPLedger.[Source] <> 'C' AND GPLedger.[Source] <> 'J'

		AND (GPLedger.TradeDt >=@FromDate AND GPLedger.TradeDt <=@ToDate) 

		AND GPLEdger.AdjPosted = 'Y'

		AND (@CoID = '' OR (BuySell.CoID = @CoID OR GPLedger.MiscCOID = @COID)) 

		AND (@TradeType = '' OR (GPLEDGER.MiscTradeType = @TradeType OR BUYSELL.TradeType = @TradeType)) 		

		AND (@WksNo IS NULL OR buysell.buysellno = @WksNo)		









    -- RECOGNIZE PROFIT BASED ON CASH RECEIVED DATE (FOR COMMISSION REPORTING).		        

	IF @UsePaymentDate = 1 BEGIN

		UPDATE #BuySelllist SET #BuySelllist.PmtRcvdDate = 

			(select TOP 1 PaymentDt from RECEIPT where Receipt.BuySellNo = #BuySelllist.BuySellNo  And (Reversed <> 1 or Reversed IS Null) ORDER BY PaymentDt DESC) 

		FROM #BuySelllist WHERE Source = 'B'	



        DELETE #BuySelllist  WHERE (Source = 'B' AND (PmtRcvdDate < @FromDate  OR PmtRcvdDate > @ToDate)) 

		

    END









    /* SALES REP ALLOCATION RECORDS FOR EVERY TRANSACTION. */

    --==========================================================



    --SALES SIDE REPS (INVENTORY).

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,VendorID,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo1,'S',.5,NULL,NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs,#BUYSELLLIST.AdjustQty,#BUYSELLLIST.Vendorid,'INV'

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        WHERE (@ReportType = 'SPLIT PROFIT') AND  BuySell.VendorID = '(INV)'  --Split Profit Only





    --SALES SIDE REPS (BROKERAGE) (2 different reps).

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,VendorID,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo1,'S',.5,NULL,NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs,#BUYSELLLIST.AdjustQty,#BUYSELLLIST.Vendorid,'BRK'

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        WHERE (@ReportType = 'SPLIT PROFIT') AND  BuySell.VendorID <> '(INV)' --Split Profit Only

        AND #BUYSELLLIST.Rep1 <> #BUYSELLLIST.Rep2 AND #BUYSELLLIST.Rep1 <> '' AND #BUYSELLLIST.Rep2 <> ''   



    --SALES SIDE REPS (BROKERAGE) (1 sales rep or same rep twice.).

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,VendorID,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo1,'S',1,NULL,NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs,#BUYSELLLIST.AdjustQty,#BUYSELLLIST.Vendorid,'BRK'

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        WHERE (@ReportType = 'SPLIT PROFIT') AND BuySell.VendorID <> '(INV)' --Split Profit Only

        AND ( (#BUYSELLLIST.Rep1 <> '' AND #BUYSELLLIST.Rep2 = '') OR (#BUYSELLLIST.Rep1 = #BUYSELLLIST.Rep2))



    --SALES SIDE REPS (BROKERAGE) (1 buy rep).

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,VendorID,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo2,'S',1,NULL,NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs,#BUYSELLLIST.AdjustQty,#BUYSELLLIST.Vendorid,'BRK'

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        WHERE (@ReportType = 'SPLIT PROFIT') AND BuySell.VendorID <> '(INV)' --Split Profit Only

        AND (#BUYSELLLIST.Rep1 = '' AND #BUYSELLLIST.Rep2 <> '')





    --BROKERAGE BUY SIDE REPS.

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo2,'B',@SplitFactor,NULL, NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs, #BUYSELLLIST.AdjustQty,'BRK'   -- *(2)

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

     WHERE 

		(

			(@ReportType = 'SPLIT PROFIT' AND BUYSELL.CustID <> '(INV)' AND BUYSELL.VendorID <> '(INV)' 

			    AND  (#BUYSELLLIST.Rep1 <> #BUYSELLLIST.Rep2 AND #BUYSELLLIST.Rep1 <> '' AND #BUYSELLLIST.Rep2 <> '') 

			) 	------Split                  (dsh changed 5/21/2012)  --added: AND BUYSELL.CustID <> '(INV)'

			OR 

			(@ReportType = 'BUYER' AND BUYSELL.VENDORID <> '(INV)')		--Buyer-Only, Brokerage

		)





    --BROKERAGE & INVENTORY SALE -- SELL SIDE REP ONLY.

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,TxType)

        SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo1,'B',@SplitFactor,NULL, NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs, #BUYSELLLIST.AdjustQty,'BRK'   -- *(2)

        FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        WHERE 

		(@ReportType = 'SELLER' )		--Seller Only, Brokerage   AND BUYSELL.VENDORID <> '(INV)'







    --INV PURCHASE ADJUSTMENT SPLIT PROFIT ONLY.     

    IF @ReportType = 'SPLIT PROFIT' BEGIN

		INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation,TotalVal,InvVal,[Source],TotalWeightLbs,AdjQty,TxType)

			SELECT BUYSELL.BuySellNo, LedgerID, ReferrerNo2,'B',@SplitFactor,NULL, NULL,#BUYSELLLIST.[Source], #BUYSELLLIST.TotalWeightLbs, 0, 'INV'

			FROM BUYSELL WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON BUYSELL.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

			WHERE 

			#BUYSELLLIST.[Source] = 'A'

			AND #BUYSELLLIST.CustID = '(INV)' AND #BUYSELLLIST.VendorID <> '(INV)'  	

			AND  (#BUYSELLLIST.Rep1 <> #BUYSELLLIST.Rep2 AND #BUYSELLLIST.Rep1 <> '' AND #BUYSELLLIST.Rep2 <> '') --(2 different reps)

	END

	









IF @ReportType <> 'SELLER' BEGIN



 

    --INV PURCHASE REPS FOR (WEIGHT) INVENTORY **NON-CONVERTING**. 

    --===========================================================

	INSERT #SalesRepAlloc(BuySellNo,LedgerID,WksDetailID,[Source],ReferrerNo,Role,VendorID,WtUOM,Wt,InvVal,AdjQty)

        SELECT WksInventory.BuySellNo, 

			#BUYSELLLIST.LedgerID,

			WksInventory.DetailID,  --Sales Worksheet Detail ID

			#BUYSELLLIST.[Source],

			Inventory.RefID,

			'I',  

			Inventory.VendorID, 

			'L',		

			sum( dbo.convert_to_uom(WksInventory.[Weight],WksInventory.UOM,'L') ),

			sum(WksInventory.wksInvValue),

			#BUYSELLLIST.AdjustQty

        FROM WksInventory WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON WksInventory.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno 

        INNER JOIN BUYSELL InvoiceBuysell WITH(NOLOCK)  ON WksInventory.BUYSELLNO = InvoiceBuysell.BUYSELLNO        

        WHERE dbo.convert_to_uom(WksInventory.[Weight],WksInventory.UOM,'L') <> 0         

			AND Coalesce(ConvertingJobNo,0) = 0		--Exclude Converting Lots.

			AND WksInventory.UOM <> 'E' 

        Group By WksInventory.BuySellNo, LedgerID, DetailID,[Source],RefID,Inventory.VendorID,#BUYSELLLIST.AdjustQty













    --INV PURCHASE REPS FOR (PER EACH) INVENTORY **NON-CONVERTING**. 

    --=============================================================

	INSERT #SalesRepAlloc(BuySellNo,LedgerID,WksDetailID,[Source],ReferrerNo,Role,VendorID,WtUOM,Wt,InvVal,AdjQty)

        SELECT WksInventory.BuySellNo, 

			#BUYSELLLIST.LedgerID,

			WksInventory.DetailID,

			#BUYSELLLIST.[Source],

			Inventory.RefID,'I',			

			Inventory.VendorID, 

			'E',		

			sum(WksInventory.[Weight]),

			sum(WksInventory.wksInvValue),

			#BUYSELLLIST.AdjustQty						

        FROM WksInventory WITH(NOLOCK)  

			INNER JOIN #BUYSELLLIST ON WksInventory.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

			INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno         

			INNER JOIN BUYSELL InvoiceBuysell WITH(NOLOCK)  ON WksInventory.BUYSELLNO = InvoiceBuysell.BUYSELLNO        

        WHERE ((InvoiceBuysell.Referrerno1 = '' OR InvoiceBuysell.ReferrerNo1 IS NULL)  OR (InvoiceBuysell.ReferrerNo2 = '' OR InvoiceBuysell.ReferrerNo2 IS NULL)) 

			 AND Coalesce(ConvertingJobNo,0) = 0	

			AND WksInventory.UOM = 'E' 			

        Group By WksInventory.BuySellNo, LedgerID, DetailID,[Source],RefID,Inventory.VendorID,#BUYSELLLIST.AdjustQty











    --INV PURCHASE REPS FOR (WEIGHT) **CONVERTING**.    

    --==============================================

    -- Breakdown by Sublot. 

	INSERT #_SubInventory

        SELECT 

			WksInventory.BuySellNo, 

			#BUYSELLLIST.LedgerID,

			WksInventory.DetailID,  --Sales Worksheet Detail ID

			WksInventory.WksInvID,  

			#BUYSELLLIST.[Source],

			InventorySubLots.LotNo,

			row_number() OVER (Partition by WksInventory.WksInvID, #BUYSELLLIST.LedgerID order by WksInventory.WksInvID, #BUYSELLLIST.LedgerID), 	

			InventorySubLots.VendorID, 

			InventorySubLots.RefID,

			InventorySubLots.InvValuePrct,

			InventorySubLots.WeightPrct,

			WksInventory.WksInvValue,

			dbo.convert_to_uom(WksInventory.[Weight],WksInventory.UOM,'L'),

			NULL,NULL,

			#BUYSELLLIST.AdjustQty

        FROM WksInventory WITH(NOLOCK)  INNER JOIN #BUYSELLLIST ON WksInventory.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

        INNER JOIN Inventory WITH(NOLOCK)  ON WksInventory.Lotno = Inventory.Lotno 

        INNER JOIN InventorySubLots ON InventorySubLots.Lotno = Inventory.LotNo 

        INNER JOIN BUYSELL InvoiceBuysell WITH(NOLOCK)  ON WksInventory.BUYSELLNO = InvoiceBuysell.BUYSELLNO        

        WHERE 

			dbo.convert_to_uom(WksInventory.[Weight],WksInventory.UOM,'L') <> 0 							

			AND Coalesce(ConvertingJobNo,0) <> 0 



		UPDATE #_SubInventory SET 

			InvValue = Round(TotalInvValue * InvValuePrct,2),

			InvWeight = Round(TotalWeightLbs * WeightPrct,0)















		/* NEW METHOD 2 */

		-- TRUE-UP COST ALLOCATION VARIANCES. (DSH 05/21/2012 Added LedgerDI to fix dup on lotno with adj + buysell on same wks no)

		CREATE TABLE #_TrueUp(WksInvID INT, LedgerID INT, Lotno CHAR(10), TotalCost Money, TotalWeight Decimal(18,5), 

						AllocCost Money, AllocWeight Decimal(18,5), Variance Money DEFAULT 0, 

						WtVariance Decimal(18,5) DEFAULT 0)

		

		INSERT #_TrueUp( WksInvID, LedgerID, Lotno, TotalCost, TotalWeight, AllocCost, AllocWeight )

			SELECT WksInvID, LedgerID, Lotno, TotalInvValue, TotalWeightLbs, SUM(InvValue), SUM(InvWeight)  

			FROM #_SubInventory GROUP BY WksInvID, LedgerID, Lotno, TotalInvValue, TotalWeightLbs



		UPDATE #_TrueUp SET Variance = TotalCost - AllocCost,

							WtVariance = TotalWeight - AllocWeight



		--True-up Cost.					

		UPDATE #_SubInventory 

			SET	#_SubInventory.InvValue = #_SubInventory.InvValue + Variance 		

			FROM #_SubInventory INNER JOIN #_TrueUp ON #_SubInventory.LedgerID = #_TrueUp.LedgerID AND #_SubInventory.WksInvID = #_TrueUp.WksInvID

			WHERE #_SubInventory.LotSeq = 1 AND Variance <> 0	

		--True-up Weight.					

		UPDATE #_SubInventory 

			SET	#_SubInventory.InvWeight = #_SubInventory.InvWeight + WtVariance 		

			FROM #_SubInventory INNER JOIN #_TrueUp ON #_SubInventory.LedgerID = #_TrueUp.LedgerID AND #_SubInventory.WksInvID = #_TrueUp.WksInvID

			WHERE #_SubInventory.LotSeq = 1 AND WtVariance <> 0	



		-- POPULATE REP ALLOCATION TABLE.

		INSERT #SalesRepAlloc(BuySellNo,LedgerID,WksDetailID,[Source],ReferrerNo,Role,VendorID,AdjQty,WtUOM,Wt,InvVal)

			SELECT BuySellNo, 

				LedgerID,

				DetailID,

				[Source],

				RefID,'I',			

				VendorID, 

				AdjQty,

				'L',		

				sum(InvWeight),

				sum(InvValue)			

			FROM #_SubInventory GROUP BY BuySellNo, LedgerID, DetailID,[Source],RefID, VendorID, AdjQty



	END

	--================================================================================================================















    --ADDIN: GP ADJUSTMENTS AND WRITE-OFFS FROM GP ADJUSTMENT LEDGER.

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,[Source],ReferrerNo,Role,Allocation,TotalVal,InvVal)

        SELECT '', GpLedger.LedgerID, #BUYSELLLIST.[Source], ReferrerNo, 'G',1,NULL, NULL

        FROM GPLedger WITH(NOLOCK)  INNER JOIN #BuySellList ON GPLedger.LedgerID = #BuySellList.LedgerID

    WHERE #BUYSELLLIST.[Source] = 'E' OR #BUYSELLLIST.[Source] = 'I'





	-- HANDLE EMPTY INVENTORY SALES    

    UPDATE #SalesRepAlloc SET 		

		Allocation = 1		

        FROM #SalesRepAlloc 

     WHERE 

			#SalesRepAlloc.VendorID = '(INV)'

    UPDATE #SalesRepAlloc SET ReferrerNo = '(U)' WHERE Role = 'I'

        AND (ReferrerNo = '' OR ReferrerNo IS NULL OR ReferrerNo = 'NULL')



    --TOTALLY UNASSIGNED WORKSHEET.

    SELECT #BUYSELLLIST.LedgerID, #BUYSELLLIST.BUYSELLNO,'(U)' AS REFID,'U' AS Role,1 AS Allocation

		INTO #UnAssignedList

		FROM #BUYSELLLIST

		WHERE NOT EXISTS(SELECT * FROM #SalesRepAlloc WHERE #SalesRepAlloc.BuySellno = #BUYSELLLIST.BuySellNo)

    INSERT #SalesRepAlloc(BuySellNo,LedgerID,ReferrerNo,Role,Allocation)

        SELECT BUYSELLNO,LedgerID,REFID,ROLE,Allocation FROM #UnassignedList





    --UNASSIGNED HALF OF INVENTORY SALE WITH NO INVENTORY.

    UPDATE #SalesRepAlloc 

		SET Allocation = 1 

		FROM BUYSELL WITH(NOLOCK)  INNER JOIN #SalesRepAlloc ON BUYSELL.BUYSELLNO = #SalesRepAlloc.BUYSELLNO

		WHERE 

		(

			(BUYSELL.VENDORID = '(INV)' AND NOT EXISTS(SELECT * FROM WKSINVENTORY WITH(NOLOCK)  WHERE WKSINVENTORY.BuySellNo = #SalesRepAlloc.BuySellNo ))

		)

		AND ((Referrerno1 = '' OR ReferrerNo1 IS NULL) OR  (Referrerno2 = '' OR ReferrerNo2 IS NULL))





    -- BUILD TOTAL VALUE & WEIGHT TABLE FOR EACH WORKSHEET.

    SELECT BuySellno, 

		LedgerID, 

		Sum(isnull(InvVal,0)) AS Total_InvVal, 

		Sum( Wt) AS Total_WeightLbs

    INTO #TotalInvVal

    FROM #SalesRepAlloc

    WHERE (ROLE='I' OR ROLE = 'J' ) 

    Group By BuySellNo, LedgerID

    Create index #PK_TotalInvVal ON #TotalInvVal(BuySellNo,LedgerID)









	-- GET MISSING EA WEIGHT -- 

	SELECT BuySellNo, SUM(Wt) TotalQtyEA INTO #TotalQtyEA FROM #SalesRepAlloc WHERE Role = 'I' AND WtUOM = 'E' GROUP BY BuySellNo

	UPDATE #SalesRepAlloc  SET 

		TotalWeightLbs = TotalQtyEA,

	-- BUILD TABLE OF CHARGE-BACKS FROM INVENTORY SALES WHERE WKS DETAIL LINES HAVE NO RELATED INVENTORY.  

	INSERT #ChargeBackItems

		SELECT WksDetail.BuySellNo, Sum(SAmount)

			FROM WksDetail INNER JOIN #BUYSELLLIST ON WksDetail.BUYSELLNO = #BUYSELLLIST.BUYSELLNO

			WHERE #BUYSELLLIST.VendorID = '(INV)' AND Source = 'B'

				AND NOT EXISTS(SELECT WksInvID FROM WksInventory WITH (NOLOCK) WHERE WksInventory.DetailID = WksDetail.DetailID )

			GROUP BY WksDetail.BuysellNo





	--GET TOTAL WEIGHT OF INVENTORY PURCHASES. 

	--(Note: Can not use worksheet totalweightlbs field because sale wt may be diff then purchase weights)

    UPDATE #SalesRepAlloc SET 

		TotalWeightLbs = Total_WeightLbs,

		TotalProfit= (BuySell.SalesAmt+BuySell.SFreightChg-BuySell.PFreightChg)- Total_InvVal		

    FROM #SalesRepAlloc INNER JOIN #TotalInvVal ON #SalesRepAlloc.BuySellNo = #TotalInvVal.BuysellNo

        AND #SalesRepAlloc.LedgerID = #TotalInvVal.LedgerID

        INNER JOIN BUYSELL WITH(NOLOCK) ON BUYSELL.BUYSELLNO = #SalesRepAlloc.BUYSELLNO

    WHERE ROLE='I' OR ROLE = 'J' 















	---- ALLOCATE SALES (BY WORKSHEET LINE ITEM) *WEIGHT*

    UPDATE #SalesRepAlloc 		

		SET 

			--(Amount of sale to assign to inventory based on qty)

			InvSale=( (WksDetail.SAmount / dbo.convert_to_uom(WksDetail.PWeight,WksDetail.PWeightUOM,'L')) * Wt), 

			InvSaleChgBk = (((BuySell.SFreightChg + COALESCE(#ChargeBackItems.ChargeBackAmt,0)) / #SalesRepAlloc.TotalWeightLbs) * Wt),			

			ExpVal =((( BuySell.PFreightChg) / #SalesRepAlloc.TotalWeightLbs) * Wt)		 	  -- 

    FROM WKSDETAIL WITH(NOLOCK) INNER JOIN #SalesRepAlloc ON WKSDETAIL.DetailID = #SalesRepAlloc.WksDetailID

		INNER JOIN BUYSELL WITH(NOLOCK) ON BUYSELL.BUYSELLNO = #SalesRepAlloc.BuySellNo		

		LEFT JOIN #ChargeBackItems ON BUYSELL.BuySellNo = #ChargeBackItems.BuySellNo

	WHERE (Role = 'I' OR Role = 'J') AND (Source = 'B') AND WTUOM <> 'E'

		AND Wt <> 0 and WksDetail.PWeight <> 0    --AND WksDetail.SAmount <> 0   DSH 05/21/2012 removed to handle chargebacks with zero sales.



	

	

	---- ALLOCATE SALES (BY WORKSHEET LINE ITEM) *EACH*

    UPDATE #SalesRepAlloc 		

		SET 

			InvSale=( (WksDetail.SAmount / WksDetail.PWeight) * Wt), 

			InvSaleChgBk = (((BuySell.SFreightChg + COALESCE(#ChargeBackItems.ChargeBackAmt,0)) / #SalesRepAlloc.TotalWeightLbs) * Wt),			

			ExpVal =  ((( BuySell.PFreightChg) / #SalesRepAlloc.TotalWeightLbs) * Wt)		 	  -- 

    FROM WKSDETAIL WITH(NOLOCK) INNER JOIN #SalesRepAlloc ON WKSDETAIL.DetailID = #SalesRepAlloc.WksDetailID

		INNER JOIN BUYSELL WITH(NOLOCK) ON BUYSELL.BUYSELLNO = #SalesRepAlloc.BuySellNo		

		LEFT JOIN #ChargeBackItems ON BUYSELL.BuySellNo = #ChargeBackItems.BuySellNo

		WHERE (Role = 'I' OR Role = 'J') AND (Source = 'B') AND WTUOM = 'E'





















	-- ASSIG PROFIT.

	UPDATE #SalesRepAlloc 	

		SET ProfitPerItem = round((InvSale + InvSaleChgBk) - ExpVal - InvVal,2),

			InvSale = InvSale+InvSaleChgBk

	WHERE (Role = 'I' OR Role = 'J') AND (Source = 'B')









	-- (a.) CALCULATE ALLOCATION-FACTOR BY WEIGHT (For Adjustments). 	

    UPDATE #SalesRepAlloc 

    SET 		

		Allocation = (Wt/TotalWeightLbs)

    FROM #SalesRepAlloc INNER JOIN #TotalInvVal ON #SalesRepAlloc.BuySellNo = #TotalInvVal.BuysellNo

        AND #SalesRepAlloc.LedgerID = #TotalInvVal.LedgerID

    WHERE (ROLE='I' OR ROLE = 'J') AND TotalWeightLbs <> 0 AND TotalWeightLbs IS NOT NULL 

			AND (Source = 'A' Or Source='F' Or Source='N' OR Source = 'D')  --*7/6/11  'D'









    -- (b.) CALCULATE ALLOCATION-FACTOR BY PROFIT (For Worksheets). 

    UPDATE #SalesRepAlloc 

    SET 		

		Allocation = (ProfitPerItem/TotalProfit)

    FROM #SalesRepAlloc INNER JOIN #TotalInvVal ON #SalesRepAlloc.BuySellNo = #TotalInvVal.BuysellNo

        AND #SalesRepAlloc.LedgerID = #TotalInvVal.LedgerID

    WHERE	(ROLE='I' OR ROLE = 'J') 

			AND TotalProfit <> 0 

			AND TotalProfit IS NOT NULL 

			AND

			( 

				(Source <> 'A' And Source <> 'F' AND Source <> 'N' AND Source <> 'D' AND ISNULL(TotalWeightLbs,0) <> 0)

				OR

				((Source = 'A' OR Source='F' OR Source='D' OR Source='N') AND ISNULL(TotalWeightLbs,0) = 0)    --*7/6/11  'D'

			)











    -- SPLIT FACTORS IN HALF FOR INVENTORY PURCHASE REPS (OTHER HALF IS FINAL WKS SALES REP)    

	UPDATE #SalesRepAlloc 

		SET ALLOCATION = ((ALLOCATION*100)* @SplitFactor)/100 WHERE (Role='I' OR Role = 'J')





    -- ASSIGN SUPPLIERS FROM NON-INVENTORY SALES.

    UPDATE #SalesRepAlloc 

		SET VendorID = #BuySellList.VendorID 

        FROM #SalesRepAlloc INNER JOIN #BuySellList ON #BuySellList.BuysellNo = #SalesRepAlloc.BuysellNo

        WHERE #SalesRepAlloc.VendorID IS NULL















	--**NEED TO FIX WEIGHT ALLOCATION SECTION -- Used for reporting converions created rounding errors.**

    -- ASSIGN WEIGHT ALLOCATION FOR REPORTING.

    UPDATE #SalesRepAlloc SET 

        WeightAlloc = 0        

        WHERE (Source <> 'B') 

    

    UPDATE #SalesRepAlloc SET 

        WeightAlloc = round(TotalWeightLbs*Allocation,0) 

        WHERE (Source = 'B' ) AND (Role <> 'I' AND Role <> 'J')  



    UPDATE #SalesRepAlloc SET 

        WeightAlloc = round(AdjQty*Allocation,0) 

        WHERE (Source = 'A') 



          

    UPDATE #SalesRepAlloc SET 

        WeightAlloc = (Wt * @SplitFactor) 

        WHERE (Source = 'B') AND (Role = 'I' OR Role = 'J')



    UPDATE #SalesRepAlloc SET 

        WeightAlloc = (Wt * Allocation) 

        WHERE (Source = 'B') AND Role = 'S' AND WtUom = 'E'





	---- ALLOCATE SALES, COSTS AND GROSS PROFIT USING EITHER ALLOCATION FACTOR OR ACTUAL COSTS IF INVENTORY *(1)

	UPDATE #SalesRepAlloc SET

		Rep_Sales = round(Sale*Allocation,2),		

		Rep_Cost = round(Cost*Allocation,2),

		Rep_GrossProfit = round(GrossProfit*Allocation,2)   

        FROM #BuySellList INNER JOIN #SalesRepAlloc ON #BuySellList.BuysellNo = #SalesRepAlloc.BuysellNo

			AND #BuySellList.LedgerID = #SalesRepAlloc.LedgerID

		WHERE Role <> 'I' OR ((Role = 'I' OR Role = 'J') and #BuySellList.Source <> 'B')

    



	UPDATE #SalesRepAlloc SET

		Rep_Sales = round(InvSale* @SplitFactor,2),		

		Rep_Cost = round((InvVal+ExpVal)* @SplitFactor,2),

		Rep_GrossProfit = round((InvSale-((InvVal+ExpVal))) * @SplitFactor ,2) 		

		FROM #BuySellList INNER JOIN #SalesRepAlloc ON #BuySellList.BuysellNo = #SalesRepAlloc.BuysellNo

			AND #BuySellList.LedgerID = #SalesRepAlloc.LedgerID

			WHERE (Role = 'I' OR Role = 'J') AND (#BuySellList.Source = 'B') -- OR #BuySellList.Source = 'D' )

			

	

	IF @ReportType = 'SELLER' 

		UPDATE 	#SalesRepAlloc SET Role ='I' where vendorid = '(INV)'









			

---------------------------------------DETAIL---------------------------------------------------------------------



    --** PRODUCE FINAL DETAIL REPORT WITH A RECORD FOR EACH REP'S ALLOCATION. **

    SELECT 

		Referrer.Name ReferrerName, 

        #SalesRepAlloc.BuySellNo, 

        #BuySellList.TradeDt, 

		--#BuySellList.ClosePeriodDt, -- ADDED BY MN

        Customer.CompanyNm Customer, 

        Supplier.CompanyNm Supplier,

        dbo.SETTRADETYPE(#BuySellList.[Source],#BuySellList.AdjType,#BuySellList.VendorID) AS TransType,		

        #BuySellList.AdjReason,         

        #SalesRepAlloc.Role,            

        CASE WHEN @ReportUOM='L' THEN WeightAlloc ELSE dbo.convert_to_uom(WeightAlloc,'L',@ReportUOM) END  AS Weight, 

        Rep_Sales,  

        Rep_Cost, 

        Rep_GrossProfit						       

		INTO #Result

        FROM #BuySellList INNER JOIN #SalesRepAlloc ON #BuySellList.BuysellNo = #SalesRepAlloc.BuysellNo

			AND #BuySellList.LedgerID = #SalesRepAlloc.LedgerID

        LEFT JOIN BUYSELL WITH(NOLOCK)  ON #BuySellList.BUYSELLNO = BUYSELL.BUYSELLNO

        left join Referrer ON  #SalesRepAlloc.ReferrerNo=Referrer.ReferrerNo

        left join Counterparty Customer ON Customer.CPID = #BuySellList.CustID

        left join Counterparty Supplier ON Supplier.CPID = #SalesRepAlloc.VendorID

        WHERE (@ExclUnassigned=0 OR (@ExclUnassigned=1 AND Referrer.Name IS NOT NULL )) 

        AND   (@ExclSalesDisc = 0 OR (@ExclSalesDisc = 1 AND #BuySellList.[Source] <> 'D'))    

        AND (@ReferrerNo = '' OR (#SalesRepAlloc.ReferrerNo = @ReferrerNo ))

        order by Referrer.Name, TradeDT, #SalesRepAlloc.Role DESC



PRINT '** PRODUCE FINAL DETAIL REPORT WITH A RECORD FOR EACH REP ALLOCATION. ** - table #result'







		IF @DoSummary = 0 BEGIN

			SELECT	ReferrerName,

					#Result.BuySellNo,

					TradeDt,

					--#Result.ClosePeriodDt,

					Customer,

					[Weight],

					Rep_Sales,

			ORDER BY ReferrerName

		ELSE

			BEGIN

			SELECT isnull(ReferrerName, '(UNASSIGNED)') as ReferrerName,  TradeDt, Rep_GrossProfit

			INTO #RawResult

			FROM #Result

            Declare @OffsetDate datetime, @OffsetYTD_Date datetime, @OffsetYTD_Mnth int
