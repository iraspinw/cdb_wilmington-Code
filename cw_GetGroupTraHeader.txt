



CREATE proc [dbo].[cw_GetGroupTraHeader] @TransportationNo int as



--11/25/2020 Alex: Created for Group Transportation



	SELECT CompanyNm, CompanyAddr1, CompanyAddr2, CompanyCity, CompanyRegion, CompanyPostalCd,

	CompanyPhone, CompanyFax, CompanyEmail, LogoImage,

	 PrintCompanyNm, LogoSize, LogoWidth, LogoHeight,

	CompanyInfo.CompanyCtry as Country,AltLogoImage,

	Country.Name as CountryFullName

	INTO #DocHeader

	FROM CompanyInfo WITH (NOLOCK) 

	INNER JOIN GroupTransportation WITH (NOLOCK) ON CompanyInfo.COID = GroupTransportation.COID

	left join Country on Country.CountryID = CompanyInfo.CompanyCtry

	--LEFT JOIN Country WITH (NOLOCK) ON GroupTransportation.CompanyCtry = Country.CountryID

	WHERE GroupTransportation.TransportationNo = @TransportationNo





	UPDATE #DocHeader SET AltLogoImage = (SELECT AltLogoImage FROM CompanyInfo WHERE COID = '00')

	

	UPDATE #DocHeader SET LogoImage = NULL WHERE LogoSize=0





	IF ISNULL((select optionValue from options where optionname = 'OmitAddrPrint'),0)=1

		UPDATE #DocHeader 

		SET CompanyNm='', CompanyAddr1='', CompanyAddr2='', CompanyCity='', CompanyRegion='',

			CompanyPostalCd='', CompanyPhone='', CompanyFax='', CompanyEmail='',Country=''





	select * from #DocHeader
