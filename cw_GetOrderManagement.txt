SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

DECLARE @filler decimal(18,5), @Cnt bigint, @i bigint

SELECT @filler = 0.000





DECLARE @PType bit

SET @PType = 0

IF @PriceType = 'Contract' SET @PType = 1





	--FILTER DATA INTO TEMP TABLE

	SELECT DISTINCT PODetail.PO_ID as pID,

        PO.PONumber,

        PO.[Source],

        PODate,

        Grade.Description,

        PODesc, 

        dbo.CnvFromBaseQtyUOM(dbo.CnvToBaseQtyUOM(Weight,WeightUOM),@ToWtUOM) as OrderQty,

      Units,

        PODetail.UnitType,

        Price,

        PriceUOM,

	 PriceBasis as ShippingTerm,

        REPLACE(REPLACE(Specifications,char(10),''),char(13),' ') Specifications,

        @filler as ShippedQty,

        (SELECT TOP 1 referrer.name FROM POSalesReps INNER JOIN referrer ON POSalesReps.ReferrerNo = referrer.ReferrerNo WHERE POSalesReps.POID = PO.POID) As SalesRep,        

        PO.Status,

        c.CompanyNm,

        PO.POReference,

        PO.MonthlyContract, 

        Grade.GradeID,

		ProductGroupID,

        Address.country,

        c.CpID as CounterpartyID,

        SPACE(30) as CounterpartyReference, 

        ShipTo.CompanyNm as CounterpartyName, 

        SPACE(15) as CounterpartyOrder, 

        PO.AllocPeriod,

		WeightUOM as OriginalWtUOM,

		PODetail.CurrencyCd,

		PriceFormula.FormulaDesc,

		PO.AddrType,Comment,

		ExpirationDt,

		UnitType.PropertySet AS UnitTypePpt,

		PODetail.Amount as OriginalAmount,

		PODetail.Amount as OpenAmount,

		dbo.CnvFromBaseQtyUOM(dbo.CnvToBaseQtyUOM(PO.MinWeight,PO.MinWeightUOM),@ToWtUOM) as MinWt,

		@filler as LLQty,

		CompanyDesc Dept,

		BasisWt, 

		BasisWtUOM,

		Caliper,

		RollWd, 

		RollWdUOM,  

		Color,

		Diameter, 

		DiameterUOM,

	    dbo.RemoveTrailingZeros(CoreSize) CoreSize,

		CoresizeUOM,

		LinearLen,

		LengthUOM,

		SheetSize, 

	    SheetSizeUOM,

		NoSheets,

		dbo.RemoveTrailingZeros(MWeight) MWeight,

		PlasticMelt,

		PlasticType,

		PlasticForm,

		PODetail.FormulaID,

		case when po.MonthlyContract = 1 then 'CONTRACT' else 'SPOT' end as OrderType,

		PO.ExpirationDt As [Expiration],

		UserDef.UserNm

	INTO #Report

	FROM PO WITH(NOLOCK)

	INNER JOIN PODetail WITH(NOLOCK) on PO.PONumber = PODetail.PONumber

	INNER JOIN Grade WITH(NOLOCK) ON PODetail.GradeID = Grade.GradeID

	INNER JOIN Counterparty c WITH(NOLOCK) on PO.CpID = c.CpID

	LEFT JOIN Address WITH(NOLOCK) ON PO.ShipAddrID = Address.CpID AND PO.ShipAddrType = Address.[Type]

	LEFT JOIN POSalesReps p WITH(NOLOCK) ON p.POID = PO.POID

	LEFT JOIN PriceFormula WITH(NOLOCK) ON PODetail.FormulaID = PriceFormula.FormulaID		

	LEFT JOIN UnitType WITH(NOLOCK) ON UnitType.TypeID = PODetail.UnitType

	LEFT JOIN Counterparty ShipTo WITH(NOLOCK) on PO.ShipAddrID = ShipTo.CpID

	LEFT JOIN CompanyInfo ON PO.COID = CompanyInfo.COID

	LEFT JOIN UserDef on UserDef.UserID = PO.UserID

	WHERE (@Type = '' OR PO.Source = @Type)

	AND (@Status = '' OR @Status = PO.Status)

	AND (@RefNo = '' OR p.ReferrerNo = @RefNo)

	AND (@TradeType = '' OR PO.TradeType = @TradeType)

	AND (@GradeID = '' OR (PODetail.GradeID = @GradeID AND @ProductType = 'GRADE')	OR (Grade.ProductCategoryID = @GradeID AND @ProductType = 'CATEGORY'))		

	AND (@FromDt = ''  OR (@FromDt <= PO.PODate AND PO.PODate <=@ToDt))

	AND (@SchFromDate = '' OR (@SchFromDate <= PO.SchShipDate AND PO.SchShipDate <= @SchToDate))

	AND (@CpID = '' OR PO.CpID = @CpID)

	AND (@COID = '' OR PO.COID = @COID)

	AND (@ToWtUOM NOT IN ('E','Q') OR (@ToWtUOM = PoDetail.WeightUOM))

	AND (@PriceType='' OR @PType = ISNULL(PO.MonthlyContract,0))

	AND (@SalesRepNo = '' OR (@SalesRepNo <> '' AND EXISTS(SELECT * FROM POSalesReps WHERE POSalesReps.POID = PO.POID AND POSalesReps.ReferrerNo = @SalesRepNo))) 

	and (@Location = '' or (PO.AddrType = @Location))

	AND (@OrderType = 'ALL' or(

		(@OrderType = 'SPOT' AND (PO.MonthlyContract <> 1)) OR (@OrderType = 'CONTRACT' AND (PO.MonthlyContract = 1)) OR (@OrderType = 'PROVISIONAL' AND (PO.IsProvisional = 1))

		))

	AND (@UserID = '' OR PO.UserID = @UserID)





	--GET SHIPPED WEIGHTS FROM WORKSHEET DETAIL

	CREATE TABLE #ShippedWeight(

		pID INT,

		Wt DECIMAL(22,8),

		NumItems INT,

		CONSTRAINT pkShippedWeight PRIMARY KEY (pID)

	)



	IF @Type = 'SO' BEGIN

		INSERT #ShippedWeight

		SELECT PODetail.PO_ID as pID,

			Sum(dbo.CnvFromBaseQtyUOM(dbo.CnvToBaseQtyUOM(SWeight,SWeightUOM),@ToWtUOM)) as Wt, 

			Count(*) as NumItems			

			FROM WksDetail WITH(NOLOCK) 

				INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = WksDetail.BuySellNo

				INNER JOIN PODetail WITH(NOLOCK) ON PODetail.PO_ID = wksDetail.SPO_ID 

				INNER JOIN #Report ON PODetail.PO_ID = #Report.pID

			WHERE BuySell.[status] <> 'X'

			AND (@ShippingOK = '' OR BuySell.ShippingOK = @ShippingOK)

			GROUP BY PODetail.PO_ID	

	END

	

	IF @Type = 'PO' BEGIN	

		INSERT #ShippedWeight

		SELECT PODetail.PO_ID as pID, 

			Sum(dbo.CnvFromBaseQtyUOM(dbo.CnvToBaseQtyUOM(PWeight,PWeightUOM),@ToWtUOM)) as Wt,

			Count(*) as NumItems

			FROM WksDetail WITH(NOLOCK) 

				INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = WksDetail.BuySellNo

				INNER JOIN PODetail WITH(NOLOCK) ON PODetail.PO_ID = wksDetail.PPO_ID 

				INNER JOIN #Report ON PODetail.PO_ID = #Report.pID

			WHERE BuySell.[status] <> 'X'

			AND (@ShippingOK = '' OR BuySell.ShippingOK = @ShippingOK)

			AND (COALESCE(BuySell.BaleRouteID,0)=0 OR (COALESCE(BuySell.BaleRouteID,0) <> 0 AND BuySell.VendorID <> '(INV)' AND BuySell.ShipFromID <> '[IN TRANSIT]'))

			GROUP BY PODetail.PO_ID

	END





		

	--INSERT SHIPPED WEIGHTS (BASED ON ORDER NO., GRADE, AND UNITTYPE)

	UPDATE #Report SET ShippedQTY = #ShippedWeight.Wt

	FROM #Report 

	INNER JOIN #ShippedWeight ON #Report.pID = #ShippedWeight.pID

	--		PODetail.GradeID, PODetail.UnitType

	--		FROM WksDetail WITH(NOLOCK) 

	--		INNER JOIN BuySell WITH(NOLOCK) ON BuySell.BuySellNo = WksDetail.BuySellNo

	--		INNER JOIN Booksheet WITH(NOLOCK) ON BuySell.BookingNo = Booksheet.BookingNo

	--		INNER JOIN PO WITH(NOLOCK) ON PO.PONumber = Booksheet.PONumber AND PO.[Source] = 'SO'

	--		INNER JOIN PODetail WITH(NOLOCK) ON PODetail.PONumber = PO.PONumber AND PODetail.GradeID = WksDetail.GradeID AND PODetail.UnitType = WksDetail.UnitType

	--		INNER JOIN #Report WITH(NOLOCK) ON PODetail.PO_ID = #Report.pID

	--		WHERE BuySell.ShippingOK IN ('S','A','R') AND @Type = 'SO'  AND BuySell.[status] Not In ('I', 'X')

	--		AND NOT EXISTS(SELECT LotNo FROM WksInventory WITH(NOLOCK) WHERE WksInventory.LotNo = WksDetail.LotNo)

	--		GROUP BY PODetail.PO_ID, PO.PONumber, PODetail.GradeID, PODetail.UnitType

			

	--	SELECT @Cnt = COUNT(*) FROM #Committed WITH(NOLOCK) 

		

	--	SET @i = 1

	--	-- LOOP THROUGH AND UPDATE COMIT WT AND DELETE ANY DUPE RECORDS

	--	WHILE @i <= @Cnt

	--	BEGIN

	--		SET @i = @i + 1

	--	END

		

	--	DROP TABLE #Committed

	--END	







	--GET SUPPLIER'S ORDER NO.

	IF @Type = 'SO' BEGIN

		UPDATE #Report SET CounterpartyOrder = 

			(SELECT TOP 1 PO.PONumber

			FROM PO INNER JOIN  WksDetail ON PO.PONumber = WksDetail.PPO

			WHERE WksDetail.SPO = #Report.PONumber)		



        -- From Linked Purchase Order to Sales Order --

		UPDATE #Report SET CounterpartyOrder = 

			(SELECT TOP 1 PO.PONumber FROM PO INNER JOIN  PODetail ON PO.PONumber = PoDetail.PoNumber WHERE PoDetail.SPO = #Report.PONumber) 

            WHERE (CounterpartyOrder = '' OR CounterpartyOrder IS NULL)

	

	END



    -- GET CUSTOMERS ORDER NO.

	IF @Type = 'PO' BEGIN    

	

            -- Get Order Number from Drop-Ship Worksheet.

		    UPDATE #Report SET CounterpartyOrder = 

			    (SELECT TOP 1 PO.PONumber

			    FROM PO INNER JOIN WksDetail ON PO.PONumber = WksDetail.SPO

			    WHERE WksDetail.PPO = #Report.PONUmber)		



            -- From Linked Purchase Order to Sales Order --

	      	UPDATE #Report SET CounterpartyOrder = 

		    	(SELECT TOP 1 SPO FROM PODETAIL WHERE PoDetail.PONumber = #Report.PONumber) 

                WHERE (CounterpartyOrder = '' OR CounterpartyOrder IS NULL)



	END





	--GET CUSTOMER/SUPPLIER NAME AND REF. NO 

	UPDATE #Report SET CounterpartyName = Counterparty.CompanyNm, CounterpartyReference = PO.POReference

		FROM #Report 

		INNER JOIN PO ON #Report.CounterpartyOrder = PO.PONumber

		INNER JOIN Counterparty ON PO.CpID = Counterparty.CpID	





	--IF NOT MONTHLY CONTRACT FORMULA DESC SHOULD BE EMPTY

	UPDATE #Report SET FormulaDesc = '' WHERE MonthlyContract<>1



	-- WHEN REPORTING BY SALES REP SET NAME	COLUMN.

	IF @SalesRepNo <> '' BEGIN

		DECLARE @SalesRepNm VARCHAR(30)

		SELECT @SalesRepNm = [Name] FROM Referrer WHERE ReferrerNo = @SalesRepNo

		UPDATE #REPORT SET SalesRep = @SalesRepNm 

	END































	--RETURN DATA

	if (@ToWtUOM = 'L' or @ToWtUOM='K') 	begin

	SELECT

		MonthlyContract,

		PONumber As [Order No],

		[Source] As [ref ORDER TYPE],

		POReference As Reference,

		Dept,

		PODate As [Order Date],

		CompanyNm As Account,

		AddrType As Location,

		[Description] As Product,

		IsNull(dbo.CnvUnitName(UnitType), '') As [Packaging],

		dbo.cw_RemoveTrailingZeros(Price, 0) + '/' + dbo.CnvUOM_Names(PriceUOM) As Price,

		ShippingTerm As [Terms],



		dbo.cw_RemoveTrailingZeros(IsNull(cast((OrderQty) as decimal(16,0)),0),0) As [Order Qty],

		dbo.cw_RemoveTrailingZeros(IsNull(cast((ShippedQty) as decimal(16,0)) ,0), 0) As [Shipped Qty],

		dbo.cw_RemoveTrailingZeros(IsNull(cast(([Committed]) as decimal(16,0)),0), 0) As [Committed Qty],

		dbo.cw_RemoveTrailingZeros((cast((IsNull(OrderQty, 0)-IsNull(ShippedQty,0)-IsNull([Committed],0)) as  decimal(16,0))),0) as [Remaining Qty],

		IsNull(SalesRep,'') As [Sales Rep],

		CONVERT(varchar, CAST(OriginalAmount AS money), 1) As [Order Value],

		CONVERT(varchar, CAST(OpenAmount AS money), 1) As [Remaining Value],

		pID As [ref POID],

		BasisWt [Basis Wt], 

		BasisWtUOM [Basis UOM],

		Caliper  [{C}Caliper],

		RollWd [Roll Width], 

		RollWdUOM [Roll Width UOM],  

		#Report.color,

		Diameter, 

		DiameterUOM [Diameter UOM],

		CoreSize [Core Size],

		CoresizeUOM [Core Size UOM],

		LinearLen [Length],

		LengthUOM [Length UOM],

		SheetSize [Sheet Size], 

	    SheetSizeUOM [Sheet Size UOM],

		NoSheets Sheets,

		MWeight [M-Weight],

		PlasticMelt [Meltflow Index],

		PlasticType [Type-Grade],

		PlasticForm [Form],

		FormulaID AS [Index Pricing],

		OrderType as [Order Type],

		[Expiration],

		ShipAddrType as [Ship To],

		UserNm as [Created By]

		FROM #Report ORDER BY PODate,PONumber,Description

	end



	ELSE 

	begin

	SELECT

	PONumber As [Order No],

		[Source] As [ref ORDER TYPE],

		POReference As Reference,

		Dept,

		PODate As [Order Date],

		CompanyNm As Account,

		AddrType As Location,

		[Description] As Product,

		IsNull(dbo.CnvUnitName(UnitType), '') As [Packaging],

		dbo.cw_RemoveTrailingZeros(Price, 0) + '/' + dbo.CnvUOM_Names(PriceUOM) As Price,

		ShippingTerm As [Terms],



		dbo.cw_RemoveTrailingZeros(IsNull(cast((OrderQty) as decimal(16,3)),0),0) As [Order Qty],

		dbo.cw_RemoveTrailingZeros(IsNull(cast((ShippedQty) as decimal(16,3)) ,0), 0) As [Shipped Qty],

		--dbo.cw_RemoveTrailingZeros(IsNull(cast(([Committed]) as decimal(16,3)),0), 0) As [Committed Qty],

		dbo.cw_RemoveTrailingZeros((cast((IsNull(OrderQty, 0)-IsNull(ShippedQty,0)-IsNull([Committed],0)) as  decimal(16,3))),0) as [Remaining Qty],

		IsNull(SalesRep,'') As [Sales Rep],

		CONVERT(varchar, CAST(OriginalAmount AS money), 1) As [Order Value],

		CONVERT(varchar, CAST(OpenAmount AS money), 1) As [Remaining Value],

		pID As [ref POID],

		BasisWt [Basis Wt], 

		BasisWtUOM [Basis UOM],

		Caliper  [Caliper],

		RollWd [Roll Width], 

		RollWdUOM [Roll Width UOM],  

		#Report.Color,

	    Diameter, 

		DiameterUOM [Diameter UOM],

		CoreSize [Core Size],

		CoresizeUOM [Core Size UOM],

		LinearLen [Length],

		LengthUOM [Length UOM],

		SheetSize [Sheet Size], 

	    SheetSizeUOM [Sheet Size UOM],

		NoSheets Sheets,

		MWeight [M-Weight],

		PlasticMelt [Meltflow Index],

		PlasticType [Type-Grade],

		PlasticForm [Form],

		FormulaID AS [Index Pricing],

		OrderType as [Order Type],

		[Expiration],

		ShipAddrType as [Ship To],

		UserNm as [Created By]

	FROM #Report ORDER BY PODate,PONumber,Description
