

CREATE PROC [dbo].[cw_GetUOMS] as



--09/13/2016 MEC: Created



	SET NOCOUNT ON;

	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;



	Create Table #UOMS (UOM varchar(3), UOMKey char(1))

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('EA', 'E')

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('KG', 'K')

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('LBS', 'L')

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('GT', 'G')

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('MT', 'M')

	INSERT INTO #UOMS (UOM, UOMKey) VALUES ('ST', 'S')

	

	SELECT UOM, UOMKey FROM #UOMS ORDER BY UOM
