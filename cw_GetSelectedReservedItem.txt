	select Inventory.LotNo, inventory.UnitType, VendorID, inventory.GradeID, inventory.[Weight], WeightUOM, InvValue / inventory.[weight] Price, 

			WeightUOM PriceUOm, Buysellno, rcvdt, qtyonhand, WarehouseNm, round(InvValue,2) InvValue, 

			round(OrigInvValue,2) OrigInvValue, convert(int,inventory.LastEdit) as LastEdit,

			[Description] as GradeNm, SerialNo, inventory.Units, OrigUnits, SpecString, inventory.LotNo, ISNULL(containerno,'') containerno, SpecString

	from Inventory 

	inner join ReserveStageSOIntoInvShip on ReserveStageSOIntoInvShip.LotNo = Inventory.LotNo

	inner join grade on grade.gradeid = inventory.GradeID

	left join ContainerContents on ContainerContents.LotNo = inventory.LotNo

	left join container on ContainerContents.ContainerID=container.ContainerID

    where ReserveStageSOIntoInvShip.PONumber = @PONumber and ReserveStageSOIntoInvShip.CreatedDateTime=  @CreationDate
