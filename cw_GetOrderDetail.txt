CREATE proc [dbo].[cw_GetOrderDetail] @PO_ID int as 

set transaction isolation level read uncommitted

set nocount on



select unittype, GradeID, WeightUOM wtuom, Price, PriceUOM, CurrencyCd, 1 as fxrate, PO_ID, shipto, Terms, specifications

from podetail 

inner join po on po.PONumber = PODetail.PONumber

where PO_ID = @PO_ID
