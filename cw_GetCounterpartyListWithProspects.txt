CREATE   PROCEDURE [dbo].[cw_GetCounterpartyListWithProspects] AS

--11/9/2017

SET NOCOUNT ON 

SELECT CompanyNm, RTRIM(CpID) AS CpID, Role, ActiveStatus, AutoFeedAPVoucher, FreightType, CurrCode 

FROM CounterParty WITH (NOLOCK)  

WHERE RTRIM(CpID) NOT IN ('(INV)','(PLANT)','{C}')

ORDER BY CompanyNm
