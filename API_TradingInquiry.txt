CREATE PROC [dbo].[API_TradingInquiry]  

	@FromDt datetime, 

	@ToDt datetime,

	@Dept char(50),

	@TradeType varchar(25),

	@Customer varchar(50) = '',

	@Supplier varchar(50) = '',

	@InventoryType varchar(15) = '',

	@DateType char(4) = '',

	@Status char(1) = ''

AS

SET NOCOUNT ON 

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED



-- 09/27/2023 DN: order date split up between purchase and sales order, dest port split up between sales and purchas orders

-- 09/27/2023 DN: added shipfullname,po/so shipping terms, po/so ship via, poso payment terms, custid, vendorID per request of Scrapgo 

-- 10/18/2023 DN: added refEquipNo per request of Scrapgo 

-- 11/28/2023 DN: added GrossWt,TareWt,NetWt

--6/23/2025 DN: added Logistics





	CREATE TABLE #REPORT(

		Buysellno CHAR(12),

		Detailid INT,

		Department VARCHAR(200),

		TradeType  VARCHAR(200),

		[Status] VARCHAR(30),

		ShippingDt DATE,

		InvoiceDt DATE,

		SoOrderDt DATE,

		PoOrderDt DATE,

		Customer VARCHAR(200),	

		CustomerAddress VARCHAR(450),

		ShipFromID VARCHAR(200),	 

		RelNo VARCHAR(50),	 

		ContainerNo CHAR(40),	

		Supplier VARCHAR(200),

		SupplierAddress VARCHAR(450),

		ShipTo VARCHAR(200),

		PickupNo VARCHAR(200), 	

		BookingNo VARCHAR(200),

		SoDestPort VARCHAR(200),

		PoDestPort VARCHAR(200), 

		ETA DATETIME,

		ProductCat VARCHAR(200),

		ProductGrp VARCHAR(200),

		Description VARCHAR(200),

		Specifications VARCHAR(800),

		InvoiceDesc VARCHAR(200),

		Units INT,

		UnitType CHAR(100),

		WksWeightLBS DECIMAL(22,5),

		SWeight  DECIMAL(22,5),

		SWeightUOM CHAR(1),

		PWeight  DECIMAL(22,5),

		PWeightUOM CHAR(1), 

		SWeightLBS DECIMAL(22,5),

		WksSales MONEY,	

		WksChgBack MONEY,

		WksExp MONEY,

		WksPurchases MONEY,

		WksSAdj MONEY,

		WksPAdj MONEY,

		Factor decimal(18,12),

		SPO CHAR(30),

		SCurrencyCd CHAR(3),

		SPrice money,

		SPriceUOM CHAR(1),

		SAmount	MONEY,

		PO CHAR(30),

		PCurrencyCd CHAR(3), 

		PPrice  money,

		PPriceUOM CHAR(1),	

		PAmount MONEY,

		ChargeBacks MONEY, 

		FreightExp MONEY, 

		Expenses MONEY,

		SAdjs MONEY,

		PAdjs MONEY,

		Interest MONEY,

		BankFees MONEY,	

		Discount MONEY,	

		Margin MONEY,

		TotalSale MONEY,

		TotalCost MONEY,

		GrossProfit MONEY,

		NetProfit MONEY,

		[Sell-Rep] VARCHAR(200), 

		[Buy-Rep] VARCHAR(200), 

		IsAgency BIT,

		IsNegSales BIT,

		TranSrc CHAR(3),

		UserDefined1 VARCHAR(100),

		UserDefined2 VARCHAR(100),

		UserDefined3 VARCHAR(100),

		OwnerName VARCHAR(60),

		POShippingTerms varchar(100),

		POShipVia varchar(50),

		POPaymentTerms varchar(50),

		SOShippingTerms varchar(100),

		SOShipVia varchar(50),

		SOPaymentTerms varchar(50),

		CustID char(12),

		SupplierID char(12),

		EquipNo char(45),

		GrossWt DECIMAL(22,5),

		TareWt DECIMAL(22,5),

		NetWt DECIMAL(22,5),

		Logistics char(12)

		



	)



	INSERT #REPORT

	SELECT  

		buysell.buysellno,

		wksdetail.detailid,

		W.CompanyDesc,

		BuySell.TradeType,

		[dbo].[CnvWksStatus_Name](ISNULL(buysell.[Status],''),'') [Status],

		ShippingDt,

		InvoiceDt,

		

		SalePO.PODate SODate,

		purchpo.PODate PODate,

		--Customer Supplier

		C.CompanyNm Customer,

		dbo.FormatAddress(CustAddr.Addr1, CustAddr.Addr2, CustAddr.Addr3, CustAddr.City, CustAddr.Region, CustAddr.PostalCd, CustAddr.Country, 1),

		buysell.ShipFromID, 

		Buysell.RelNo,

		isnull(wksdetail.ContainerNo,''),	

		S.CompanyNm Supplier,

		dbo.FormatAddress(SupAddr.Addr1, SupAddr.Addr2, SupAddr.Addr3, SupAddr.City, SupAddr.Region, SupAddr.PostalCd, SupAddr.Country, 1),

		buysell.shiptoid ShipTo, 

		PickupNo, 	

		Buysell.BookingNo,

		isnull(SalePO.DestCountry,'') SODestPort,

		isnull(PurchPO.DestCountry,'') PODestPort,

		(CASE WHEN BookSheet.ETA < '1/1/1980' THEN '' ELSE BookSheet.ETA END) ,



		--Grade

		isnull(ProductCategoryID, ' ') as ProductCat,

		isnull(ProductGroupID, ' ') as ProductGrp,

		Grade.Description,

		Specifications,

		Wksdetail.InvoiceDesc,



		--Quantity

		Units,

		UnitType.UnitType,

		Round(TotalWeightLBS,0,1) WksWeightLBS,

		SWeight, SWeightUOM,

		PWeight, PWeightUOM, 

		dbo.CONVERT_TO_UOM(SWeight,SWeightUOM,'L') SWeightLBS,

	

		--TOTAL WKS FINANCIALS

		SalesAmt		WksSales,	

		SFreightChg		WksChgBack, 

		PFreightChg		WksExp,

		PSupplierCost	WksPurchases,

		SAdjustment		WksSAdj, 

		PAdjustment		WksPAdj,



		--WKSDETAIL FINANCIALS	

		convert(decimal(18,12),0) as Factor, 



		--Sale	

		SPO,

		wksdetail.SCurrencyCd,

		SPrice, SPriceUOM,

		SAmount,	

	

		--Purchase

		PPO as PO,

		wksdetail.PCurrencyCd, 

		PPrice, PUOM PPriceUOm,	

		PAmount,



		--Expenses

		CONVERT(MONEY,0) AS ChargeBacks, 

		CONVERT(MONEY,0) as Expenses,

		CONVERT(MONEY,0) as SAdjs,

		CONVERT(MONEY,0) as PAdjs,

		CONVERT(MONEY,0) as Interest,

		CONVERT(MONEY,0) as BankFees,	

		CONVERT(MONEY,0) as Discount,	

		

		--Other Financials

		CONVERT(MONEY,0) as Margin,

		CONVERT(MONEY,0) as TotalSale,

		convert(money,0) as TotalCost,

		convert(money,0) as GrossProfit,

		convert(money,0) NetProfit,



		(SELECT sum(WksDelivery.Cost) from WksDelivery WITH (NOLOCK) WHERE WksDelivery.BuySellNo = buysell.BuySellNo and WksDelivery.isFreight=1) as FreightExp ,--Alex pend·ing change



		--SALES REPS

		isnull(referrer.[name], ' ') as Referrer, 

		isnull(PurchRep.[name], ' ') as Referrer2,

	

		BuySell.IsAgency,

		0 as IsNegSales,

		(CASE WHEN VendorID = '(INV)' THEN 'INV' ELSE 'BRK' END),

		buysell.UserDefined1,

		buysell.UserDefined2,

		buysell.UserDefined3,



		UserDef.UserNm,

		isnull(PurchPO.PriceBasis,'') POShipTerms,

		isnull(PurchPO.ShipVia,'') POShipVia,

		isnull(PurchPO.Terms,'') POShipTerms,

		isnull(SalePO.PriceBasis,'') SOShipTerms,

		isnull(SalePO.ShipVia,'') SOShipVia,

		isnull(SalePO.Terms,'') SOShipTerms,

		CustID,

		VendorID,

		buysell.RefEquipNo,

		isnull(WKSDetail.GrossWt,0),

		isnull(WKSDetail.TareWt,0),

		isnull(WKSDetail.GrossWt - WKSDetail.TareWt,0),

		--BasisWt,BasisWtUOM,RollWd,RollWdUOM,SheetSize,SheetSizeUOM,Diameter, DiameterUOM,CoreSize,CoreSizeUOM,Caliper,LinearLen,MSF

		isnull(upper([dbo].[CnvShipStatus_Name](BuySell.ShippingOK)),'')

		FROM buysell with(nolock)

		inner join wksdetail with(nolock) on wksdetail.buysellno = buysell.buysellno

		inner join grade with(nolock) on grade.gradeid = wksdetail.gradeid

		inner join Counterparty C with(nolock) on C.CPID = buysell.CustID

		inner join Counterparty S with(nolock) on S.CPID = buysell.vendorID

		INNER JOIN [CompanyInfo] W WITH(NOLOCK) ON W.COID = Buysell.COID

		LEFT JOIN UnitType WITH(NOLOCK) on wksdetail.UnitType = UnitType.TypeID

		left join referrer with(nolock) on referrer.referrerno = buysell.referrerno1

		left join referrer PurchRep with(nolock) on PurchRep.referrerno = buysell.referrerno2

		left join BookSheet on BookSheet.BookingNo=buysell.BookingNo

		LEFT JOIN PO SalePO ON SPO = SalePO.PONumber

		LEFT JOIN PO PurchPO ON ppo = PurchPO.PONumber

		LEFT JOIN Address As CustAddr ON CustAddr.CpID = C.CpID AND CustAddr.Type = buysell.ShipToID

		LEFT JOIN Address As SupAddr ON SupAddr.CpID = S.CpID AND SupAddr.Type = BuySell.ShipFromID

		LEFT JOIN UserDef ON BuySell.UserID = UserDef.UserID

		WHERE 	

			(custid <> '(PLANT)')

			AND (@Dept = '' OR (@Dept = buysell.COID or w.CompanyDesc = @Dept ))			

			AND (@TradeType = '' OR BuySell.TradeType = @TradeType)

			AND (@Customer = '' or (buysell.CustID = @Customer or C.CompanyNm = @Customer))

			AND (@Supplier = '' or (buysell.VendorID = @Supplier or S.CompanyNm = @Supplier))

			AND (@Status = '' or (@Status = 'I' and buysell.status = 'I') or (@Status = 'N' and buysell.status not in ('I','X')) or (@Status = 'A' and buysell.status <> 'X'))

			AND 

				((@DateType = 'SHIP' and (buysell.ShippingDt >= @FromDt and buysell.ShippingDt <= @ToDt))

					or

				((@DateType = 'POST' and (buysell.InvoiceDt >= @FromDt and buysell.InvoiceDt <= @ToDt)))

				)

			AND

				(@InventoryType = '' or

				(@InventoryType = 'RCV' and (buysell.CustID = '(INV)'))

					or

				(@InventoryType = 'SHIP' and (buysell.VendorID = '(INV)'))

				)







		/* FLAG WORKSHEETS WITH NEGATIVE SALES */

		CREATE TABLE #NegativeSales(BuySellNo Char(12), UNIQUE (BuySellNo) )

		INSERT #NegativeSales select distinct buysellno FROM #REPORT where SAmount < 0

		UPDATE #REPORT SET IsNegSales = 1 FROM #REPORT INNER JOIN #NegativeSales ON #REPORT.BuySellNo = #NegativeSales.BuySellNo



		/* DETERMINE ALLOCATION FACTOR BY SALES AMOUNT */	

		update #REPORT set factor = (SAmount  / WksSales) where WksSales <> 0 AND IsNegSales = 0

		update #REPORT set factor = (SWeightLBS / WksWeightLBS) where (WksSales = 0 OR IsNegSales = 1) and WksWeightLBS <> 0 AND Factor = 0

		update #REPORT set factor = (PAmount / WksPurchases) where WksSales = 0 and WksPurchases <> 0 AND WksWeightLBS = 0 AND Factor = 0 



		--/* CALCULATE ALLOCATED EXPENSES */

		UPDATE #REPORT SET 

			ChargeBacks = Round(WksChgBack * Factor,2),

			Expenses	= Round(WksExp * Factor,2),

			SAdjs		= Round(WksSAdj * Factor,2),

			PAdjs		= Round(WksPAdj * Factor,2),

			FreightExp  = Round(FreightExp * Factor,2)





		/* CALCULATE GROSS PROFIT */		

		update #REPORT set TotalSale = SAmount + ChargeBacks + SAdjs

		update #REPORT set TotalCost = PAmount + Expenses + PAdjs

		update #REPORT set GrossProfit = Round(TotalSale - TotalCost,2)

		update #REPORT set Margin = case when TotalCost<>0 then  (GrossProfit/TotalCost)*100 else 0 end  --Alex 7/16/2020

		update #REPORT set Expenses = Expenses - FreightExp



		/* CALCULATE Gross/Net/Tare Wt*/

		--update #REPORT set WksGrossWt = sum(WKSDetail.GrossWt), WksTareWt = sum(WKSDetail.TareWt), WksNetWt = sum(WKSDetail.GrossWt) - sum(WKSDetail.TareWt) from WKSDetail where WKSDetail.BuySellNo = #REPORT.Buysellno



SELECT 

	rtrim(Buysellno) WorksheetNo,

	rtrim(Detailid) Detailid,

	rtrim(Department) Department,

	rtrim(TradeType)  TradeType,

	rtrim([Status]) [Status],

	rtrim(ShippingDt) ShippingDt,

	rtrim(cast(InvoiceDt as char)) InvoiceDt,

	rtrim(SoOrderDt) SoOrderDt,

	rtrim(PoOrderDt) PoOrderDt,

	rtrim(Customer) Customer,	

	rtrim(CustomerAddress) CustomerAddress,

	rtrim(ShipFromID) ShipFromID,	 

	rtrim(RelNo) RelNo,	 

	rtrim(ContainerNo) ContainerNo,	

	rtrim(Supplier) Supplier,

	rtrim(SupplierAddress) SupplierAddress,

	rtrim(ShipTo) ShipTo,

	rtrim(PickupNo) PickupNo, 	

	rtrim(BookingNo) BookingNo,

	rtrim(SoDestPort) SoDestPort,

	rtrim(PoDestPort) PoDestPort, 

	rtrim(isnull(ETA,'')) ETA,

	rtrim(ProductCat) ProductCat,

	rtrim(ProductGrp) ProductGrp,

	rtrim(Description) Description,

	rtrim(Specifications) Specifications,

	rtrim(InvoiceDesc) InvoiceDesc,

	rtrim(Units) Units,

	rtrim(UnitType) UnitType,

	rtrim(WksWeightLBS) WksWeightLBS,

	rtrim(SWeight) SWeight,

	rtrim(dbo.[CnvUOM_Names](SWeightUOM)) SWeightUOM,

	rtrim(PWeight) PWeight,

	rtrim(dbo.[CnvUOM_Names](PWeightUOM)) PWeightUOM, 

	rtrim(SWeightLBS) SWeightLBS,

	rtrim(WksSales) WksSales,	

	rtrim(WksChgBack) WksChgBack,

	rtrim(WksExp) WksExp,

	rtrim(WksPurchases) WksPurchases,

	rtrim(WksSAdj) WksSAdj,

	rtrim(WksPAdj) WksPAdj,

	rtrim(Factor) Factor,

	rtrim(SPO) SPO,

	rtrim(SCurrencyCd) SCurrencyCd,

	rtrim(SPrice) SPrice,

	rtrim(dbo.[CnvUOM_Names](SPriceUOM)) SPriceUOM,

	rtrim(SAmount) SAmount,

	rtrim(PO) PO,

	rtrim(PCurrencyCd) PCurrencyCd, 

	rtrim(PPrice) PPrice,

	rtrim(dbo.[CnvUOM_Names](PPriceUOM)) PPriceUOM,	

	rtrim(PAmount) PAmount,

	rtrim(ChargeBacks) ChargeBacks, 

	rtrim(FreightExp) FreightExp, 

	rtrim(Expenses) Expenses,

	rtrim(SAdjs) SAdjs,

	rtrim(PAdjs) PAdjs,

	rtrim(Interest) Interest,

	rtrim(BankFees) BankFees,	

	rtrim(Discount) Discount,	

	rtrim(Margin) Margin,

	rtrim(TotalSale) TotalSale,

	rtrim(TotalCost) TotalCost,

	rtrim(GrossProfit) GrossProfit,

	rtrim(NetProfit) NetProfit,

	rtrim([Sell-Rep]) [Sell-Rep], 

	rtrim([Buy-Rep]) [Buy-Rep],  

	rtrim(IsAgency) IsAgency,

	rtrim(IsNegSales) IsNegSales,

	rtrim(TranSrc) TranSrc,

	rtrim(UserDefined1) UserDefined1,

	rtrim(UserDefined2) UserDefined2,

	rtrim(UserDefined3) UserDefined3,

	rtrim(OwnerName) OwnerName,

	rtrim(POShippingTerms) POShippingTerms,

	rtrim(POShipVia) POShipVia,

	rtrim(POPaymentTerms) POPaymentTerms,

	rtrim(SOShippingTerms) SOShippingTerms,

	rtrim(SOShipVia) SOShipVia,

	rtrim(SOPaymentTerms) SOPaymentTerms,

	rtrim(CustID) CustID,

	rtrim(SupplierID) SupplierID,

	rtrim(EquipNo) EquipNo,

	rtrim(GrossWt) GrossWt,

	rtrim(TareWt) TareWt,

	rtrim(NetWt) NetWt,

	rtrim(Logistics) Logistics

FROM #REPORT
