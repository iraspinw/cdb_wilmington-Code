CREATE PROC [dbo].[cw_GetShipStatReportDetails] 

	@FromDate char(10), 

	@ToDate char(10), 

	@CoID char(2), 

	@SupplierID char(12), 

	@InvoiceNo varchar(20), 

	@ByDate int,

	@SupplierNm varchar(65) = '', 

	@SupplierType char(45) = '',

	@WtUOM char(1) = 'S'



AS



-- cw_GetShipStatReportDetails   '','','','43410','MARCEMILYTEST',0,'','','L'

-- cw_GetShipStatReportSuppliers '','','','43410','MARCCONVTEST2',0,'',''

-- cw_GetShipStatReportDetails '09/28/2018','09/28/2018','','43410','',1, '', ''



-- IsShipStmtMetal



--10/09/2015 RS: Created from cw_GetShippingStatementReport

--10/16/2015 RS: Default SupplierAddr to SupplierNm

--11/03/2015 DSH Altered InvoiceNo Filter.

--11/12/2015 RS: Fixed Adjustment Grade Description

--01/01/2016 RS: Show GLAccount.Name instead of GPLedger.AdjReason

--03/01/2016 RS: Replaced filter @SupplierNm with @SupplierID

--04/05/2016 RS: Added BLNo, MillRcvNo, GrossWt, TareWt for Scrap report

--08/29/2016 MEC: Added POReference for Scrap Metal

--09/02/2016 RS: Fixed PriceUOM

--07/17/2017 RS: Added @ReptType to have "By Location" option

--04/12/2018 MN: Change select order so expense is called after grade	

--06/05/2018 RS: Fixed last change (Order By) to keep all Transactions for the same load together



SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

SET ANSI_NULLS ON 

SET NOCOUNT ON



	DECLARE @ByProduct int 

	DECLARE @IncludeWaste bit

	DECLARE @ReverseOnly bit 

	DECLARE @Invoiced bit

	DECLARE @ReverseBatchNo int

	DECLARE @ExcludeZeroAmounts bit

	DECLARE @Debug bit

	DECLARE @IsDotNet bit

	DECLARE @IncExpenses bit 

	DECLARE @ReptType char(1)



	SET @ByProduct  = 3 

	SET @ReptType = ISNULL((SELECT optionValue FROM Options WHERE optionName = 'IsShipStmtMetal'),'')

	

	SET @IncludeWaste  = 0 

	SET @ReverseOnly  = 0 

	SET @Invoiced  = 0

	SET @ReverseBatchNo  = 0 

	SET @ExcludeZeroAmounts  = 0 

	SET @Debug  = 0 

	SET @IsDotNet  = 0 

	SET @IncExpenses  = 0

	DECLARE @WtPrecision int

	SET @WtPrecision = 0

	DECLARE @FmtNum varchar(20)

	SET @FmtNum = '#####0'

	

	DECLARE @Query nvarchar(4000)

	DECLARE @ParmDefinition nvarchar(4000)

	

	SET @ParmDefinition = '@FromDate DateTime, @ToDate DateTime, @CoID char(2), @SupplierID char(12), @InvoiceNo varchar(20), 

                @ByProduct int, @ByDate int, @IncludeWaste bit, @SupplierType char(45), @ReverseOnly bit, @Invoiced bit,

                @ReverseBatchNo int, @ExcludeZeroAmounts bit'



    CREATE TABLE #TempReport(        

		BuySellNo char(12),

        Jobno char(10),

        RelNo varchar(20),

        FShippingDt datetime,

        DetailDescription varchar(185),

        PWeight decimal(18,5),

        PweightUOM char(10),

        PPrice decimal(25,14),

        PUOM char(1),

        Amt money,

        PFxAmount money,

        [Type] char(1),

        PickupNo varchar(25),

        TrailerNo varchar(30),

        ShippingDt datetime,

        ContainerNo varchar(30),

        RecordID int,

        TruckNo varchar(25),

        ShippingNotes text ,

        Location char(45),

        Units int,

		UnitType varchar(30),  

        IsTrash int,   

        AltGradeName varchar(185),

        GradeID char(15),

        CustID char(12),

		PoCallNumber char(25),

		CurrencyCd char(3),

        SupplierNm varchar(65),

        SupplierCPID char(12),

        POUnitType varchar(30),

        WksAltGradeName varchar(185),

        IsPerEachCnv bit,

        PerEachCnvPrice money,

		ItemSeq INT,

		RowID INT Identity(1,1),

		BLNo varchar(25),

		MillRcvNo varchar(50),

		GrossWt decimal(18,5),

		TareWt decimal(18,5),

		POReference char(25),

		IsExpense bit

        )

        

	CREATE TABLE #TempSumReport(

		SupplierNM VARCHAR(65),

		SupplierCPID CHAR(12),

		Location VARCHAR(45),

		GradeID CHAR(15),

		GradeNm VARCHAR(85),		

		CurrencyCd CHAR(3),

		PWeightUOM CHAR(1),

		AvgPrice Decimal(22,12),		

		[Weight] Decimal(22,2),

		Rebate MONEY,

		AllocAmt MONEY,

		Category VARCHAR(30),

		NumOfItems INT,

		ISTrash INT,

		[Type] CHAR(1),

		IsStaticPrice INT

	)        



	CREATE TABLE #TempGradeCons(

		BuySellNo CHAR(12),

		GradeID CHAR(15),

		UnitType CHAR(30),

		AltGradeName varchar(185),

		WksAltGradeName varchar(185),

		RowID INT,

		ItemCnt INT, 

		HCAmount MONEY, 

		FXAmount MONEY, 

		Quantity DECIMAL(22,5), 

		Units INT,

		OrigPrice MONEY

		)



	CREATE TABLE #_PreserveOrigPrice(

		BuySellNo CHAR(12),

		GradeID CHAR(15),

		UnitType CHAR(30),

		--AltGradeName varchar(185),

		--WksAltGradeName varchar(185),

		PPrice MONEY, 

		ItemCnt INT

		)

	





	-- A.) GET PURCHASE DETAILS.

   ----------------------------

    SET @Query = 'INSERT #TempReport

	SELECT 

        WksDetail.BuySellNo, 

        cast(Grading.ScaleTicketNo AS varchar(50)) as Job, 

        BuySell.RelNo, 

		BuySell.ShippingDt AS fShippingDt, 

		Case When COALESCE(InvoiceDesc,'''') = '''' Then Grade.[Description] Else COALESCE(InvoiceDesc,'''') End as DetailDescription, 

        wksDetail.PWeight, wksDetail.PWeightUOM, 

        wksDetail.PPrice, 

        wksDetail.PUOM, 

        wksDetail.PAmount as Amt, wksDetail.PFxAmount, 

		''S'' as [Type], 

		BuySell.PickupNo, 

		Grading.EquipmentNo,       		

		BuySell.ShippingDt, 

		WksDetail.ContainerNo, 

		wksDetail.DetailID, 

		Grading.TruckNo AS TruckNo,

		BuySell.ShippingNotes,

		BuySell.ShipFromID AS Location,

		WksDetail.Units,

        ISNULL(UnitType.UnitType,'''') AS UnitType,        

        ISNULL(Grade.IsWaste,0) as IsTrash, 		

		COALESCE(PODesc,'''') AS AltGradeName,

		WksDetail.GradeID, 

		BuySell.CustID, 

		'''' RefNo,

        WksDetail.PcurrencyCd,

        Counterparty.Companynm,

        Counterparty.CPID,

        ISNULL(u.UnitType,'''') AS UnitType,

        COALESCE(InvoiceDesc,'''') AS WksAltGradeName,

        WksDetail.IsPerEachCnv,

        WksDetail.PerEachCnvPrice,

		ROW_NUMBER() OVER(PARTITION BY wksDetail.buysellno, wksDetail.GradeID, wksDetail.UnitType, InvoiceDesc, COALESCE(PODesc,'''') 

			ORDER BY  wksDetail.buysellno, wksDetail.GradeID, wksDetail.UnitType, InvoiceDesc, COALESCE(PODesc,'''') ) AS ItemSeq,

		BuySell.BLNo,

		BuySell.MillRcvNo,

		wksDetail.GrossWt,

		wksDetail.TareWt,

		POReference,

		0

	FROM dbo.WksDetail     

    INNER JOIN dbo.BuySell WITH(NOLOCK) ON dbo.wksDetail.BuySellNo = dbo.BuySell.BuySellNo    

	INNER JOIN dbo.Grade WITH(NOLOCK) ON dbo.Grade.GradeID = dbo.wksDetail.GradeID

    INNER JOIN dbo.Counterparty ON dbo.BuySell.VendorID = dbo.Counterparty.CPID

	LEFT JOIN dbo.UnitType WITH(NOLOCK) ON dbo.WksDetail.UnitType = dbo.UnitType.TypeID

	LEFT JOIN dbo.Grading WITH(NOLOCK) ON dbo.Grading.GradingID = dbo.BuySell.GradingID

	LEFT JOIN dbo.PODetail  WITH (NOLOCK) ON dbo.PODetail.PO_ID = dbo.WksDetail.PPO_ID 

	LEFT JOIN dbo.PO WITH (NOLOCK) ON dbo.PO.PONumber = dbo.BuySell.RefPONo 

	LEFT JOIN dbo.UnitType u WITH(NOLOCK) ON dbo.PODetail.UnitType = u.TypeID

	WHERE ISNULL(BUYSELL.[status],'''')<>''X'' '

	IF NOT @ByDate = 0 AND @ReverseBatchNo = 0

	BEGIN

		IF @InvoiceNo = '' BEGIN 

			IF @ByDate = 1 

				SET @Query = @Query + 'AND (BuySell.InvoiceDt >= @FromDate AND BuySell.InvoiceDt <= @ToDate) '

			IF @ByDate = 2

				SET @Query = @Query + 'AND (BuySell.ShippingDt >= @FromDate AND BuySell.ShippingDt <= @ToDate) '

		END

     END               

    IF NOT @COID = ''

        SET @Query = @Query + 'AND (@COID = BuySell.COID) '

	IF NOT @SupplierID = ''

		SET @Query = @Query + 'AND (@SupplierID = BuySell.VendorID) '

	IF NOT @SupplierType = ''

		SET @Query = @Query + 'AND (BuySell.ShipFromID = @SupplierType) '

    IF NOT @InvoiceNo = ''

		SET @Query = @Query + 'AND EXISTS(SELECT PayableID FROM Payables WITH(NOLOCK) WHERE Payables.BuySellNo = BuySell.BuySellNo AND Payables.ItemIDSource = ''S'' AND (reversed=0 OR Reversed IS NULL) AND Payables.InvoiceNo = @InvoiceNo) '

	IF NOT @Invoiced = 0

		SET @Query = @Query + 'AND BuySell.Status = ''I'' '





	/* When ReverseOnly=1 -- INCLUDE ONLY UNPOSTED PAYABLES. */

    IF @ReverseOnly = 1 

		SET @Query = @Query + 'AND NOT EXISTS(SELECT PayableID FROM Payables WITH(NOLOCK) WHERE Payables.BuySellNo = BuySell.BuySellNo 

		AND Payables.ItemIDSource = ''S'' 

		AND	Posted = 1 AND (reversed=0 OR Reversed IS NULL) ) 		

		AND (Counterparty.ReverseBilling = 1) '

/*	

	-- Need to look at adding ReverseBatch Number to BuySell Table and removing it from Payables.

	-- Should be changed to filter on a new ReverseBatch field on Worksheet.

	IF NOT @ReverseBatchNo = 0

		SET @Query = @Query + 'AND EXISTS(SELECT PayableID FROM Payables WITH(NOLOCK) WHERE Payables.ReverseBatchNo = @ReverseBatchNo AND Payables.BuySellNo = BuySell.BuySellNo) '	

*/	



	/* Only Include Payables for ReverseBatch */	

	IF NOT @ReverseBatchNo = 0

   		SET @Query = @Query + 'AND BuySell.ReverseBatchNo = @ReverseBatchNo '

		

		







	

		



	IF @Debug = 1 

		PRINT @Query

	ELSE

		EXEC sp_executesql @Query,@ParmDefinition,@FromDate=@FromDate,@ToDate=@ToDate,@CoID=@CoID,@SupplierID=@SupplierID,@InvoiceNo=@InvoiceNo, 

			@ByProduct=@ByProduct,@ByDate=@ByDate,@IncludeWaste=@IncludeWaste,@SupplierType=@SupplierType,@ReverseOnly=@ReverseOnly,

			@Invoiced=@Invoiced,@ReverseBatchNo=@ReverseBatchNo,@ExcludeZeroAmounts=@ExcludeZeroAmounts





	-- B.) GET DELIVERY PAYABLES

    -----------------------------

    SET @Query = 'INSERT #TempReport

	SELECT Payables.BuySellNo, 

    '''', 

    BuySell.RelNo,  

    BuySell.ShippingDt AS fShippingDt, 

	''Expense-'' + isnull(InvoiceDesc, '''') as detailDescription, 

    0, 

    '''', 

    0, 

    '''', 

    Payables.Amount, 

    Payables.FxAmount, 

    ''D'', 

    BuySell.PickupNo, 

    Wksdelivery.TrailerNo, 

    BuySell.ShippingDt, 

    '''' AS ContainerNo, 

    Payables.PayableID, 

    '''',

	BuySell.ShippingNotes,

    BuySell.ShipFromID,

    0 AS Units,

    '''' AS UnitType,

    0 as IsTrash, 

    '''' AS AltGradeName,  

    '''' AS GradeID, 

    BuySell.CustID,

    '''',

    Payables.CurrencyCd,

    Counterparty.Companynm,

    Counterparty.CPID,

    '''' AS POUnitType,

     '''' AS WksAltGradeName,

     0,0

	 ,0,

 	BuySell.BLNo,

	BuySell.MillRcvNo,0,0,'''',

	1

	FROM dbo.Payables WITH(NOLOCK) 

	INNER JOIN dbo.WksDelivery WITH(NOLOCK)  ON dbo.wksDelivery.BuySellNo = dbo.Payables.BuySellNo AND dbo.wksDelivery.ItemID = dbo.Payables.ItemID 

	INNER JOIN  dbo.BuySell ON dbo.BuySell.BuySellNo = dbo.Payables.BuySellNo	

    INNER JOIN dbo.Counterparty ON dbo.WksDelivery.VendorID = dbo.Counterparty.CPID

	WHERE ISNULL(BUYSELL.[status],'''')<>''X'' 

	AND Payables.ItemIDSource = ''D'' 

	AND (Reversed=0 OR Reversed IS NULL) 

	AND NOT EXISTS(SELECT PayableID FROM Payables WHERE ChgBackWks = BuySell.BuySellNo) '

	IF NOT @ByDate = 0 AND @ReverseBatchNo = 0 

	BEGIN

		IF @ByDate = 1 

			SET @Query = @Query + 'AND (BuySell.InvoiceDt >= @FromDate AND BuySell.InvoiceDt <= @ToDate) '

		IF @ByDate = 2

			SET @Query = @Query + 'AND (BuySell.ShippingDt >= @FromDate AND BuySell.ShippingDt <= @ToDate) '

    END

    IF NOT @COID = ''

		SET @Query = @Query + 'AND (@COID = BuySell.COID) '

	IF NOT @SupplierID = ''

		SET @Query = @Query + 'AND (@SupplierID = Payables.CpID) '

	IF NOT @SupplierType = ''

		SET @Query = @Query + 'AND (BuySell.ShipFromID = @SupplierType) '

	IF NOT @InvoiceNo = ''

		SET @Query = @Query + 'AND (Payables.InvoiceNo = @InvoiceNo) '

	IF NOT @ReverseOnly = 0

		SET @Query = @Query + 'AND 

			(

				EXISTS(SELECT * FROM Payables WITH(NOLOCK) WHERE (Posted = 0 OR Posted IS NULL) AND Payables.BuySellNo = BuySell.BuySellNo AND (reversed=0 OR Reversed IS NULL)) AND

				 EXISTS(SELECT ReverseBilling FROM Counterparty WHERE CpID = Payables.CpID AND ReverseBilling = 1)

			) '

	IF NOT @Invoiced = 0

		SET @Query = @Query + 'AND (BuySell.Status = ''1'') '

	IF NOT @ReverseBatchNo = 0

		SET @Query = @Query + 'AND (Payables.ReverseBatchNo = @ReverseBatchNo)	'		

	

	IF @Debug = 1 

		PRINT @Query

	ELSE

		EXEC sp_executesql @Query,@ParmDefinition,@FromDate=@FromDate,@ToDate=@ToDate,@CoID=@CoID,@SupplierID=@SupplierID,@InvoiceNo=@InvoiceNo, 

			@ByProduct=@ByProduct,@ByDate=@ByDate,@IncludeWaste=@IncludeWaste,@SupplierType=@SupplierType,@ReverseOnly=@ReverseOnly,

			@Invoiced=@Invoiced,@ReverseBatchNo=@ReverseBatchNo,@ExcludeZeroAmounts=@ExcludeZeroAmounts







	-- B.) GET EXPENSE ALLOCATION PAYABLES

    -----------------------------

    SET @Query = 'INSERT #TempReport

	SELECT Payables.BuySellNo, 

    '''', 

    '''',  

    ExpenseAllocation.TransDate AS fShippingDt, 

	''Expense-Allocation'' as detailDescription, 

    0, 

    '''', 

    0, 

    '''', 

    Payables.Amount, 

    Payables.FxAmount, 

    ''E'', 

    '''', 

    '''', 

    ExpenseAllocation.TransDate, 

    '''' AS ContainerNo, 

    Payables.PayableID, 

    '''',

	'''',

    '''',

    0 AS Units,

    '''' AS UnitType,

    0 as IsTrash, 

    '''' AS AltGradeName,  

    '''' AS GradeID, 

    '''' CustID,

    '''',

    Payables.CurrencyCd,

    Counterparty.Companynm,

    Counterparty.CPID,

    '''' AS POUnitType,

    '''' AS WksAltGradeName,

    0,0,0,

 	'''',

	'''',0,0,'''',1

	FROM dbo.Payables WITH(NOLOCK) 

	INNER JOIN  dbo.ExpenseAllocation ON dbo.ExpenseAllocation.BuySellNo = dbo.Payables.BuySellNo	

	LEFT JOIN dbo.Counterparty ON Payables.CpID = dbo.Counterparty.CPID

	WHERE Payables.ItemIDSource = ''E'' 

	AND Payables.Posted = 1 

	AND (Reversed=0 OR Reversed IS NULL) '

	IF NOT @ByDate = 0 AND @ReverseBatchNo = 0 

	BEGIN

		IF @ByDate = 1 

			SET @Query = @Query + 'AND (ExpenseAllocation.PostingDt >= @FromDate AND ExpenseAllocation.PostingDt <= @ToDate) '

		IF @ByDate = 2

			SET @Query = @Query + 'AND (ExpenseAllocation.TransDate >= @FromDate AND ExpenseAllocation.TransDate <= @ToDate) '

    END

    IF NOT @COID = ''

		SET @Query = @Query + 'AND (@COID = ExpenseAllocation.COID) '

	IF NOT @SupplierID = ''

		SET @Query = @Query + 'AND (@SupplierID = Payables.CpID) '

	IF NOT @InvoiceNo = ''

		SET @Query = @Query + 'AND (Payables.InvoiceNo = @InvoiceNo) '

	

	IF @Debug = 1 

		PRINT @Query

	ELSE

		EXEC sp_executesql @Query,@ParmDefinition,@FromDate=@FromDate,@ToDate=@ToDate,@CoID=@CoID,@SupplierID=@SupplierID,@InvoiceNo=@InvoiceNo, 

			@ByProduct=@ByProduct,@ByDate=@ByDate,@IncludeWaste=@IncludeWaste,@SupplierType=@SupplierType,@ReverseOnly=@ReverseOnly,

			@Invoiced=@Invoiced,@ReverseBatchNo=@ReverseBatchNo,@ExcludeZeroAmounts=@ExcludeZeroAmounts







	-- B.) GET EXPENSE CONVERTING PAYABLES

    -----------------------------

    SET @Query = 'INSERT #TempReport

	SELECT Payables.BuySellNo, 

    '''', 

    '''',  

    ConvertingJob.JobDate AS fShippingDt, 

	''Converting-Expense'' as detailDescription, 

    0, 

    '''', 

    0, 

    '''', 

    Payables.Amount, 

    Payables.FxAmount, 

    ''C'', 

    '''', 

    '''', 

    ConvertingJob.PostDt, 

    '''' AS ContainerNo, 

    Payables.PayableID, 

    '''',

	'''',

    '''',

    0 AS Units,

    '''' AS UnitType,

    0 as IsTrash, 

    '''' AS AltGradeName,  

    '''' AS GradeID, 

    '''' CustID,

    '''',

    Payables.CurrencyCd,

    Counterparty.Companynm,

    Counterparty.CPID,

    '''' AS POUnitType,

    '''' AS WksAltGradeName,

    0,0,0,

 	'''',

	'''',0,0,'''',1

	FROM dbo.Payables WITH(NOLOCK) 

	INNER JOIN dbo.ConvertingJob ON dbo.ConvertingJob.BuySellNo = dbo.Payables.BuySellNo	

	LEFT JOIN dbo.Counterparty ON Payables.CpID = dbo.Counterparty.CPID

	WHERE Payables.ItemIDSource = ''C'' 

	AND Payables.Posted = 1 

	AND (Reversed=0 OR Reversed IS NULL) '

	IF NOT @ByDate = 0 AND @ReverseBatchNo = 0 

	BEGIN

		IF @ByDate = 1 

			SET @Query = @Query + ' AND (ConvertingJob.PostDt >= @FromDate AND ConvertingJob.PostDt <= @ToDate) '

		IF @ByDate = 2

			SET @Query = @Query + ' AND (ConvertingJob.JobDate >= @FromDate AND ConvertingJob.JobDate <= @ToDate) '

    END

    IF NOT @COID = ''

		SET @Query = @Query + ' AND (@COID = ConvertingJob.COID) '

	IF NOT @SupplierID = ''

		SET @Query = @Query + ' AND (@SupplierID = Payables.CpID) '

	IF NOT @InvoiceNo = ''

		SET @Query = @Query + ' AND (Payables.InvoiceNo = @InvoiceNo) '

	

	IF @Debug = 1 

		PRINT @Query

	ELSE

		EXEC sp_executesql @Query,@ParmDefinition,@FromDate=@FromDate,@ToDate=@ToDate,@CoID=@CoID,@SupplierID=@SupplierID,@InvoiceNo=@InvoiceNo, 

			@ByProduct=@ByProduct,@ByDate=@ByDate,@IncludeWaste=@IncludeWaste,@SupplierType=@SupplierType,@ReverseOnly=@ReverseOnly,

			@Invoiced=@Invoiced,@ReverseBatchNo=@ReverseBatchNo,@ExcludeZeroAmounts=@ExcludeZeroAmounts









	-- C.) GET ADJUSTMENTS (By Reporting Date).

    ------------------------------------------------

    SET @Query = 'INSERT #TempReport 

	SELECT 

        BuySell.BuySellNo, 

        cast(Grading.ScaleTicketNo AS varchar(50)) as Job,

        '''', 

        GPLedger.AdjReportDt AS fShippingDt, 

	    ''Adjustment-'' + GLAccount.Name as DetailDescription, 

        0, '''', 0, '''', 

        GPLedger.Cost,  

        GPLedger.AdjFxAmount, 

        ''A'', 

        BuySell.PickupNo, 

        '''', 

        BuySell.ShippingDt, 

        '''' AS ContainerNo, 

        GPLedger.LedgerID, 

        '''',

	    rtrim(dbo.getfield(GPLedger.AdjNotes,27,char(9))),   '--+' '+convert(varchar(200),BuySell.ShippingNotes), ''''COMMENTED OUT BF DONT NEED THIS ON ADJUSTMENT

        + 'BuySell.ShipFromID,

        0 AS Units,

        '''' AS UnitType,

        0 as IsTrash,

        GLAccount.Name AS AltGradeName, 

        '''' AS GradeID, 

        BuySell.CustID, 

        '''' RefNo,

        AdjCurrencyCd,

        Counterparty.Companynm,

        Counterparty.CPID,

        '''' AS POUnitType,

  '''' AS WksAltGradeName,

        0,0

		,0,

 		BuySell.BLNo,

		BuySell.MillRcvNo,0,0,'''',

		0

    FROM dbo.GPLedger WITH(NOLOCK) 

    INNER JOIN  dbo.BuySell WITH(NOLOCK) ON dbo.BuySell.BuySellNo = dbo.GPLEDGER.BuySellNo	

    INNER JOIN dbo.Counterparty ON dbo.GPLedger.AdjCPID = dbo.Counterparty.CPID

	LEFT JOIN dbo.GLAccount WITH(NOLOCK) ON dbo.GPLedger.GLAcct = dbo.GLAccount.GLAcct

	LEFT JOIN dbo.Grading WITH(NOLOCK) ON dbo.Grading.GradingID = dbo.BuySell.GradingID

	WHERE GPLedger.AdjType = ''P'' 

		AND GPLedger.Source = ''A'' 			

		AND GPLedger.AdjPosted = ''Y'' 

		AND ISNULL(BUYSELL.[status],'''')<>''X'' '

		

		IF NOT @COID = ''

			SET @Query = @Query + 'AND (@COID = BuySell.COID) '

		IF NOT @SupplierID = ''

			SET @Query = @Query + 'AND (@SupplierID = GPledger.AdjCPID) '

		IF NOT @SupplierType = ''

			SET @Query = @Query + 'AND (BuySell.ShipFromID = @SupplierType) '

		IF NOT @InvoiceNo = ''

			SET @Query = @Query + 'AND EXISTS(SELECT PayableID FROM Payables WITH(NOLOCK) WHERE Payables.ItemID = GPLedger.LedgerID AND Payables.BuySellNo = BuySell.BuySellNo AND Payables.InvoiceNo = @InvoiceNo) '

		IF NOT @ReverseOnly = 0

			SET @Query = @Query + 'AND 

				(

					 EXISTS(SELECT * FROM Payables WITH(NOLOCK) WHERE (Posted = 0 OR Posted IS NULL) AND Payables.BuySellNo = BuySell.BuySellNo AND (reversed=0 OR Reversed IS NULL)) AND

					 EXISTS(SELECT ReverseBilling FROM Counterparty WHERE CpID = GPLedger.AdjCpID AND ReverseBilling = 1)

				) '

		IF NOT @Invoiced = 0 

			SET @Query = @Query + 'AND (BuySell.Status = ''I'') '

		IF NOT @ReverseBatchNo = 0

			SET @Query = @Query + 'AND (GPLedger.ReverseBatchNo = @ReverseBatchNo) '	

		IF @ReverseBatchNo = 0 AND @InvoiceNo = ''

			SET @Query = @Query + 'AND (GPLedger.AdjReportDt >= @FromDate AND GPLedger.AdjReportDt <= @ToDate) '

		

	IF @Debug = 1 

		PRINT @Query

	ELSE

		EXEC sp_executesql @Query,@ParmDefinition,@FromDate=@FromDate,@ToDate=@ToDate,@CoID=@CoID,@SupplierID=@SupplierID,@InvoiceNo=@InvoiceNo, 

			@ByProduct=@ByProduct,@ByDate=@ByDate,@IncludeWaste=@IncludeWaste,@SupplierType=@SupplierType,@ReverseOnly=@ReverseOnly,

			@Invoiced=@Invoiced,@ReverseBatchNo=@ReverseBatchNo,@ExcludeZeroAmounts=@ExcludeZeroAmounts







	-- D.) GET UN-POSTED ADJUSTMENTS (By Worksheet Shipping Date).

    --------------------------------------------------------------

    IF @InvoiceNo = '' 

    BEGIN

		SET @Query = 'INSERT #TempReport 

		SELECT 

			BuySell.BuySellNo, 

			cast(Grading.ScaleTicketNo AS varchar(10)) as Job,

			'''', 

			BuySell.ShippingDt AS fShippingDt, 

			GLAccount.Name as DetailDescription, 

			0, '''', 0, '''', 

			GPLedger.Cost,   

			GPLedger.AdjFxAmount, 

			''A'', 

			BuySell.PickupNo, 

			'''', 

			BuySell.ShippingDt, 

			'''' AS ContainerNo, 

			GPLedger.LedgerID, 

			'''',
