CREATE proc [dbo].[cw_GetSystemLog] @FromDt datetime = '', @ToDt datetime = '', @User varchar(30) = '', @Object varchar(30) = '', @ObjectID varchar(30) = '' as

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

SET NOCOUNT ON;

--11/11/2015 MCM: Added @FromDt, @ToDt, @User, @Object, @ObjectID

--04/07/2021 MEC: Added Timezone Assignment to datestamp





	SELECT 

		Case When TimeZone.TimeZoneNm is null Then Datestamp Else DateAdd(HOUR, (IsNull(TimeZone.GMTOffsetVAL,0) + 5), Datestamp) End AS [Date], 

		RTRIM([Object]) AS [Object], ObjectID AS [ID], UserNm AS [User Name], Comment

		FROM systemlog s

		inner join userdef on userdef.userid= s.userid

		left join TimeZone on UserDef.TimeZoneNm = TimeZone.TimeZoneNm

		WHERE (s.UserID = @User OR @User = '')

		AND (CAST(Datestamp as Date) >= @FromDt OR @FromDt = '')

		AND (CAST(Datestamp as Date) <= @ToDt OR @ToDt = '')

		AND (ObjectID = @ObjectID OR @ObjectID = '')

		AND ([Object] = @Object OR @Object = '')

		order by datestamp desc
