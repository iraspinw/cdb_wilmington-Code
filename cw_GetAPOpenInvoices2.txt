CREATE PROCEDURE [dbo].[cw_GetAPOpenInvoices2] 

    @SupplierID varchar(12), 

    @ReportType varchar(12), 

    @ReportingByAPDate varchar(12), 

    @Include varchar(12), 

    @FromDt datetime, 

    @ToDt datetime,

    @COID varchar(2), 

    @TradeType varchar(25), 

    @SortBy char(12) ='', 

    @CurrencyCd char(3) ='',

    @InclUnpostedAdj bit = 0

AS

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

SET ANSI_WARNINGS OFF

-- 11/7/2016 DSH: Rewritten for .Net 

-- 04/13/2022 DSH Add multicurrency columns

-- 09/13/2022 DSH Added Multicurrencyu filters



-- 5/2/25, checking

 --select top(10) * from payables where buysellno = '36317'

 --select top(10) * from receipt where buysellno = '36317'



IF @FromDt = '' SET @FromDt = '07/01/2023'

IF @ToDt = '' SET @ToDt = GETDATE()



	CREATE TABLE #OPENAPREPORT(

		OpenDt DATE,

		PostDt DATE,

		Src CHAR(3),

		TransNo CHAR(12),

		TranType CHAR(3),

		VendorNm VARCHAR(65),

		AccountDesc VARCHAR(100),

		HCAmount MONEY,

		PickupNo varchar(25),

		RelNo varchar(20),

		FxAmount MONEY,

		CurrencyCd CHAR(3),

		FxRate DECIMAL(22,8)

		,COID char(2)

		,SupplierID varchar(12)

	)





	-- OPEN WORKSHEET EXPENSES

	INSERT #OPENAPREPORT(OpenDt,PostDt,Src,TransNo,TranType,VendorNm,AccountDesc,HCAmount,PickupNo, RelNo,FxAmount,CurrencyCd,FxRate, COID, SupplierId)

    SELECT  

		BuySell.InvoiceDt, 			

		Payables.RcvDt, 			

		case when custid='(INV)' then 'Rec' when vendorid='(INV)' then 'Shp' else 'Wks' end,

		Payables.BuySellNo, 

		CASE WHEN Payables.ItemIDSource ='S' THEN 'Pur' ELSE 'Exp' END, 

		CounterParty.CompanyNm, 

		Payables.glacct + ' ' + GLAccount.Name, 	

		Payables.Amount,

		BuySell.PickupNo,

		BuySell.RelNo,

		Payables.FxAmount,

		Payables.CurrencyCd,		

		Round(Payables.FxAmount / Payables.Amount,3)

		, buysell.COID

		,payables.CpID

	FROM 

	    Payables INNER JOIN CounterParty WITH(NOLOCK) ON CounterParty.CpID = Payables.CpID

		INNER JOIN BuySell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo 

        LEFT JOIN GLAccount WITH(NOLOCK) ON Payables.GLAcct = GLAccount.GLAcct 

	WHERE 

        BuySell.Status = 'I'  

		AND Payables.ItemIDSource <> 'A'   --Exclude Adjustments  

		AND Amount <> 0        

        AND (@COID = '' OR  BuySell.COID = @COID) 

        AND (@TradeType = '' OR @TradeType = BuySell.TradeType)

        AND (@SupplierID = '' OR Payables.CpID = @SupplierID)  

		AND (@CurrencyCd= '' OR Payables.CurrencyCd = @CurrencyCd)		

		AND (Payables.Posted=0  



		--02/15/24 I.R.

		--OR (Payables.Posted=1 AND Payables.InvoiceDt > @ToDt ))	

		

		OR (Payables.Posted=1 AND Payables.RcvDt > @ToDt ))		



		AND(

			(@ReportType = 'BYASOFDATE' AND BuySell.InvoiceDt <= @ToDt  ) 

			OR

			(@ReportType = 'BYPERIOD' AND (BuySell.InvoiceDt >= @FromDt AND BuySell.InvoiceDt <= @ToDt))

		)



		--SELECT Payables.InvoiceDt, * FROM Payables WHERE buysellno = 'PR-2497'



	-- OPEN EXPENSE ALLOCATIONS COSTS

	INSERT #OPENAPREPORT(OpenDt,PostDt,Src,TransNo,TranType,VendorNm,AccountDesc,HCAmount,PickupNo, RelNo,FxAmount,CurrencyCd,FxRate, COID, SupplierID)

    SELECT  

		EAC.PostingDt OpenDt, 			

		Payables.RcvDt, 			

		'Alc',

		Payables.BuySellNo, 

		'Exp', 

		CounterParty.CompanyNm, 

		Payables.glacct + ' ' + GLAccount.Name, 	

		Payables.Amount,

		'' PickupNo,

		'' RelNo,

		Payables.FxAmount,

		Payables.CurrencyCd,

		CASE WHEN Payables.FxAmount <> 0 THEN Payables.Amount/ Payables.FxAmount ELSE 1 END

		, eac.CoID

		, payables.CpID

	FROM Payables 

		INNER JOIN ExpenseAllocation EAC ON EAC.BuySellNo = Payables.BuySellNo  AND ItemIDSource = 'E' 

		INNER JOIN Counterparty ON Payables.cpid = Counterparty.cpid		

		LEFT JOIN GLAccount ON Payables.GLacct = GLaccount.glacct  	

	WHERE 

        EAC.ISPosted = 1 AND Amount <> 0		

        AND (@COID = '' OR  EAC.COID = @COID) 

        AND (@TradeType = '' OR @TradeType = EAC.TradeType)

        AND (@SupplierID = '' OR Payables.CpID = @SupplierID)  

		AND (@CurrencyCd= '' OR Payables.CurrencyCd = @CurrencyCd)



		AND (Payables.Posted=0  

		--02/15/24 I.R.

		--OR (Payables.Posted=1 AND Payables.InvoiceDt > @ToDt ))	

		

		OR (Payables.Posted=1 AND Payables.RcvDt > @ToDt ))		



		AND(

			(@ReportType = 'BYASOFDATE' AND EAC.PostingDt <= @TODT  ) 

			OR

			(@ReportType = 'BYPERIOD' AND (EAC.PostingDt >= @FromDt AND EAC.PostingDt <= @ToDt))

		)





	-- OPEN CONVERTING JOB COSTS

	INSERT #OPENAPREPORT(OpenDt,PostDt,Src,TransNo,TranType,VendorNm,AccountDesc,HCAmount,PickupNo,RelNo,FxAmount,CurrencyCd,FxRate, COID, SupplierID)

	SELECT 

		CNV.PostDt OpenDt, 			

		Payables.RcvDt, 			

		'Cnv',

		Payables.BuySellNo, 

		'Exp', 

		CounterParty.CompanyNm, 

		Payables.glacct + ' ' + GLAccount.Name, 	

		Payables.Amount

		,

		BuySell.PickupNo,

		BuySell.RelNo,

		Payables.FxAmount,

		Payables.CurrencyCd,

		CASE WHEN Payables.FxAmount <> 0 THEN Payables.Amount/ Payables.FxAmount ELSE 1 END

		, BuySell.coid

		, payables.CpID



	FROM Payables 

		INNER JOIN ConvertingJob CNV ON CNV.BuySellNo = Payables.BuySellNo AND ItemIDSource = 'C' 

		INNER JOIN Counterparty ON Payables.cpid = Counterparty.cpid

		LEFT JOIN BuySell WITH(NOLOCK) ON Payables.BuySellNo = BuySell.BuySellNo 

		LEFT JOIN GLAccount ON Payables.GLacct = GLaccount.glacct  	

	WHERE 

		CNV.[Status] = 'I' AND Amount <> 0 				

		AND (CNV.COID = @CoID OR @CoID = '') 

		AND (CNV.TradeType = @TradeType OR @TradeType = '')

		AND (@CurrencyCd= '' OR Payables.CurrencyCd = @CurrencyCd)

		AND (@SupplierID = '' OR Payables.CpID = @SupplierID)

		AND (Payables.Posted=0  

		--02/15/24 I.R.

		--OR (Payables.Posted=1 AND Payables.InvoiceDt > @ToDt ))	

		

		OR (Payables.Posted=1 AND Payables.RcvDt > @ToDt ))		

		AND(

			(@ReportType = 'BYASOFDATE' AND CNV.PostDt <= @ToDt) 

			OR

			(@ReportType = 'BYPERIOD' AND (CNV.PostDt >= @ToDt AND CNV.PostDt <= @ToDt))

		)



		TRUNCATE TABLE CietradeOpenAPInvoices



		INSERT INTO CietradeOpenAPInvoices

		( [Billing Company]



      ,[WksNo]

      ,[Vendor Name]

	  ,[CpiD]

      ,[Open Date]

      ,[Post Date]

      ,[Account Desc]

      ,[Outstanding Payables Amount]

      ,[Transaction Type]

      ,[Open Date Period])

	SELECT (SELECT CompanyNm from CompanyInfo WHERE COID= #OPENAPREPORT.coid) [Billing Company]

	    ,TransNo As [WksNo]

	    ,VendorNm As [Vendor Name]

		,SupplierID

		, OpenDt As [Open Date],

		ISNULL(CAST(PostDt AS VARCHAR(10)), '') As [Post Date]



		--,TransNo As [WksNo]

		,AccountDesc As [Account Desc],

		HCAmount As [Outstanding Payables Amount]

		,

		--Src [Source],

		TranType As [Transaction Type]

		, FORMAT(OpenDt, 'MM/yyyy') [Open Date Period]

		--,



		--,PickupNo As [Pickup No],

		--RelNo As [Release No],

		--FxAmount,

		--CurrencyCd,

		--FxRate



			--

		

	from #OPENAPREPORT 



	--WHERE TransNo = 'PR-9783' 

	ORDER BY 1, 2, 3



	--OpenDt, TransNo
