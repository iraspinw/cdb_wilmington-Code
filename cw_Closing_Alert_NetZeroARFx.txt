

CREATE PROC [dbo].[cw_Closing_Alert_NetZeroARFx] @DoColumnDefOnly BIT, @FromDt datetime,@ToDT datetime,@COID char(2)

AS

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

--Expenses Against Unposted Sales Details

--10/11/2012 RS: Created for Closing Adjustments Ledger.



	IF @COID IS NULL	

		SET @COID = ''



    -- GET ACCOUNTING SYSTEM.

    Declare @AccountingSystem varchar(30)    

    SELECT TOP 1 @AccountingSystem = AccountingSystem from AccountingCompany WITH(NOLOCK) WHERE AccountingSystem IS NOT NULL AND AccountingSystem <> '(NONE)'



    -- GET FUNCTIONAL CURRENCY

    Declare @SysCurrency char(3)

    SELECT @SysCurrency = CurrCode from systeminfo WITH(NOLOCK) 



	/* DEFINE REPORT COLUMNS SECTION. */

	IF @DoColumnDefOnly = 1 BEGIN

	

		CREATE TABLE #ReportDef(

			ColName char(25),

			ColAlign char(1),

			ColWidth int,

			ColDataType char(10),

			ColNumFormat char(15)

		)

		

		INSERT #ReportDef SELECT 'Customer','L',2600,'TEXT',''

		INSERT #ReportDef SELECT 'Check No.','L',1250,'TEXT',''		

		INSERT #ReportDef SELECT 'Payment Date','L',1250,'DATE',''

		INSERT #ReportDef SELECT 'Amount','R',1300,'NUM','#,###,###0.00'		

		INSERT #ReportDef SELECT 'Currency','L',700,'TEXT',''

		INSERT #ReportDef SELECT 'HC Amount','R',1300,'NUM','#,###,###0.00'				

		

		SELECT ColName, ColAlign, ColWidth, ColDataType, ColNumFormat FROM #ReportDef

		RETURN

	END	

	

	

	CREATE TABLE #Report( 	

		CompanyNm VARCHAR(65),		 

		CheckNo CHAR(15),		

		PaymentDt DATETIME,  

		FxAmount MONEY,

		CurrencyCd CHAR(3),

		Amount MONEY,

		Seq Int

	)



	/* PRODUCE REPORT DETAIL  */	

	INSERT #Report

	SELECT 

		Counterparty.CompanyNm,		 

		Receipt.CheckNo,		

		Receipt.PaymentDt,

		SUM((-1) * Receipt.FxAmount) AS FxAmount,

		Receipt.CurrencyCd,

		SUM(Receipt.Amount) As Amount,

		1 

	FROM Receipt WITH(NOLOCK) 	

	LEFT JOIN Buysell WITH(NOLOCK) ON Receipt.buysellno = Buysell.Buysellno

	LEFT JOIN Counterparty WITH(NOLOCK) ON Buysell.CustID = Counterparty.CpID

	WHERE 	

		Receipt.PaymentDt >= @FromDt and Receipt.PaymentDt <= @ToDt 

		AND Receipt.CurrencyCd <> @SysCurrency 			

		AND (@CoID = '' OR @CoID = buysell.CoID) 

	GROUP BY Counterparty.CompanyNm,Receipt.CheckNo,Receipt.PaymentDt, Receipt.CurrencyCd

	HAVING SUM(Receipt.FXAmount) = 0

	ORDER BY Receipt.PaymentDt		





	INSERT #REPORT(CompanyNm,Amount,Seq)			

		SELECT 

			'TOTAL', 

			SUM(Amount),2

		FROM #Report 

	

	

	--SELECT * FROM #Report Order by Seq, PaymentDt

	   SELECT

	    CompanyNm [Vendor],		 

		CheckNo ,		

		PaymentDt ,  

		FxAmount ,

		CurrencyCd ,

		Amount 

		FROM #Report Order by Seq, PaymentDt
