

--12/13/2022 MEC: Converted PO_ID to allow "C" prefix for contract grades



CREATE proc [dbo].[cw_getIOGrades] @SONO varchar(15) = '', @IOType char(1) as

			SELECT 

				Grade.Description, 

				PODetail.GradeID,

				UnitType,

				Specifications,

				Price,

				PriceUOM, 

				CurrencyCd, 

				100 As Factor, 

				PODesc, 

				CONVERT(varchar(100), PO_ID) PO_ID,

				po.Terms,

				po.PriceBasis,

				Grade.InventoryType

			FROM PO WITH(NOLOCK) 

			

			inner join podetail on po.PONumber = podetail.PONumber

			left JOIN Grade WITH(NOLOCK) ON PODetail.GradeID = Grade.GradeID	

			WHERE PO.PONumber = @SONO

			

		SELECT

				Grade.Description, 

				Grade.GradeID							

			FROM PO WITH(NOLOCK) 

			inner join podetail on po.PONumber = podetail.PONumber

			LEFT JOIN Grade WITH(NOLOCK) ON PODetail.GradeID = Grade.GradeID	

			WHERE Grade.WhProductType = @IOType

			GROUP BY Grade.Description, Grade.GradeID
