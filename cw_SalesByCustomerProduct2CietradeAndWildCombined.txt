		   ,FreightCharge

		   ,OtherChargesAvg

		   ,[Cust Adj Avg]

		   ,[FOB Price]

		   , CASE WHEN TonsTotal = 0 then 0

		     else [AmountTotal]/TonsTotal 

			 end PricePerTon

			 ,a1.[Period]

			 ,SupplierID

			 ,[Datasource]

	SELECT

	RTRIM(COID) AS BillCompany

	, '' [FOB Price]

	,  SupplierID

	#detail

	#detail.COID,

	#detail.CompanyNm,



	 [dbo].[GetCietradeGradeName](#detail.[Grade]) 



	,#detail.Grade

	,#detail.WksType

	,#detail.TransactionPeriod

	,#detail.SupplierID

	,#detail.Delivery) a1

   (

   SELECT 

       #r23.Company

      ,#r23.[Index_Value]

	  ,#r23.[Period]

   ) a4

   ON TRIM(a4.[Company]) = TRIM(a1.BillCompany)

   AND TRIM(a4.[Period]) = TRIM(a1.[Period])

   AND TRIM(a4.[CieTradeGradeName]) = TRIM(a1.CieTradeGradeName)

SELECT [BillCompany]

      ,[CustomerID]

      ,[CustDescript]

      ,[CustGrade]

      ,[CieTrade Grade Name] [CieTradeGradeName]

	  ,ISNULL(a5.[Index], '') [Index]



      -- 06/20/24

      ,IIF(CONVERT(VARCHAR(6),[Index_Value]) = '0',' ', CONVERT(VARCHAR(6), a5.[Index_Value]))

	  --,ISNULL(a5.[Index_Value], 0) [Index_Value]



      ,[AmountTotal]

      ,[Delivery]

      ,[TonsTotal]

      ,CAST([FreightCharge] AS VARCHAR(30)) [FreightCharge]

      ,CAST([OtherChargesAvg] AS VARCHAR(30)) [OtherChargesAvg]

	  ,CAST([Cust Adj Avg] AS VARCHAR(30)) [Cust Adj Avg]

      ,CAST([FOB Price] AS VARCHAR(30)) [FOB Price]

      ,[PricePerTon]

      ,[tblWildCustomerspbi] .[Period]

      ,[SupplierID]

      ,CAST([Net Revenue] AS VARCHAR(30)) [Net Revenue] 

	  ,'Wild' [Datasource]

  FROM [WPSQL2K18].[WildRMR].[dbo].[tblWildCustomerspbi] 

  LEFT JOIN 

  (

  SELECT 

       #r23.Company

      ,#r23.[Index]

      ,#r23.[Index_Value]

      ,#r23.[Period]

	  ,a1.[CieTradeGradeName]

  FROM #r23

  LEFT JOIN 

  (SELECT [CieTradeGradeName]

      ,[Index]

  FROM [cdb_Wilmington].[dbo].[CietradeGradeNameIndexes]

	   ) a1

  on a1.[Index] = #r23.[Index]

  ) a5

  ON TRIM([tblWildCustomerspbi].[BillCompany]) = TRIM(a5.[Company])

  AND TRIM([tblWildCustomerspbi].[Period]) = TRIM(a5.[Period])

  AND TRIM([tblWildCustomerspbi].[CieTrade Grade Name]) = TRIM(a5.[CieTradeGradeName])



 update [cdb_Wilmington].[dbo].[tblCietradeAndWildCustomerCombined]

 set [index] = ISNULL(#r25.[Index], '') , index_Value = ISNULL(#r25.[Index_Value], 0) 

 from tblCietradeAndWildCustomerCombined

 JOIN #r25 on

 tblCietradeAndWildCustomerCombined.BillCompany = #r25.[Company] and 

 tblCietradeAndWildCustomerCombined.[Period] = #r25.[Period] and

 tblCietradeAndWildCustomerCombined.CietradeGradeName = #r25.CietradeGradeName
