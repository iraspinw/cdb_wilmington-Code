











CREATE  PROCEDURE [dbo].[cw_GetFreightRate_Ocean_TEST]  @CpID char(12), @Expiration datetime = '1/1/1900',

									 @ExpirationType int = 0, @POL varchar(50) = '', @POD varchar(50) = '', 

									 @POR varchar(50) = '', @PlaceOfDeliv varchar(50) = '',

									 @ExpFromDt datetime = '', @ExpToDt datetime = '',

									 @FreightFwdr char(12) = '',@PerUOM varchar(15)='',

									 @POLCountryID char(3) = '',@PODCountryID char(3) = '', @RateID int = 0, @FreightExpID int = 0







AS

SET NOCOUNT ON

--08/27/2009 RS: Created from [q_GetFreightRate] for Ocean Freight

declare @CountNewImport int 



select @CountNewImport = count(*) from FreightRate_Ocean where NewImport =1



	SELECT RateID, FreightRate_Ocean.CarrierID,

	 Counterparty.CompanyNm as [Ocean Carrier],

	 RateDate as [Date], 

	 --convert(char(10),RateDate,101) as [Date], 

	       POL as [POL],

		   POD  as [POD],

		   PlOfRcpt  as [PlOfRcpt],

		   PlOfDeliv  as [PlOfDeliv],

		     (isnull(POL,'') + case when isnull(CPol.Name,'') <>'' then  ', ' +  isnull(CPol.Name,'') else '' end) as [Port of Loading],

		     (isnull(POD,'') + case when isnull(Cpod.Name,'') <>'' then  ', ' +    isnull(Cpod.Name,'')else '' end) as [Port of Discharge],

		     (isnull(PlOfRcpt,'') + case when isnull(CPlOfRcpt.Name,'') <>'' then ', ' +  isnull(CPlOfRcpt.Name,'')else '' end) as [Place of Receipt],

		     (isnull(PlOfDeliv,'') + case when isnull(CPlOfDeliv.Name,'') <>'' then ', ' +    isnull(CPlOfDeliv.Name,'')else '' end) as [Place of Delivery],

		   ExpiryDate as [Expiry Date],

		    Notes,

			 Charge as [{C}Charge],

			 ChargeCurCode,ReferenceNo as [Ref No],

		   FreightFwdr,

		  -- dbo.CnvUOM_Names(PerUOM) as [Per /],

		  case when PerUOM ='' then 'FLAT' when PerUOM ='M'  then 'MT' when PerUOM ='S' then 'ST' else 'CONTAINER' end as [Per /],

		  PerUOM as PerValue,

		   FF.CompanyNm as   [Freight Forwarder],

		      FF.CompanyNm as   [-hidden-Freight Forwarder],

		   EqType as [Equip. Type],

		   PortFreetime as [Port Freetime],

		   CPol.Name as [POL Country],

		   Cpod.Name as [POD Country],

		   TransitTime as [Transit Time],

		   FreightRate_Ocean.StickyNotes,

		   (select count(*) from BookSheet where BookSheet.FreightRateID=FreightRate_Ocean.RateID) as '-hidden-NoAssigned'

    FROM FreightRate_Ocean WITH(NOLOCK)

	INNER JOIN Counterparty WITH(NOLOCK) ON FreightRate_Ocean.CarrierID = Counterparty.CpID

	LEFT JOIN Counterparty FF WITH(NOLOCK) ON FreightRate_Ocean.FreightFwdr = FF.CpID

	LEFT JOIN Ports POL WITH(NOLOCK) ON POL.PortNm = FreightRate_Ocean.POL

	LEFT JOIN Ports POD WITH(NOLOCK) ON POD.PortNm = FreightRate_Ocean.POD

	LEFT JOIN Ports PlOfRcpt WITH(NOLOCK) ON PlOfRcpt.PortNm = FreightRate_Ocean.PlOfRcpt

	LEFT JOIN Ports PlOfDeliv WITH(NOLOCK) ON PlOfDeliv.PortNm = FreightRate_Ocean.PlOfDeliv

	left join Country CPol on CPol.CountryID = POL.CountryID

	left join Country Cpod on Cpod.CountryID = POD.CountryID

	left join Country CPlOfRcpt on CPlOfRcpt.CountryID = PlOfRcpt.CountryID

		left join Country CPlOfDeliv on CPlOfDeliv.CountryID = PlOfDeliv.CountryID



	WHERE (

			(@RateID <> 0 and RateID = @RateID) OR ((@RateID = 0 and

			(@CpID = '' OR @CpID = FreightRate_Ocean.CarrierID)

			AND (@POL = '' OR FreightRate_Ocean.POL = @POL)

			AND (@POD = '' OR FreightRate_Ocean.POD = @POD)

			AND (@POR = '' OR FreightRate_Ocean.PlOfRcpt = @POR)

			AND (@PlaceOfDeliv = '' OR FreightRate_Ocean.PlOfDeliv = @PlaceOfDeliv)

			AND (@FreightFwdr = '' OR (@FreightFwdr = '(None)' AND RTRIM(ISNULL(FreightRate_Ocean.FreightFwdr,'')) = '') OR FreightRate_Ocean.FreightFwdr = @FreightFwdr)

			AND (@POLCountryID = '' OR POL.CountryID = @POLCountryID)

			AND (@PODCountryID = '' OR POD.CountryID = @PODCountryID)

			--EXCLUDE EXPIRED

			AND (@Expiration = '1/1/1900' 

				OR ((@ExpirationType = 0 

					AND 

					(FreightRate_Ocean.ExpiryDate >= @Expiration 

						OR FreightRate_Ocean.ExpiryDate = '' 

						OR FreightRate_Ocean.ExpiryDate is null)) 

				OR (@ExpirationType = 1 AND FreightRate_Ocean.ExpiryDate < @Expiration)))

			AND (@ExpFromDt = '' 

				OR ((FreightRate_Ocean.RateDate >= @ExpFromDt AND FreightRate_Ocean.RateDate <= @ExpToDt) 

					OR FreightRate_Ocean.RateDate = '' OR FreightRate_Ocean.RateDate is null))))

			--	OR ((FreightRate_Ocean.ExpiryDate >= @ExpFromDt AND FreightRate_Ocean.ExpiryDate <= @ExpToDt) 

					--OR FreightRate_Ocean.ExpiryDate = '' OR FreightRate_Ocean.ExpiryDate is null))))

		and @CountNewImport =0)

		or

	   (NewImport=1)





	update FreightRate_Ocean

	set NewImport = 0





    SELECT e.RateID, VendorID, c.CompanyNm, GL.GLAcct, GL.Name, e.[Description], isnull(Charge,0) as Charge, CurrencyCd, UOM, e.[RateExpenseID]

    FROM OceanFreightRate_Expense e WITH(NOLOCK)

    INNER JOIN Counterparty c WITH(NOLOCK) ON e.VendorID = c.CpID

    left JOIN GLAccount GL WITH(NOLOCK) ON e.GLAcct = GL.GLAcct

	where e.RateID =@RateID

		And (@FreightExpID = 0 Or e.[RateExpenseID] = @FreightExpID)
