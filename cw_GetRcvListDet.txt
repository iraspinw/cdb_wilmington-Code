	Declare @ShowInUomRnd int

	Declare @ShowUOM char(4)



	

	-- SET REPORT UNIT OF MEASURE CODE Lbs/Kgs.

	SELECT TOP 1 @ShowUOM = SWeightUOM FROM WksDetail WITH(NOLOCK) WHERE BuySellNo = @BuySellNo



	

	SET @ShowInUomRnd = 3

	IF @ShowUOM = 'L' OR @ShowUOM = 'K'

		SET @ShowInUomRnd = 0





	SELECT 

		WksDetail.LotNo,

		WksDetail.Units, 

		RTRIM(ISNULL(dbo.CnvUnitName(WksDetail.UnitType),'')) as UnitType, 

		dbo.FormatQty(dbo.CONVERT_TO_UOM(WksDetail.PWeight, WksDetail.PWeightUOM, @ShowUOM), @ShowUOM) AS WtInShowUOM,

		dbo.FormatQty(dbo.CONVERT_TO_UOM(WksDetail.TareWt, WksDetail.PWeightUOM, @ShowUOM), @ShowUOM) AS TareWtInShowUOM,

		dbo.FormatQty(dbo.CONVERT_TO_UOM(WksDetail.GrossWt, WksDetail.PWeightUOM, @ShowUOM), @ShowUOM) AS GrossWtInShowUOM,

		RTRIM(dbo.CnvUOM_Names(@ShowUOM)) AS WtShowUOM,

		Grade.Description as Product,

		WksDetail.InvoiceDesc as InvoiceDesc, 

		WksDetail.Specifications,

		RTRIM(WksDetail.SerialNo) AS [SerialNo],

		dbo.RemoveTrailingZeros(WksDetail.PPrice) as Price,

		RTRIM(dbo.CnvUOM_Names(WksDetail.PUOM)) as PriceUOM,

		WksDetail.PFxAmount as FxAmount,WksDetail.PCurrencyCd as CurrencyCd,WksDetail.PAmount as Amount,

		WarehouseLocation.LocationName

	FROM WksDetail WITH (NOLOCK) 

	INNER JOIN Grade WITH (NOLOCK) ON WksDetail.GradeID = Grade.GradeID

	INNER JOIN BuySell WITH (NOLOCK) ON WksDetail.BuySellNo = BuySell.BuySellNo 

	left join WarehouseLocation on WarehouseLocation.ID = WksDetail.WarehouseLocation

	WHERE WksDetail.BuySellNo = @BuySellNo

	--AND BuySell.Status <> 'I'

	ORDER BY LotNo
