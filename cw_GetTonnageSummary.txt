		INSERT #wrkReport

			dbo.convert_to_uom(PWeight,PWeightUOM,@ReportUOM), --PWeight, PWeightUOM,

			WksDetail.PAmount,

			BuySell.BuySellNo			

		FROM WksDetail 

		INNER JOIN BuySell ON WksDetail.BuySellNo = BuySell.BuySellNo

		INNER JOIN GRADE ON WksDetail.GradeID = Grade.GradeID

		WHERE (@StartDt = '' OR ShippingDt >= @StartDt)

		AND (@EndDt = '' OR ShippingDt <= @EndDt) 

		AND (BuySell.[Status] <> 'X' AND BuySell.Status <> 'W')

		AND PWeight <> 0 

		AND (@DEPTID = '' OR BuySell.COID = @DEPTID)

		AND (@TradeType = '' OR BuySell.TradeType= @TradeType)

		AND (@WarehouseNm = '' OR BUYSELL.ShipToID = @WarehouseNm)

		AND (@ProductCategoryID = '' OR Grade.ProductCategoryID = @ProductCategoryID)

		AND ISNULL(BuySell.IsAgency,0) <> 1

		AND BuySell.VendorID <> '(INV)'



	END

		AND (@TradeType = '' OR BuySell.TradeType= @TradeType)

		AND (@WarehouseNm = '' OR BUYSELL.ShipToID = @WarehouseNm)

		AND (@ProductCategoryID = '' OR Grade.ProductCategoryID = @ProductCategoryID)

		AND ISNULL(BuySell.IsAgency,0) <> 1

		AND BuySell.CustID <> '(INV)'



	END





	IF @GroupBy = 'GRADE+PKG'

	INSERT #REPORT (GradeNm,Quantity,Cost,AvgCost,ShipmentCnt)

	SELECT GradeNm, SUM(Quantity), SUM(Cost),0,0 FROM #wrkReport GROUP BY GradeNm ORDER BY GradeNm

		



	--Shipments

	Select GradeNm, BuySellNo, Count(*) Cnt INTO #wrkShipments FROM #wrkReport GROUP BY GradeNm, BuySellNo

	select GradeNm, Count(*) Shipments INTO #Shipments FROM #wrkShipments  Group by GradeNm

	UPDATE #REPORT SET ShipmentCnt = Shipments FROM #Shipments S INNER JOIN #REPORT R ON S.GradeNm = R.GradeNm



		

	IF @ReportUOM = 'L' 

		UPDATE #REPORT  SET AvgCost = Cost/(Quantity/2000) WHERE Quantity <> 0

	IF @ReportUOM = 'K' 

		UPDATE #REPORT  SET AvgCost = Cost/(Quantity/1000) WHERE Quantity <> 0

	IF @ReportUOM <> 'L' AND @ReportUOM <> 'K'

		UPDATE #REPORT  SET AvgCost = Cost/Quantity WHERE Quantity <> 0







	-- FINAL REPORT
