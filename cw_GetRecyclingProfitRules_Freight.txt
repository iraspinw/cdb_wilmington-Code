CREATE Procedure [dbo].[cw_GetRecyclingProfitRules_Freight] AS 

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

--11/8/2022 ZT: Created from [q_GetRecyclingProfitRules_Freight]

--11/9/2022 ZT: Added  EquipmentType name



SELECT [ID]

      ,[WarehouseID]

      , CASE WHEN ISNULL([VendorID], '') = '' THEN '(Any)' ELSE [CompanyNm] END Carrier

      ,[VendorID]

      ,CASE WHEN ISNULL(EquipmentType, '') = '' THEN '(Any)' ELSE  (SELECT TOP 1 EqType FROM EquipmentType WHERE ShortName =  [RecyProfitRules_Freight].EquipmentType ) END Equiptment

      ,[Charge]	[Freight/Hour]

      ,[DefaultTime]   [Default Run Time]   

  FROM [RecyProfitRules_Freight] With(NOLOCK) 

  LEFT JOIN COUNTERPARTY ON VendorID = CPID
