CREATE   PROCEDURE [dbo].[cw_GetPort] @Name varchar(25)

AS

BEGIN

--01/28/2016 MCM: Created

--01/29/2016 MCM: Added email

	SET NOCOUNT ON;

	SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;



    SELECT [PortNm],ISNULL(CountryID,'') CountryID,[Ports].PortID

		FROM [Ports] WHERE [PortNm] = @Name

END
