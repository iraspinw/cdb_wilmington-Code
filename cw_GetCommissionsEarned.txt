	--@UnpostedOnly bit = 0, @Dept char(2) = '', @TradeType char(25) = '', @SalesRep CHAR(10) = ''

SET NOCOUNT ON

SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED

	CREATE TABLE #ReportDetail(

		SeqNo INT IDENTITY(1,1),

		RecType CHAR(1),

		InvoiceDt DATE,

		LoadNo CHAR(12),

		SoldToCustomer VARCHAR(100),

		ShipToLocation  VARCHAR(100),

		CustomerPO CHAR(35),

		ProductNm VARCHAR(200), 

		SalesPrice VARCHAR(50),		

		SalesExtension MONEY,

		QuantityWt DECIMAL(22,3),		

		CommissionRate VARCHAR(50),

		CommissionAmt MONEY,		

		TotalQty DECIMAL(22,3),		

		AllocQty DECIMAL(22,3),	

		AllocWtQty DECIMAL(22,3),		

		AllocEaQty DECIMAL(22,3),

		AllocType CHAR(1),

		WksDetailID INT,

		WksDetailSeq INT,

		WksType char(1)

	)

	--------------------------------

	INSERT #ReportDetail(RecType,InvoiceDt,LoadNo,SoldToCustomer,ShipToLocation,CustomerPO,ProductNm,SalesPrice,SalesExtension,QuantityWt,CommissionRate,CommissionAmt,TotalQty,AllocQty,AllocWtQty,AllocEaQty,AllocType,WksDetailID,WksDetailSeq,WksType) 		

	SELECT 

		'B',

		BuySell.InvoiceDt,

		BuySell.BuySellNo,

		Customer.CompanyNm,																			--Customer

		A.City+' '+A.Region,

		ISNULL(PO.POReference,''),																	--Customer Order

		Grade.[Description]+' '+ISNULL(dbo.CnvUnitName(WksDetail.UnitType),''), 

		CONVERT(VARCHAR(20),CONVERT(FLOAT,WksDetail.SPrice))+' /'+ rtrim(dbo.CnvUOM_Names2(WksDetail.SPriceUOM))+' '+BuySell.CurrencyCd,		

		WksDetail.SAmount,

		dbo.convert_to_Uom(SWeight,SWeightUOM,@UOM),												--Weight

		Convert(VARCHAR(20),CONVERT(FLOAT,wksDelivery.Price)) + ' '+rtrim(dbo.CnvUOM_Names2(wksDelivery.PricePer)),		

		WksDelivery.FxAmount,

		0,

		0,

		(CASE WHEN SWeightUom <> 'E' THEN dbo.convert_to_Uom(SWeight,SWeightUOM,'L') ELSE 0 END),

		(CASE WHEN SWeightUom = 'E' THEN SWeight ELSE 0 END),

		'W',

		WksDetail.DetailID,

		ROW_NUMBER()  OVER(PARTITION BY WksDetail.DetailID ORDER BY WksDetail.DetailID ASC),

		CASE

			WHEN buysell.CustID = '(INV)' THEN 'R'

			WHEN buysell.VendorID = '(INV)' THEN 'S'

			ELSE 'B'

		END as WksType

		FROM WksDelivery 

			INNER JOIN Buysell WITH(NOLOCK) ON BuySell.BuySellNo = wksDelivery.BuySellNo

			INNER JOIN GLAccount WITH(NOLOCK) ON GLAccount.GLAcct = wksDelivery.GLAcct 						

			INNER JOIN Counterparty Customer WITH(NOLOCK) ON BuySell.CustID = Customer.CpID

			INNER JOIN [Address] A ON A.CPID = BuySell.CustID AND A.[Type] = BuySell.ShipToID

			LEFT JOIN PO WITH(NOLOCK) ON BuySell.RefSoNo = PO.PONUmber	

			LEFT JOIN WksDetail WITH(NOLOCK) ON WksDetail.BuySellNo = BuySell.BuySellNo

			LEFT JOIN Grade WITH(NOLOCK) ON Grade.GradeID = wksDetail.GradeID

		WHERE WksDelivery.VendorID = @VendorID

			AND BuySell.[STATUS] = 'I'

			AND GLAccount.Cat = 'COMMISSION'

			AND (@FromDt = '' OR InvoiceDt >= @FromDt)

			AND (@ToDt = '' OR InvoiceDt <= @ToDt)





	UPDATE #ReportDetail SET ProductNm = '', SalesPrice = 'na', SalesExtension = 0 WHERE ProductNm Is NULL





--delete from #ReportDetail  where loadno <> '13276'





	--ASSIGN QUANTITY TYPE		

	UPDATE #ReportDetail SET AllocType = 'E' WHERE EXISTS(Select * from Wksdetail where WksDetail.BuysellNo = #ReportDetail.LoadNo AND SWeightUOM = 'E')

			



	

	

	-- ALLOCATE COMMMISSION EXPENSES BY GRADE --

	--------------------------------------------------------------------------------------------------------



	-- GET LIST OF WORKSHEETS 

	CREATE TABLE #Loads(LoadNo CHAR(12))

	INSERT #Loads

		SELECT DISTINCT LoadNo FROM #ReportDetail

	

	UPDATE #ReportDetail SET AllocQty = AllocEaQty WHERE AllocEaQty <> 0  AND  AllocType = 'E'

	UPDATE #ReportDetail SET AllocQty = AllocWtQty WHERE AllocWtQty  <> 0  AND  AllocType = 'W'



	

	---- GET TOTAL WEIGHT OF LOADS.

	CREATE TABLE #TotalWeight(LoadNo CHAR(12), TotalQty DECIMAL(22,5))

	INSERT #TotalWeight

		SELECT DISTINCT LoadNo, SUM(AllocQTy) FROM #ReportDetail WHERE WksDetailSeq=1 GROUP BY LoadNo





	UPDATE #ReportDetail SET TotalQty = #TotalWeight.TotalQty  FROM #ReportDetail INNER JOIN #TotalWeight ON #ReportDetail.LoadNo = #TotalWeight.LoadNo	

	UPDATE #ReportDetail SET CommissionAmt = Round(CommissionAmt * (AllocQty / TotalQty),2) WHERE TotalQty <> 0 



	

	--------------------------------------------------------------------------------------------------------





	-- COMMISSION ADJUSTMENTS

	-------------------------

	INSERT #ReportDetail(RecType,InvoiceDt,LoadNo,SoldToCustomer,ShipToLocation,CustomerPO,ProductNm,SalesPrice,SalesExtension,QuantityWt,CommissionRate,CommissionAmt,TotalQty,AllocQty,WksType) 		

	SELECT 

		'A',

		CASE WHEN GPLedger.AdjPosted = 'N' THEN BuySell.InvoiceDt ELSE GPLedger.TradeDt END,

		GPLedger.BuySellNo,

		Customer.CompanyNm,	

		A.City+' '+A.Region,

		ISNULL(PO.POReference,''),																	--Customer Order

		'Commission Adjustment',

		' ',

		0,

		0,

		' ',

		GPLedger.AdjFxAmount,

		NULL,NULL,

		CASE

			WHEN buysell.CustID = '(INV)' THEN 'R'

			WHEN buysell.VendorID = '(INV)' THEN 'S'

			ELSE 'B'

		END as WksType

	FROM GPLedger

			INNER JOIN Buysell WITH(NOLOCK) ON BuySell.BuySellNo = GPLedger.BuySellNo

			INNER JOIN GLAccount WITH(NOLOCK) ON GLAccount.GLAcct = GPLedger.GLAcct 

			INNER JOIN Counterparty Customer WITH(NOLOCK) ON BuySell.CustID = Customer.CpID

			INNER JOIN [Address] A ON A.CPID = BuySell.CustID AND A.[Type] = BuySell.ShipToID

			LEFT JOIN PO WITH(NOLOCK) ON BuySell.RefSONo = PO.PONUmber	

		WHERE GPLedger.AdjCPID = @VendorID

			AND 

			(

				(GPLedger.AdjPosted = 'Y' AND ((@FromDt = '' OR TradeDt >= @FromDt) and (@ToDt = '' OR TradeDt <= @ToDt)))

				OR

				(GPLedger.AdjPosted = 'N' AND  ((@FromDt = '' OR InvoiceDt >= @FromDt) and (@ToDt = '' OR InvoiceDt <= @ToDt)))

			)

			AND BuySell.[STATUS] = 'I'

			AND GPLedger.Source = 'A'

			AND (GLAccount.Cat = 'COMMISSION')

		





	-- FINAL REPORT

	---------------

	SELECT	InvoiceDt [Invoice Date],

			LoadNo [Load#],

			SoldToCustomer [Customer],

			ShipToLocation [Ship To Location],

			CustomerPO [PO#], 

			ProductNm [Product/Description],

			SalesPrice [Sales Price],

			SalesExtension [Extension],

			QuantityWt [Quantity],

			CommissionRate [Comm Rate],

			CommissionAmt [Comm Amount],

			WksType [-hidden-WksType]	

		FROM #ReportDetail

		Order By InvoiceDt, LoadNo
