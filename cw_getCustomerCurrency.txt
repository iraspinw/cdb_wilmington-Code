CREATE proc [dbo].[cw_getCustomerCurrency] @CPID varchar(15), @Date datetime as 

set transaction isolation level read uncommitted

select top 1 currcode, isnull(FxRate,1) as fxRate, GLAcctSupplier

from counterparty

left join CurrencyRate on CurrencyRate.CurCode = counterparty.CurrCode and exchdate <= @Date

where cpid = @Cpid 



order by ExchDate desc
