declare @TrackBySupplier bit 

	select @TrackBySupplier = isnull(optionvalue,0) from options where optionname='TrackBySupplier'



if @TrackBySupplier = 0 

	set @Supplier = ''



--select sum(quantity), sum(value), case when sum(value) <> 0 then sum(quantity) / sum(value) else 0 end

--from InventoryBalance

declare @BalQuantity decimal(22,5) = 0

declare @BalValue money = 0

declare @BalUnitCost decimal(22,5) = 0



	exec cw_GetBulkQuantityValue @Warehouse, @GradeID, @Class, @Location, @Supplier, @BalQuantity output, @BalValue output
