	----Accounts Receivable.	

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,HCAmount,ActionTag) 

    SELECT 

		1,'AR','SALES',1,			

		COALESCE(BuySell.TradeType,''),			

		@ARAccrualAC, --@ARAcct, 

		'Accrued Sales', 

		SUM(round(SalesAmt+SFreightChg+COALESCE(SalesTaxAmt,0),2)),'TOTAL' 		

    FROM BuySell WITH(NOLOCK) 

    WHERE InvoiceDt >= @dt1 and InvoiceDt <= @dt2 

		AND (CustID <> '(INV)'  AND CustID <> '{C}' AND CustID <> '(PLANT)')  

		AND Status = 'I'  and (IsAgency IS NUlL OR IsAgency <> 1)

		AND (  (@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )
	Group By  BuySell.TradeType

	order by TradeType	

	

	

	--Product Sales.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount) 

	SELECT 

		1,'AR','SALES',2,			

		COALESCE(BuySell.TradeType,''),			

		COALESCE(GLAccount.GLAcct,@SalesAcct),   

		COALESCE(GLAccount.Name,'Sales (Default)'), 

		BuySell.BuySellNo,

		BuySell.InvoiceDt,

		Round(SAmount*-1,2)		

	FROM BuySell WITH(NOLOCK) 

		INNER JOIN wksDetail WITH(NOLOCK) ON BuySell.BuySellNo = wksDetail.BuySellNo		

		INNER JOIN Grade WITH(NOLOCK) ON Grade.GradeID = wksDetail.GradeID 

		LEFT JOIN GLAccount WITH(NOLOCK) ON GLAccount.GLAcct = Grade.GLAcct

	WHERE BuySell.InvoiceDt >= @dt1 and BuySell.InvoiceDt <= @dt2 and (BuySell.CustID <> '(INV)' AND BuySell.CustID <> '{C}'  AND CustID <> '(PLANT)')

		And Status = 'I'  and (IsAgency IS NUlL OR IsAgency <> 1)

		AND (  (@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )



	

	--Sales Chargebacks.

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,AltAcctNm,HCAmount,IsOffsetTx,ExpItemID) 

    SELECT 

		1,'AR','SALES',2,  --3	

		COALESCE(BuySell.TradeType,''),	

		COALESCE(SalesGL.GLAcct,@SalesAcct),

		ISNULL('Sales-' + GLAccount.Name,'(SALES-CHGBACK)'),		

		BuySell.BuySellNo,

		BuySell.InvoiceDt,

		SalesGL.Name,		

		round(ChgToCustAmt*-1,2),

		CASE WHEN UPPER(BuySell.VendorID) <> '(OTHER)' THEN 0 ELSE  -1 END IsOffsetTx,

		WksDelivery.ItemID

    FROM WksDelivery WITH(NOLOCK) 

		INNER JOIN BuySell WITH(NOLOCK) On WKSDelivery.BuySellNo = BuySell.BuySellNo

        left join glaccount  WITH(NOLOCK) ON WksDelivery.glacct = glaccount.glacct

        left join glaccount SalesGL WITH(NOLOCK) ON GLAccount.Salesglacct = SalesGL.glacct

    WHERE InvoiceDt >= @dt1 and InvoiceDt <= @dt2 

        AND (CustID <> '(INV)'  AND CustID <> '{C}'  AND CustID <> '(PLANT)')  

        AND Status = 'I'  

		and (IsAgency IS NUlL OR IsAgency <> 1)

        AND ChgToCustAmt <> 0

        AND ((@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')))

	INSERT #LedgerDetail(SectionSeq,AcctClass,AcctGrp,AcctGrpOrd,TradeType,AcctNo,AcctNm,TranNo,TranDt,HCAmount)

        COALESCE(BuySell.TradeType,''),	

		SalesTax.SalesTaxAcct,								

		'Sales-Tax '+SalesTax.TaxDescription,   

		BuySell.BuySellNo,

		BuySell.InvoiceDt,		            

        (COALESCE(TaxAmount,0)*-1)                  

    FROM BuySell WITH(NOLOCK)

    LEFT JOIN wksSalesTax WITH(NOLOCK)  ON wksSalesTax.BuySellNo = BuySell.BuySellNo

    LEFT JOIN SalesTax WITH(NOLOCK) ON SalesTax.SalesTaxID = wksSalesTax.SalesTaxID		    

    WHERE 

    BuySell.InvoiceDt >= @dt1 and BuySell.InvoiceDt <= @dt2 

    and (BuySell.CustID <> '(INV)' AND BuySell.CustID <> '{C}'  AND CustID <> '(PLANT)')

    AND wksSalesTax.Source = 'B'

    And Status = 'I' 

    and COALESCE(IsAgency,0) <> 1

    AND SalesTaxAmt <> 0 

    AND (  (@CoID = '')  OR (BuySell.CoID = @CoID OR (BuySell.CoID='' AND @CoID = '00')) )

		20,'ALERTS','',8,'','',

		' Inventory Write-off',	

		SUM(cost),

		'ALERT-WRITEOFF'

        from gpledger WITH(NOLOCK) 

        where TradeDt >= @dt1 and TradeDt <= @dt2

        and source = 'I' 



	

	DELETE FROM #LedgerDetail WHERE AcctClass = 'ALERTS' AND HCAmount IS NULL	

	UPDATE #LedgerDetail SET DebitAmt = abs(HCAmount), CreditAmt =0, HCAmount=ABS(HCAmount) WHERE AcctClass = 'ALERTS'







	/* ASSIGN DEBIT/CREDIT COLUMNS */

	---------------------------------



	--1. Assign basic debit and credit values.

    UPDATE #LedgerDetail SET DebitAmt = abs(HCAmount), CreditAmt = 0  WHERE HcAmount > 0 

    UPDATE #LedgerDetail SET CreditAmt = abs(HCAmount), DebitAmt = 0  WHERE HcAmount <= 0 



	--2. For Sales: Assign +Debits --> -Credits.

	UPDATE #LedgerDetail SET DebitAmt = 0, CreditAmt = (HCAmount*-1) WHERE AcctGrp = 'SALES' AND COALESCE(ISTotalLn,0) <> 1 AND COALESCE(ActionTag,'') <> 'TOTAL' AND HCAmount > 0 



	--3. For Posted Sales Adjustments: Assign -Credits --> +Debits.

	UPDATE #LedgerDetail SET DebitAmt = (HCAmount), CreditAmt = 0   WHERE AcctClass = 'POSTED-SALESADJ' AND COALESCE(ActionTag,'') <> 'TOTAL' AND HCAmount < 0 



    -- 4.  For Expenses and taxes: Assign -credits --> +Debits.  

	UPDATE #LedgerDetail SET DebitAmt = HCAmount, CreditAmt = 0  WHERE ( AcctClass = 'EXPENSES' OR  AcctGrp = 'TAXES-PAID') AND AcctGrpOrd=1 AND COALESCE(ActionTag,'') <> 'TOTAL' and HCAmount < 0 

   

	-- 5. For Inventory

	--Get Inventory Balance

	DECLARE @InvBal Money

	SELECT @InvBal  = HcAmount FROM #LedgerDetail WHERE AcctCLass = 'INVENTORY' AND COALESCE(ActionTag,'') = 'TOTAL'  

	IF @InvBal <= 0 --Credit 	

		UPDATE #LedgerDetail SET DebitAmt = HCAmount, CreditAmt = 0  WHERE AcctClass = 'INVENTORY' AND AcctGrpOrd=1 AND COALESCE(ActionTag,'') <> 'TOTAL' and HCAmount < 0 	

	IF @InvBal > 0 --Debit

		UPDATE #LedgerDetail SET DebitAmt = 0, CreditAmt = HCAmount*-1  WHERE AcctClass = 'INVENTORY' AND AcctGrpOrd=1 AND COALESCE(ActionTag,'') <> 'TOTAL' and HCAmount > 0 	





	--6. For Inventory: Swap Debit & Credit columns for negative values. 

	SELECT AcctNm, TRadeType, sum(DebitAmt) DB, sum(CreditAmt) CR into #_fixDBCr from #LedgerDetail 

		WHERE AcctCLass  = 'INVENTORY' AND  coalesce(ActionTag,'') <> 'TOTAL' AND (DebitAmt <> 0 or CreditAmt <> 0 )

		GROUP BY AcctNm, TRadeType

	

	UPDATE #LedgerDetail SET DebitAmt = CreditAmt * -1, CreditAmt = 0 FROM 

		#LedgerDetail INNER JOIN #_fixDBCr ON #_fixDBCr.AcctNm = #LedgerDetail.AcctNm AND #_fixDBCr.TRadeType = #LedgerDetail.TRadeType

			WHERE #_fixDBCr.CR < 0 

		

	UPDATE #LedgerDetail SET CreditAmt  = DebitAmt * -1, DebitAmt = 0 FROM 

		#LedgerDetail INNER JOIN #_fixDBCr ON #_fixDBCr.AcctNm = #LedgerDetail.AcctNm AND #_fixDBCr.TRadeType = #LedgerDetail.TRadeType

			WHERE #_fixDBCr.DB < 0 

	

	/* COPY DEBIT ON TOTAL LINE */

	UPDATE #LedgerDetail SET CreditAmt = HCAmount, DebitAmt= HCAmount WHERE ISTotalLn = 1 AND HCAmount <> 0

		SET @pAcctNo= dbo.GetField(@DetailReportTag,5,'|')				

		SET @pTradeType= dbo.GetField(@DetailReportTag,6,'|')



		--select @DetailReportTag			

		--select @pAcctClass, @pAcctGrp, @pAcctNm, @pAcctNo, @pTradeType

		

		INSERT #LedgerDrillDown

		SELECT 

			TranNo [Worksheet No],

			TranDt [Posting Date],

			AcctNm [Description],

			AcctNo [Acct No],

			HCAmount [Amount]

		FROM #LedgerDetail 

		WHERE AcctClass = @pAcctClass 

		AND AcctGrp = @pAcctGrp

		AND AcctNo = @pAcctNo			

		AND AcctNm = @pAcctNm

		--AND (@pAcctClass = 'INVENTORY' OR AcctNm = @pAcctNm)

		AND TradeType = @pTradeType

		AND HCAmount <> 0 

		ORDER BY TranDt



		INSERT #LedgerDrillDown

			SELECT 'TOTAL',NULL,NULL,NULL,SUM(HCAmount)

				FROM #LedgerDrillDown

	

		SELECT 			TransNo [Worksheet No],

			TransDt [Posting Date],

			AcctNm [Description],

			AcctNo [Acct No],

			HCAmount [Amount] FROM #LedgerDrillDown
